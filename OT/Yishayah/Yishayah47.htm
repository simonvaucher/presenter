לפרק הקודם * * * תוכן העניינים

ישעיהו פרק מז

א רְדִי וּשְׁבִי עַל-עָפָר בְּתוּלַת בַּת-בָּבֶל שְׁבִי-לָאָרֶץ אֵין-כִּסֵּא בַּת-כַּשְׂדִּים  כִּי לֹא תוֹסִיפִי יִקְרְאוּ-לָךְ רַכָּה וַעֲנֻגָּה׃    ב קְחִי רֵחַיִם וְטַחֲנִי קָמַח גַּלִּי צַמָּתֵךְ חֶשְׂפִּי-שֹׁבֶל גַּלִּי-שׁוֹק עִבְרִי נְהָרוֹת׃    ג תִּגָּל עֶרְוָתֵךְ גַּם תֵּרָאֶה חֶרְפָּתֵךְ נָקָם אֶקָּח וְלֹא אֶפְגַּע אָדָם׃   פ

ד גֹּאֲלֵנוּ יְהוָה צְבָאוֹת שְׁמוֹ קְדוֹשׁ יִשְׂרָאֵל׃    ה שְׁבִי דוּמָם וּבֹאִי בַחֹשֶׁךְ בַּת-כַּשְׂדִּים  כִּי לֹא תוֹסִיפִי יִקְרְאוּ-לָךְ גְּבֶרֶת מַמְלָכוֹת׃    ו קָצַפְתִּי עַל-עַמִּי חִלַּלְתִּי נַחֲלָתִי וָאֶתְּנֵם בְּיָדֵךְ לֹא-שַׂמְתְּ לָהֶם רַחֲמִים עַל-זָקֵן הִכְבַּדְתְּ עֻלֵּךְ מְאֹד׃    ז וַתֹּאמְרִי לְעוֹלָם אֶהְיֶה גְבָרֶת עַד לֹא-שַׂמְתְּ אֵלֶּה עַל-לִבֵּךְ לֹא זָכַרְתְּ אַחֲרִיתָהּ׃   פ

ח וְעַתָּה שִׁמְעִי-זֹאת עֲדִינָה הַיּוֹשֶׁבֶת לָבֶטַח הָאֹמְרָה בִּלְבָבָהּ אֲנִי וְאַפְסִי עוֹד לֹא אֵשֵׁב אַלְמָנָה וְלֹא אֵדַע שְׁכוֹל׃    ט וְתָבֹאנָה לָּךְ שְׁתֵּי-אֵלֶּה רֶגַע בְּיוֹם אֶחָד שְׁכוֹל וְאַלְמֹן כְּתֻמָּם בָּאוּ עָלַיִךְ בְּרֹב כְּשָׁפַיִךְ בְּעָצְמַת חֲבָרַיִךְ מְאֹד׃    י וַתִּבְטְחִי בְרָעָתֵךְ אָמַרְתְּ אֵין רֹאָנִי חָכְמָתֵךְ וְדַעְתֵּךְ הִיא שׁוֹבְבָתֶךְ וַתֹּאמְרִי בְלִבֵּךְ אֲנִי וְאַפְסִי עוֹד׃    יא וּבָא עָלַיִךְ רָעָה לֹא תֵדְעִי שַׁחְרָהּ וְתִפֹּל עָלַיִךְ הֹוָה לֹא תוּכְלִי כַּפְּרָהּ וְתָבֹא עָלַיִךְ פִּתְאֹם שֹׁאָה לֹא תֵדָעִי׃    יב עִמְדִי-נָא בַחֲבָרַיִךְ וּבְרֹב כְּשָׁפַיִךְ בַּאֲשֶׁר יָגַעַתְּ מִנְּעוּרָיִךְ אוּלַי תּוּכְלִי הוֹעִיל אוּלַי תַּעֲרוֹצִי׃    יג נִלְאֵית בְּרֹב עֲצָתָיִךְ יַעַמְדוּ-נָא וְיוֹשִׁיעֻךְ הברו (הֹבְרֵי) שָׁמַיִם הַחֹזִים בַּכּוֹכָבִים מוֹדִעִים לֶחֳדָשִׁים מֵאֲשֶׁר יָבֹאוּ עָלָיִךְ׃    יד הִנֵּה הָיוּ כְקַשׁ אֵשׁ שְׂרָפָתַם לֹא-יַצִּילוּ אֶת-נַפְשָׁם מִיַּד לֶהָבָה אֵין-גַּחֶלֶת לַחְמָם אוּר לָשֶׁבֶת נֶגְדּוֹ׃    טו כֵּן הָיוּ-לָךְ אֲשֶׁר יָגָעַתְּ סֹחֲרַיִךְ מִנְּעוּרַיִךְ אִישׁ לְעֶבְרוֹ תָּעוּ אֵין מוֹשִׁיעֵךְ׃   ס
