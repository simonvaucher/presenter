לפרק הקודם * * * תוכן העניינים

ישעיהו פרק טז

א שִׁלְחוּ-כַר מֹשֵׁל-אֶרֶץ מִסֶּלַע מִדְבָּרָה אֶל-הַר בַּת-צִיּוֹן׃    ב וְהָיָה כְעוֹף-נוֹדֵד קֵן מְשֻׁלָּח תִּהְיֶינָה בְּנוֹת מוֹאָב מַעְבָּרֹת לְאַרְנוֹן׃    ג הביאו (הָבִיאִי) עֵצָה עֲשׂוּ פְלִילָה שִׁיתִי כַלַּיִל צִלֵּךְ בְּתוֹךְ צָהֳרָיִם סַתְּרִי נִדָּחִים נֹדֵד אַל-תְּגַלִּי׃    ד יָגוּרוּ בָךְ נִדָּחַי מוֹאָב הֱוִי-סֵתֶר לָמוֹ מִפְּנֵי שׁוֹדֵד  כִּי-אָפֵס הַמֵּץ כָּלָה שֹׁד תַּמּוּ רֹמֵס מִן-הָאָרֶץ׃   ס

ה וְהוּכַן בַּחֶסֶד כִּסֵּא וְיָשַׁב עָלָיו בֶּאֱמֶת בְּאֹהֶל דָּוִד שֹׁפֵט וְדֹרֵשׁ מִשְׁפָּט וּמְהִר צֶדֶק׃    ו שָׁמַעְנוּ גְאוֹן-מוֹאָב גֵּא מְאֹד  גַּאֲוָתוֹ וּגְאוֹנוֹ וְעֶבְרָתוֹ לֹא-כֵן בַּדָּיו׃    ז לָכֵן יְיֵלִיל מוֹאָב לְמוֹאָב כֻּלֹּה יְיֵלִיל  לַאֲשִׁישֵׁי קִיר-חֲרֶשֶׂת תֶּהְגּוּ אַךְ-נְכָאִים׃    ח כִּי שַׁדְמוֹת חֶשְׁבּוֹן אֻמְלָל גֶּפֶן שִׂבְמָה בַּעֲלֵי גוֹיִם הָלְמוּ שְׂרוּקֶּיהָ עַד-יַעְזֵר נָגָעוּ תָּעוּ מִדְבָּר שְׁלֻחוֹתֶיהָ נִטְּשׁוּ עָבְרוּ יָם׃    ט עַל-כֵּן אֶבְכֶּה בִּבְכִי יַעְזֵר גֶּפֶן שִׂבְמָה אֲרַיָּוֶךְ דִּמְעָתִי חֶשְׁבּוֹן וְאֶלְעָלֵה  כִּי עַל-קֵיצֵךְ וְעַל-קְצִירֵךְ הֵידָד נָפָל׃    י וְנֶאֱסַף שִׂמְחָה וָגִיל מִן-הַכַּרְמֶל וּבַכְּרָמִים לֹא-יְרֻנָּן לֹא יְרֹעָע יַיִן בַּיְקָבִים לֹא-יִדְרֹךְ הַדֹּרֵךְ הֵידָד הִשְׁבַּתִּי׃    יא עַל-כֵּן מֵעַי לְמוֹאָב כַּכִּנּוֹר יֶהֱמוּ וְקִרְבִּי לְקִיר חָרֶשׂ׃    יב וְהָיָה כִי-נִרְאָה כִּי-נִלְאָה מוֹאָב עַל-הַבָּמָה וּבָא אֶל-מִקְדָּשׁוֹ לְהִתְפַּלֵּל וְלֹא יוּכָל׃   ס

יג זֶה הַדָּבָר אֲשֶׁר דִּבֶּר יְהוָה אֶל-מוֹאָב מֵאָז׃    יד וְעַתָּה דִּבֶּר יְהוָה לֵאמֹר בְּשָׁלֹשׁ שָׁנִים כִּשְׁנֵי שָׂכִיר וְנִקְלָה כְּבוֹד מוֹאָב בְּכֹל הֶהָמוֹן הָרָב וּשְׁאָר מְעַט מִזְעָר לוֹא כַבִּיר׃   פ
