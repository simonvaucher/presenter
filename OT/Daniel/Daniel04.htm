לפרק הקודם * * * תוכן העניינים

דניאל פרק ד

א  אֲנָה נְבוּכַדְנֶצַּר שְׁלֵה הֲוֵית בְּבֵיתִי וְרַעְנַן בְּהֵיכְלִי׃ א  אֲנִי נְבוּכַדְנֶצַּר, שָׁלֵו הָיִיתִי בְּבֵיתִי, וְרַעֲנָן בְּהֵיכָלִי.
ב  חֵלֶם חֲזֵית וִידַחֲלִנַּנִי וְהַרְהֹרִין עַל-מִשְׁכְּבִי וְחֶזְוֵי רֵאשִׁי יְבַהֲלֻנַּנִי׃ ב  חֲלוֹם רָאִיתִי וְיַפְחִידֵנִי, וְהִרְהוּרִים עַל מִשְׁכָּבִי, וְחֶזְיוֹנוֹת רֹאשִׁי יְבַהֲלוּנִי.
ג  וּמִנִּי שִׂים טְעֵם לְהַנְעָלָה קָדָמַי לְכֹל חַכִּימֵי בָבֶל  דִּי-פְשַׁר חֶלְמָא יְהוֹדְעֻנַּנִי׃ ג  וּמִמֶּנִּי הוּשַׂם צַו, לְהַכְנִיס לְפָנַי אֶת כָּל חַכְמֵי בָבֶל, שֶׁפִּתְרוֹן הַחֲלוֹם יוֹדִיעוּנִי.
ד  בֵּאדַיִן עללין (עָלִּין) חַרְטֻמַּיָּא אָשְׁפַיָּא כשדיא (כַּשְׂדָּאֵי) וְגָזְרַיָּא וְחֶלְמָא אָמַר אֲנָה קֳדָמֵיהוֹן וּפִשְׁרֵהּ לָא-מְהוֹדְעִין לִי׃ ד  אֲזַי נִכְנָסִים הַחַרְטֻמִּים, הַכַּשְׂדִּים וְהַקּוֹסְמִים, וְאֶת הַחֲלוֹם אֲנִי אוֹמֵר לִפְנֵיהֶם, וּפִתְרוֹנוֹ אֵין מוֹדִיעִים לִי.
ה  וְעַד אָחֳרֵין עַל קָדָמַי דָּנִיֵּאל דִּי-שְׁמֵהּ בֵּלְטְשַׁאצַּר כְּשֻׁם אֱלָהִי וְדִי רוּחַ-אֱלָהִין קַדִּישִׁין בֵּהּ וְחֶלְמָא קָדָמוֹהִי אַמְרֵת׃ ה  וְעַד לָאַחֲרוֹנָה נִכְנַס לְפָנַי דָּנִיֵּאל, אֲשֶׁר שְׁמוֹ בֵּלְטְשַׁאצַּר כְּשֵׁם אֱלֹהַי, וַאֲשֶׁר רוּחַ אֱלֹהִים קְדוֹשִׁים בּוֹ, וְאֶת הַחֲלוֹם לְפָנָיו אָמַרְתִּי.
ו  בֵּלְטְשַׁאצַּר רַב חַרְטֻמַּיָּא דִּי אֲנָה יִדְעֵת דִּי רוּחַ אֱלָהִין קַדִּישִׁין בָּךְ וְכָל-רָז לָא-אָנֵס לָךְ חֶזְוֵי חֶלְמִי דִי-חֲזֵית וּפִשְׁרֵהּ אֱמַר׃ ו  בֵּלְטְשַׁאצַּר רַב הַחַרְטֻמִּים, אֲשֶׁר אֲנִי יָדַעְתִּי כִּי רוּחַ אֱלֹהִים קְדוֹשִׁים בָּךְ, וְכָל רָז אֵינוֹ מוֹנֵעַ אוֹתְךָ, חֶזְיוֹנוֹת חֲלוֹמִי אֲשֶׁר רָאִיתִי וּפִתְרוֹנוֹ אֱמֹר.
ז  וְחֶזְוֵי רֵאשִׁי עַל-מִשְׁכְּבִי חָזֵה הֲוֵית וַאֲלוּ אִילָן בְּגוֹ אַרְעָא וְרוּמֵהּ שַׂגִּיא׃ ז  וְחֶזְיוֹנוֹת רֹאשִׁי עַל מִשְׁכָּבִי, רוֹאֶה הָיִיתִי וְהִנֵּה אִילָן בְּתוֹךְ הָאָרֶץ, וְרוּמוֹ רָב.
ח  רְבָה אִילָנָא וּתְקִף וְרוּמֵהּ יִמְטֵא לִשְׁמַיָּא וַחֲזוֹתֵהּ לְסוֹף כָּל-אַרְעָא׃ ח  גָּדַל הָאִילָן וְחָזַק, וְרוּמוֹ יַגִּיעַ לַשָׁמַיִם, וּמַרְאֵהוּ לְסוֹף כָּל הָאָרֶץ.
ט  עָפְיֵהּ שַׁפִּיר וְאִנְבֵּהּ שַׂגִּיא וּמָזוֹן לְכֹלָּא-בֵהּ תְּחֹתוֹהִי תַּטְלֵל חֵיוַת בָּרָא וּבְעַנְפוֹהִי ידרון (יְדוּרָן) צִפְּרֵי שְׁמַיָּא וּמִנֵּהּ יִתְּזִין כָּל-בִּשְׂרָא׃ ט  עָפְיוֹ יָפֶה וּפִרְיוֹ רַב, וּמָזוֹן לַכֹּל בּוֹ, תַּחְתָּיו תִּמְצָא צֵל חַיַּת הַבָּר, וּבַעֲנָפָיו תָּדֹרְנָה צִפֳּרֵי הַשָׁמִים, וּמִמֶּנּוּ יִזּוֹן כָּל בָּשָׂר.
י  חָזֵה הֲוֵית בְּחֶזְוֵי רֵאשִׁי עַל-מִשְׁכְּבִי וַאֲלוּ עִיר וְקַדִּישׁ מִן-שְׁמַיָּא נָחִת׃ י  רוֹאֶה הָיִיתִי בְּחֶזְיוֹנוֹת רֹאשִׁי עַל מִשְׁכָּבִי, וְהִנֵּה מַלְאָךְ וְקָדוֹשׁ מִשָׁמַיִם יוֹרֵד.
יא  קָרֵא בְחַיִל וְכֵן אָמַר גֹּדּוּ אִילָנָא וְקַצִּצוּ עַנְפוֹהִי אַתַּרוּ עָפְיֵהּ וּבַדַּרוּ אִנְבֵּהּ תְּנֻד חֵיוְתָא מִן-תַּחְתּוֹהִי וְצִפְּרַיָּא מִן-עַנְפוֹהִי׃ יא  קוֹרֵא בְכֹחַ וְכֵן אוֹמֵר: גֹּזוּ הָאִילָן וְקַצְּצוּ עֲנָפָיו, הַסִּירוּ עָפְיוֹ וּפַזְּרוּ פִרְיוֹ, תָּגוּד הַחַיָּה מִתַּחְתָּיו, וְהַצִּפֳּרִים מֵעֲנָפָיו.
יב  בְּרַם עִקַּר שָׁרְשׁוֹהִי בְּאַרְעָא שְׁבֻקוּ וּבֶאֱסוּר דִּי-פַרְזֶל וּנְחָשׁ בְּדִתְאָא דִּי בָרָא וּבְטַל שְׁמַיָּא יִצְטַבַּע וְעִם-חֵיוְתָא חֲלָקֵהּ בַּעֲשַׂב אַרְעָא׃ יב  אוּלָם עִקַּר שָׁרָשָׁיו בָּאָרֶץ הַנִּיחוּ, וּבְאֵסוּר שֶׁל בַּרְזֶל וּנְחֹשֶׁת בְּדֶשֶׁא שֶׁל הַבָּר, וּבְטַל הַשָׁמַיִם יִטְבֹּל, וְעִם הַחַיָּה חֶלְקוֹ בְּעֵשֶׂב הָאָרֶץ.
יג  לִבְבֵהּ מִן-אנושא (אֲנָשָׁא) יְשַׁנּוֹן וּלְבַב חֵיוָה יִתְיְהִב לֵהּ וְשִׁבְעָה עִדָּנִין יַחְלְפוּן עֲלוֹהִי׃ יג  לִבּוֹ מֵאֱנוֹשׁ יְשַׁנּוּ, וְלֵב חַיָּה יִנָּתֵן לוֹ, וְשִׁבְעָה מוֹעֲדִים יַחְלְפוּ עָלָיו.
יד  בִּגְזֵרַת עִירִין פִּתְגָמָא וּמֵאמַר קַדִּישִׁין שְׁאֵלְתָא עַד-דִּבְרַת דִּי יִנְדְּעוּן חַיַּיָּא דִּי-שַׁלִּיט עליא (עִלָּאָה) בְּמַלְכוּת אנושא (אֲנָשָׁא) וּלְמַן-דִּי יִצְבֵּא יִתְּנִנַּהּ וּשְׁפַל אֲנָשִׁים יְקִים עליה (עֲלַהּ)׃ יד  בִּגְזֵרַת מַלְאָכִים הַדָּבָר, וּמַאֲמַר קְדוֹשִׁים הַמִּשְׁאָלָה, עַל דִּבְרַת שֶׁיֵּדְעוּ הַחַיִּים, כִּי שׁוֹלֵט הָעֶלְיוֹן בְּמַלְכוּת אֱנוֹשׁ, וּלְמִי שֶׁיִּרְצֶה יִתְּנֶנָּה, וּשְׁפַל אֲנָשִׁים יָקִים עָלֶיהָ.
טו  דְּנָה חֶלְמָא חֲזֵית אֲנָה מַלְכָּא נְבוּכַדְנֶצַּר ואנתה (וְאַנְתְּ) בֵּלְטְשַׁאצַּר פִּשְׁרֵא אֱמַר כָּל-קֳבֵל דִּי כָּל-חַכִּימֵי מַלְכוּתִי לָא-יָכְלִין פִּשְׁרָא לְהוֹדָעוּתַנִי ואנתה (וְאַנְתְּ) כָּהֵל דִּי רוּחַ-אֱלָהִין קַדִּישִׁין בָּךְ׃ טו  זֶה הַחֲלוֹם רָאִיתִי, אֲנִי הַמֶּלֶךְ נְבוּכַדְנֶצַּר, וְאַתָּה בֵּלְטְשַׁאצַּר פִּתְרוֹנוֹ אֱמֹר, כָּל-עֻמַּת שֶׁכָּל חַכְמֵי מַלְכוּתִי אֵינָם יְכוֹלִים לְהוֹדִיעֵנִי הַפִּתְרוֹן, וְאַתָּה יָכוֹל, כִּי רוּחַ אֱלֹהִים קְדוֹשִׁים בָּךְ.
טז  אֱדַיִן דָּנִיֵּאל דִּי-שְׁמֵהּ בֵּלְטְשַׁאצַּר אֶשְׁתּוֹמַם כְּשָׁעָה חֲדָה וְרַעְיֹנֹהִי יְבַהֲלֻנֵּהּ עָנֵה מַלְכָּא וְאָמַר בֵּלְטְשַׁאצַּר חֶלְמָא וּפִשְׁרֵא אַל-יְבַהֲלָךְ עָנֵה בֵלְטְשַׁאצַּר וְאָמַר מָרִאי חֶלְמָא לשנאיך (לְשָׂנְאָךְ) וּפִשְׁרֵהּ לעריך (לְעָרָךְ)׃ טז  אֲזַי דָּנִיֵּאל, אֲשֶׁר שְׁמוֹ בֵּלְטְשַׁאצַּר, הִשְׁתּוֹמֵם כְּשָׁעָה אַחַת וְרַעְיוֹנָיו יְבַהֲלוּהוּ. עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר: בֵּלְטְשַׁאצַּר, הַחֲלוֹם וּפִתְרוֹנוֹ אַל יְבַהֲלֶךָ! עוֹנֶה בֵּלְטְשַׁאצַּר וְאוֹמֵר: אֲדוֹנִי, הַחֲלוֹם לְשׂוֹנְאֶיךָ וּפִתְרוֹנוֹ לְצָרֶיךָ.
יז  אִילָנָא דִּי חֲזַיְתָ דִּי רְבָה וּתְקִף וְרוּמֵהּ יִמְטֵא לִשְׁמַיָּא וַחֲזוֹתֵהּ לְכָל-אַרְעָא׃ יז  הָאִילָן אֲשֶׁר רָאִיתָ, אֲשֶׁר גָּדַל וְחָזַק, וְרוּמוֹ מַגִּיעַ לַשָׁמַיִם, וּמַרְאֵהוּ לְכָל הָאָרֶץ.
יח  וְעָפְיֵהּ שַׁפִּיר וְאִנְבֵּהּ שַׂגִּיא וּמָזוֹן לְכֹלָּא-בֵהּ תְּחֹתוֹהִי תְּדוּר חֵיוַת בָּרָא וּבְעַנְפוֹהִי יִשְׁכְּנָן צִפְּרֵי שְׁמַיָּא׃ יח  וְעָפְיוֹ יָפֶה וּפִרְיוֹ רַב, וּמָזוֹן לַכֹּל בּוֹ; תַּחְתָּיו תָּדוּר חַיַּת הַבָּר, וּבַעֲנָפָיו תִּשְׁכֹּנָּה צִפֳּרֵי הַשָׁמַיִם.
יט  אנתה- (אַנְתְּ-) הוּא מַלְכָּא דִּי רבית (רְבַת) וּתְקֵפְתְּ וּרְבוּתָךְ רְבָת וּמְטָת לִשְׁמַיָּא וְשָׁלְטָנָךְ לְסוֹף אַרְעָא׃ יט  אַתָּה הוּא הַמֶּלֶךְ, אֲשֶׁר גָּדַלְתָּ וְחָזַקְתָּ, וּגְדֻלָּתְךָ גָּדְלָה וְהִגִּיעָה לַשָׁמַיִם וְשִׁלְטוֹנְךָ לְסוֹף הָאָרֶץ.
כ  וְדִי חֲזָה מַלְכָּא עִיר וְקַדִּישׁ נָחִת מִן-שְׁמַיָּא וְאָמַר גֹּדּוּ אִילָנָא וְחַבְּלוּהִי בְּרַם עִקַּר שָׁרְשׁוֹהִי בְּאַרְעָא שְׁבֻקוּ וּבֶאֱסוּר דִּי-פַרְזֶל וּנְחָשׁ בְּדִתְאָא דִּי בָרָא וּבְטַל שְׁמַיָּא יִצְטַבַּע וְעִם-חֵיוַת בָּרָא חֲלָקֵהּ עַד דִּי-שִׁבְעָה עִדָּנִין יַחְלְפוּן עֲלוֹהִי׃ כ  וַאֲשֶׁר רָאָה הַמֶּלֶךְ, מַלְאָךְ וְקָדוֹשׁ יוֹרֵד מִן הַשָׁמַיִם וְאוֹמֵר: גֹּזוּ הָאִילָן וְהַשְׁחִיתוּהוּ, אוּלָם עִקַּר שָׁרָשָׁיו בָּאָרֶץ הַנִּיחוּ, וּבְאֵסוּר שֶׁל בַּרְזֶל וּנְחֹשֶׁת בְּדֶשֶׁא שֶׁל הַבָּר, וּבְטַל הַשָׁמַיִם יִטְבֹּל, וְעִם חַיַּת הַבָּר חֶלְקוֹ, עַד אֲשֶׁר שִׁבְעָה מוֹעֲדִים יַחְלְפוּ עָלָיו.
כא  דְּנָה פִשְׁרָא מַלְכָּא וּגְזֵרַת עליא (עִלָּאָה) הִיא דִּי מְטָת עַל-מַרִאי מַלְכָּא׃ כא  זֶה הַפִּתְרוֹן, הַמֶּלֶךְ, וּגְזֵרַת עֶלְיוֹן הִיא, שֶׁהִגִּיעָה אֶל אֲדוֹנִי הַמֶּלֶךְ.
כב  וְלָךְ טָרְדִין מִן-אֲנָשָׁא וְעִם-חֵיוַת בָּרָא לֶהֱוֵה מְדֹרָךְ וְעִשְׂבָּא כְתוֹרִין לָךְ יְטַעֲמוּן וּמִטַּל שְׁמַיָּא לָךְ מְצַבְּעִין וְשִׁבְעָה עִדָּנִין יַחְלְפוּן עליך (עֲלָךְ)  עַד דִּי-תִנְדַּע דִּי-שַׁלִּיט עליא (עִלָּאָה) בְּמַלְכוּת אֲנָשָׁא וּלְמַן-דִּי יִצְבֵּא יִתְּנִנַּהּ׃ כב  וְאוֹתְךָ מְגָרְשִׁים מֵאֱנוֹשׁ, וְעִם חַיַּת הַבָּר יִהְיֶה מְדוֹרְךָ, וְעֵשֶׂב כִּשְׁוָרִים אוֹתְךָ יַאֲכִילוּ, וּמִטַּל הַשָׁמַיִם יִטְבְּלוּךָ, וְשִׁבְעָה מוֹעֲדִים יַחְלְפוּ עָלֶיךָ עַד שֶׁתֵּדַע, כִּי שׁוֹלֵט עֶלְיוֹן בְּמַלְכוּת אֱנוֹשׁ, וּלְמִי שֶׁיִּרְצֶה יִתְּנֶנָּה.
כג  וְדִי אֲמַרוּ לְמִשְׁבַּק עִקַּר שָׁרְשׁוֹהִי דִּי אִילָנָא מַלְכוּתָךְ לָךְ קַיָּמָא מִן-דִּי תִנְדַּע דִּי שַׁלִּטִן שְׁמַיָּא׃ כג  וַאֲשֶׁר אָמְרוּ לְהַנִּיחַ עִקַּר שָׁרָשָׁיו שֶׁל הָאִילָן, הַמַּלְכוּת לְךָ קַיֶּמֶת, מִשֶּתֵּדַע כִּי שׁוֹלְטִים הַשָׁמַיִם.
כד  לָהֵן מַלְכָּא מִלְכִּי יִשְׁפַּר עליך (עֲלָךְ) וחטיך (וַחֲטָאָךְ) בְּצִדְקָה פְרֻק וַעֲוָיָתָךְ בְּמִחַן עֲנָיִן הֵן תֶּהֱוֵה אַרְכָה לִשְׁלֵוְתָךְ׃ כד  אֶלָּא הַמֶּלֶךְ, עֲצָתִי תִּיטַב עָלֶיךָ, וַחֲטָאֶיךָ בִּצְדָקָה פְדֵה וַעֲוֹנוֹתֶיךָ בְּחֹן עֲנִיִּים, אִם תִּהְיֶה אַרְכָּה לְשַׁלְוָתֶךָ.
כה  כֹּלָּא מְּטָא עַל-נְבוּכַדְנֶצַּר מַלְכָּא׃   פ כה  הַכֹּל הִגִּיעַ אֶל נְבוּכַדְנֶצַּר הַמֶּלֶךְ.   פ

כו  לִקְצָת יַרְחִין תְּרֵי-עֲשַׂר עַל-הֵיכַל מַלְכוּתָא דִּי בָבֶל מְהַלֵּךְ הֲוָה׃ כו  לְקֵץ יְרָחִים שְׁנֵים עָשָׂר, עַל הֵיכַל הַמַּלְכוּת שֶׁל בָּבֶל מְהַלֵּךְ הָיָה.
כז  עָנֵה מַלְכָּא וְאָמַר הֲלָא דָא-הִיא בָּבֶל רַבְּתָא דִּי-אֲנָה בֱנַיְתַהּ לְבֵית מַלְכוּ בִּתְקַף חִסְנִי וְלִיקָר הַדְרִי׃ כז  עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר: הֲלֹא זוֹ הִיא בָבֶל הַגְּדוֹלָה, אֲשֶׁר אֲנִי בְנִיתִיהָ לְבֵית מַלְכוּת, בְּתֹקֶף חָסְנִי וְלִכְבוֹד הֲדָרִי.
כח  עוֹד מִלְּתָא בְּפֻם מַלְכָּא קָל מִן-שְׁמַיָּא נְפַל  לָךְ אָמְרִין נְבוּכַדְנֶצַּר מַלְכָּא מַלְכוּתָא עֲדָת מִנָּךְ׃ כח  עוֹד הַדָּבָר בְּפִי הַמֶּלֶךְ, קוֹל מִן הַשָׁמַיִם נָפַל: לְךָ אוֹמְרִים נְבוּכַדְנֶצַּר הַמֶּלֶךְ, הַמַּלְכוּת עָבְרָה מִמֶּךָּ.
כט  וּמִן-אֲנָשָׁא לָךְ טָרְדִין וְעִם-חֵיוַת בָּרָא מְדֹרָךְ עִשְׂבָּא כְתוֹרִין לָךְ יְטַעֲמוּן וְשִׁבְעָה עִדָּנִין יַחְלְפוּן עליך (עֲלָךְ)  עַד דִּי-תִנְדַּע דִּי-שַׁלִּיט עליא (עִלָּאָה) בְּמַלְכוּת אֲנָשָׁא וּלְמַן-דִּי יִצְבֵּא יִתְּנִנַּהּ׃ כט  וּמֵאֱנוֹשׁ אוֹתְךָ מְגָרְשִׁים, וְעִם חַיַּת הַבָּר מְדוֹרְךָ, עֵשֶׂב כִּשְׁוָרִים אוֹתְךָ יַאֲכִילוּ, וְשִׁבְעָה מוֹעֲדִים יַחְלְפוּ עָלֶיךָ עַד שֶׁתֵּדַע, כִּי שׁוֹלֵט עֶלְיוֹן בְּמַלְכוּת אֱנוֹשׁ, וּלְמִי שֶׁיִּרְצֶה יִתְּנֶנָּה.
ל  בַּהּ-שַׁעֲתָא מִלְּתָא סָפַת עַל-נְבוּכַדְנֶצַּר וּמִן-אֲנָשָׁא טְרִיד וְעִשְׂבָּא כְתוֹרִין יֵאכֻל וּמִטַּל שְׁמַיָּא גִּשְׁמֵהּ יִצְטַבַּע עַד דִּי שַׂעְרֵהּ כְּנִשְׁרִין רְבָה וְטִפְרוֹהִי כְצִפְּרִין׃ ל  בָּהּ בַּשָׁעָה, הַדָּבָר כָּלָה עַל נְבוּכַדְנֶצַּר, וּמֵאֱנוֹשׁ גֹּרַשׁ, וְעֵשֶׂב כִּשְׁוָרִים יֹאכַל, וּמִטַּל הַשָׁמַיִם גּוּפוֹ יִטָּבֵל עַד שֶׁשְּׁעָרוֹ כִּנְשָׁרִים גָּדַל וְצִפָּרְנָיו כְּצִפֳּרִים.
לא  וְלִקְצָת יוֹמַיָּא אֲנָה נְבוּכַדְנֶצַּר עַיְנַי לִשְׁמַיָּא נִטְלֵת וּמַנְדְּעִי עֲלַי יְתוּב ולעליא (וּלְעִלָּאָה) בָּרְכֵת וּלְחַי עָלְמָא שַׁבְּחֵת וְהַדְּרֵת  דִּי שָׁלְטָנֵהּ שָׁלְטָן עָלַם וּמַלְכוּתֵהּ עִם-דָּר וְדָר׃ לא  וּלְקֵץ הַיָּמִים, אֲנִי נְבוּכַדְנֶצַּר, עֵינַי לַשָׁמַיִם נָשָׂאתִי, וְדַעְתִּי עָלַי תָּשׁוּב, וְאֶת הָעֶלְיוֹן בֵּרַכְתִּי, וְאֶת חַי הָעוֹלָם שִׁבַּחְתִּי וּפֵאַרְתִּי, אֲשֶׁר שִׁלְטוֹנוֹ שִׁלְטוֹן עוֹלָם וּמַלְכוּתוֹ עִם דּוֹר וָדוֹר.
לב  וְכָל-דארי (דָּיְרֵי) אַרְעָא כְּלָה חֲשִׁיבִין וּכְמִצְבְּיֵהּ עָבֵד בְּחֵיל שְׁמַיָּא ודארי (וְדָיְרֵי) אַרְעָא וְלָא אִיתַי דִּי-יְמַחֵא בִידֵהּ וְיֵאמַר לֵהּ מָה עֲבַדְתְּ׃ לב  וְכָל דָּרֵי הָאָרֶץ כְּאַיִן חֲשׁוּבִים, וְכִרְצוֹנוֹ עוֹשֶׂה בְּחֵיל הַשָׁמַיִם וְדָרֵי הָאָרֶץ, וְאֵין אֲשֶׁר יְמַחֶה בְיָדוֹ וְיֹאמַר לוֹ מָה עָשִׂיתָ.
לג  בֵּהּ-זִמְנָא מַנְדְּעִי יְתוּב עֲלַי וְלִיקַר מַלְכוּתִי הַדְרִי וְזִיוִי יְתוּב עֲלַי וְלִי הַדָּבְרַי וְרַבְרְבָנַי יְבַעוֹן וְעַל-מַלְכוּתִי הָתְקְנַת וּרְבוּ יַתִּירָה הוּסְפַת לִי׃ לג  בּוֹ בַזְּמַן דַּעְתִּי תָּשׁוּב עָלַי, וְלִיקַר מַלְכוּתִי הֲדָרִי, וְזִיוִי יָשׁוּב אֵלַי, וְאוֹתִי -- יוֹעֲצַי וְשָׂרַי יְבַקְּשׁוּ, וְעַל מַלְכוּתִי הוּכַנְתִּי, וּגְדֻלָּה יְתֵרָה נוֹסְפָה לִי.
לד  כְּעַן אֲנָה נְבֻכַדְנֶצַּר מְשַׁבַּח וּמְרוֹמֵם וּמְהַדַּר לְמֶלֶךְ שְׁמַיָּא דִּי כָל-מַעֲבָדוֹהִי קְשֹׁט וְאֹרְחָתֵהּ דִּין וְדִי מַהְלְכִין בְּגֵוָה יָכִל לְהַשְׁפָּלָה׃   פ לד  כָּעֵת אֲנִי נְבוּכַדְנֶצַּר, מְשַׁבֵּחַ וּמְרוֹמֵם וּמְפָאֵר אֶת מֶלֶךְ הַשָׁמַיִם, אֲשֶׁר כָּל מַעֲשָׂיו אֱמֶת וְאָרְחוֹתָיו מִשְׁפָּט, וַאֲשֶׁר מְהַלְּכִים בְּגַאֲוָה יָכוֹל לְהַשְׁפִּיל.   פ
