לפרק הקודם * * * תוכן העניינים

דניאל פרק ז

א  בִּשְׁנַת חֲדָה לְבֵלְאשַׁצַּר מֶלֶךְ בָּבֶל דָּנִיֵּאל חֵלֶם חֲזָה וְחֶזְוֵי רֵאשֵׁהּ עַל-מִשְׁכְּבֵהּ בֵּאדַיִן חֶלְמָא כְתַב רֵאשׁ מִלִּין אֲמַר׃ א  בִּשְׁנַת אַחַת לְבֵלְשַׁאצַּר מֶלֶךְ בָּבֶל, דָּנִיֵּאל חֲלוֹם רָאָה וְחֶזְיוֹנוֹת רֹאשׁוֹ עַל מִשְׁכָּבוֹ; אֲזַי אֶת הַחֲלוֹם כָּתַב, רֹאשׁ מִלִּים אָמַר.
ב  עָנֵה דָנִיֵּאל וְאָמַר חָזֵה הֲוֵית בְּחֶזְוִי עִם-לֵילְיָא וַאֲרוּ אַרְבַּע רוּחֵי שְׁמַיָּא מְגִיחָן לְיַמָּא רַבָּא׃ ב  עוֹנֶה דָּנִיֵּאל וְאוֹמֵר: רוֹאֶה הָיִיתִי בַּחֲזוֹנִי עִם לַיְלָה, וְהִנֵּה אַרְבַּע רוּחוֹת הַשָׁמַיִם מְגִיחוֹת לַיָּם הַגָּדוֹל.
ג  וְאַרְבַּע חֵיוָן רַבְרְבָן סָלְקָן מִן-יַמָּא שָׁנְיָן דָּא מִן-דָּא׃ ג  וְאַרְבַּע חַיּוֹת גְּדוֹלוֹת עוֹלוֹת מִן הַיָּם, שׁוֹנוֹת זוֹ מִזּוֹ.
ד  קַדְמָיְתָא כְאַרְיֵה וְגַפִּין דִּי-נְשַׁר לַהּ חָזֵה הֲוֵית עַד דִּי-מְּרִיטוּ גפיה (גַפַּהּ) וּנְטִילַת מִן-אַרְעָא וְעַל-רַגְלַיִן כֶּאֱנָשׁ הֳקִימַת וּלְבַב אֱנָשׁ יְהִיב לַהּ׃ ד  הָרִאשׁוֹנָה כְּאַרְיֵה וּכְנָפַיִם שֶׁל נֶשֶׁר לָהּ, רוֹאֶה הָיִיתִי עַד אֲשֶׁר נִמְרְטוּ כְנָפֶיהָ וְנִשְׂאָה מִן הָאָרֶץ, וְעַל רַגְלַיִם כְּאָדָם הוּקָמָה, וּלְבַב אָדָם נִתַּן לָהּ.
ה  וַאֲרוּ חֵיוָה אָחֳרִי תִנְיָנָה דָּמְיָה לְדֹב וְלִשְׂטַר-חַד הֳקִמַת וּתְלָת עִלְעִין בְּפֻמַּהּ בֵּין שניה (שִׁנַּהּ) וְכֵן אָמְרִין לַהּ קוּמִי אֲכֻלִי בְּשַׂר שַׂגִּיא׃ ה  וְהִנֵּה חַיָּה אַחֶרֶת שְׁנִיָּה, דּוֹמָה לְדֹב, וּלְצַד אֶחָד הָעָמְדָה, וְשָׁלֹשׁ צְלָעוֹת בְּפִיהָ בֵּין שִׁנֶּיהָ, וְכֵן אוֹמְרִים לָהּ: קוּמִי אִכְלִי בָּשָׂר רָב.
ו  בָּאתַר דְּנָה חָזֵה הֲוֵית וַאֲרוּ אָחֳרִי כִּנְמַר וְלַהּ גַּפִּין אַרְבַּע דִּי-עוֹף עַל-גביה (גַּבַּהּ) וְאַרְבְּעָה רֵאשִׁין לְחֵיוְתָא וְשָׁלְטָן יְהִיב לַהּ׃ ו  אַחֲרֵי זֶה רוֹאֶה הָיִיתִי, וְהִנֵּה אַחֶרֶת כְּנָמֵר, וְלָהּ כְּנָפַיִם אַרְבַּע שֶׁל עוֹף עַל גַּבָּהּ, וְאַרְבָּעָה רָאשִׁים לַחַיָּה, וְשִׁלְטוֹן נִתַּן לָהּ.
ז  בָּאתַר דְּנָה חָזֵה הֲוֵית בְּחֶזְוֵי לֵילְיָא וַאֲרוּ חֵיוָה רביעיה (רְבִיעָאָה) דְּחִילָה וְאֵימְתָנִי וְתַקִּיפָא יַתִּירָה וְשִׁנַּיִן דִּי-פַרְזֶל לַהּ רַבְרְבָן אָכְלָה וּמַדֱּקָה וּשְׁאָרָא ברגליה (בְּרַגְלַהּ) רָפְסָה וְהִיא מְשַׁנְּיָה מִן-כָּל-חֵיוָתָא דִּי קדמיה (קָדָמַהּ) וְקַרְנַיִן עֲשַׂר לַהּ׃ ז  אַחֲרֵי זֶה רוֹאֶה הָיִיתִי בְּחֶזְיוֹנוֹת לַיְלָה, וְהִנֵּה חַיָּה רְבִיעִית נוֹרָאָה וַאֲיֻמָּה וַחֲזָקָה יוֹתֵר, וְשִׁנַּיִם שֶׁל בַּרְזֶל לָהּ גְּדוֹלוֹת, אוֹכֶלֶת וּמְדִקָּה, וְהַשְׁאָר בְּרַגְלָהּ רוֹפֶסֶת, וְהִיא מְשֻׁנָּה מִכָּל הַחַיּוֹת אֲשֶׁר לְפָנֶיהָ, וְקַרְנַיִם עֶשֶׂר לָהּ.
ח  מִשְׂתַּכַּל הֲוֵית בְּקַרְנַיָּא וַאֲלוּ קֶרֶן אָחֳרִי זְעֵירָה סִלְקָת ביניהון (בֵּינֵיהֵן) וּתְלָת מִן-קַרְנַיָּא קַדְמָיָתָא אתעקרו (אֶתְעֲקַרָה) מִן-קדמיה (קֳדָמַהּ) וַאֲלוּ עַיְנִין כְּעַיְנֵי אֲנָשָׁא בְּקַרְנָא-דָא וּפֻם מְמַלִּל רַבְרְבָן׃ ח  מִסְתַּכֵּל הָיִיתִי בַּקַּרְנַיִם, וְהִנֵּה קֶרֶן אַחֶרֶת זְעִירָה עָלְתָה בֵּינֵיהֶן, וְשָׁלֹשׁ מִן הַקַּרְנַיִם הָרִאשׁוֹנוֹת נֶעֶקְרוּ מִפָּנֶיהָ, וְהִנֵּה עֵינַיִם כְּעֵינֵי אָדָם בְּקֶרֶן זוֹ, וּפֶה מְדַבֵּר גְּדוֹלוֹת.
ט  חָזֵה הֲוֵית עַד דִּי כָרְסָוָן רְמִיו וְעַתִּיק יוֹמִין יְתִב לְבוּשֵׁהּ כִּתְלַג חִוָּר וּשְׂעַר רֵאשֵׁהּ כַּעֲמַר נְקֵא כָּרְסְיֵהּ שְׁבִבִין דִּי-נוּר גַּלְגִּלּוֹהִי נוּר דָּלִק׃ ט  רוֹאֶה הָיִיתִי עַד אֲשֶׁר כִּסְאוֹת הֻנְּחוּ וְעַתִּיק יָמִים יָשַׁב, לְבוּשׁוֹ כְּשֶׁלֶג לָבָן וּשְׂעַר רֹאשׁוֹ כְּצֶמֶר נָקִי, כִּסְאוֹ שְׁבִיבִים שֶׁל אֵשׁ, גַּלְגַּלָּיו אֵשׁ דּוֹלֶקֶת.
י  נְהַר דִּי-נוּר נָגֵד וְנָפֵק מִן-קֳדָמוֹהִי אֶלֶף אלפים (אַלְפִין) יְשַׁמְּשׁוּנֵּהּ וְרִבּוֹ רבון (רִבְבָן) קָדָמוֹהִי יְקוּמוּן דִּינָא יְתִב וְסִפְרִין פְּתִיחוּ׃ י  נָהָר שֶׁל אֵשׁ נִמְשָׁךְ וְיוֹצֵא מִלְּפָנָיו, אֶלֶף אֲלָפִים יְשַׁמְּשׁוּהוּ וְרִבּוֹ רְבָבוֹת לְפָנָיו יַעֲמֹדוּ, הַדִּין יָשַׁב וּסְפָרִים נִפְתָּחוּ.
יא  חָזֵה הֲוֵית בֵּאדַיִן מִן-קָל מִלַּיָּא רַבְרְבָתָא דִּי קַרְנָא מְמַלֱּלָה חָזֵה הֲוֵית עַד דִּי קְטִילַת חֵיוְתָא וְהוּבַד גִּשְׁמַהּ וִיהִיבַת לִיקֵדַת אֶשָּׁא׃ יא  רוֹאֶה הָיִיתִי, אֲזַי מִקּוֹל הַדְּבָרִים הַגְּדוֹלִים שֶׁל הַקֶּרֶן הַמְּדַבֶּרֶת, רוֹאֶה הָיִיתִי, עַד אֲשֶׁר הוּמְתָה הַחַיָּה וְהָאֳבַד גּוּפָהּ וְנִתְּנָה לִיקוֹד אֵשׁ.
יב  וּשְׁאָר חֵיוָתָא הֶעְדִּיו שָׁלְטָנְהוֹן וְאַרְכָה בְחַיִּין יְהִיבַת לְהוֹן עַד-זְמַן וְעִדָּן׃ יב  וּשְׁאָר הַחַיּוֹת הָעָבְרוּ מִשִּׁלְטוֹנָן, וְאַרְכָּה בַחַיִּים נִתְּנָה לָהֶן עַד זְמָן וּמוֹעֵד.
יג  חָזֵה הֲוֵית בְּחֶזְוֵי לֵילְיָא וַאֲרוּ עִם-עֲנָנֵי שְׁמַיָּא כְּבַר אֱנָשׁ אָתֵה הֲוָא וְעַד-עַתִּיק יוֹמַיָּא מְטָה וּקְדָמוֹהִי הַקְרְבוּהִי׃ יג  רוֹאֶה הָיִיתִי בְּחֶזְיוֹנוֹת הַלַּיְלָה וְהִנֵּה עִם עַנְנֵי הַשָׁמַיִם כְּבֶן אָדָם בָּא הָיָה, וְעַד עַתִּיק הַיָּמִים הִגִּיעַ, וּלְפָנָיו הִקְרִיבוּהוּ.
יד  וְלֵהּ יְהִב שָׁלְטָן וִיקָר וּמַלְכוּ וְכֹל עַמְמַיָּא אֻמַּיָּא וְלִשָּׁנַיָּא לֵהּ יִפְלְחוּן שָׁלְטָנֵהּ שָׁלְטָן עָלַם דִּי-לָא יֶעְדֵּה וּמַלְכוּתֵהּ דִּי-לָא תִתְחַבַּל׃   פ יד  וְלוֹ נִתַּן שִׁלְטוֹן וִיקָר וּמַלְכוּת, וְכָל הָעַמִּים, הָאֻמּוֹת וְהַלְּשׁוֹנוֹת אוֹתוֹ יַעֲבֹדוּ, שִׁלְטוֹנוֹ שִׁלְטוֹן עוֹלָם אֲשֶׁר לֹא יַעֲבֹר וּמַלְכוּתוֹ -- אֲשֶׁר לֹא תִּשָׁחֵת.   פ

טו  אֶתְכְּרִיַּת רוּחִי אֲנָה דָנִיֵּאל בְּגוֹ נִדְנֶה וְחֶזְוֵי רֵאשִׁי יְבַהֲלֻנַּנִי׃ טו  נִפְעֲמָה רוּחִי, אֲנִי דָּנִיֵּאל, בְּתוֹךְ זֶה, וְחֶזְיוֹנוֹת רֹאשִׁי יְבַהֲלוּנִי.
טז  קִרְבֵת עַל-חַד מִן-קָאֲמַיָּא וְיַצִּיבָא אֶבְעֵא-מִנֵּהּ עַל-כָּל-דְּנָה וַאֲמַר-לִי וּפְשַׁר מִלַּיָּא יְהוֹדְעִנַּנִי׃ טז  קָרַבְתִּי אֶל אֶחָד הָעוֹמְדִים, וְאֶת הַנָּכוֹן אֲבַקֵּשׁ מִמֶּנּוּ עַל כָּל זֶה, וְאָמַר לִי, וּפִתְרוֹן הַדְּבָרִים יוֹדִיעֵנִי.
יז  אִלֵּין חֵיוָתָא רַבְרְבָתָא דִּי אִנִּין אַרְבַּע אַרְבְּעָה מַלְכִין יְקוּמוּן מִן-אַרְעָא׃ יז  אֵלֶּה הַחַיּוֹת הַגְּדוֹלוֹת שֶׁהֵן אַרְבַּע, אַרְבָּעָה מְלָכִים יָקוּמוּ מִן הָאָרֶץ.
יח  וִיקַבְּלוּן מַלְכוּתָא קַדִּישֵׁי עֶלְיוֹנִין וְיַחְסְנוּן מַלְכוּתָא עַד-עָלְמָא וְעַד עָלַם עָלְמַיָּא׃ יח  וִיקַבְּלוּ אֶת הַמַּלְכוּת קְדוֹשֵׁי עֶלְיוֹנִים, וְיִירְשׁוּ אֶת הַמַּלְכוּת עַד עוֹלָם וְעַד עוֹלָם עוֹלָמִים.
יט  אֱדַיִן צְבִית לְיַצָּבָא עַל-חֵיוְתָא רְבִיעָיְתָא דִּי-הֲוָת שָׁנְיָה מִן-כלהון (כָּלְּהֵן)  דְּחִילָה יַתִּירָה שניה (שִׁנַּהּ) דִּי-פַרְזֶל וטפריה (וְטִפְרַהּ) דִּי-נְחָשׁ אָכְלָה מַדְּקָה וּשְׁאָרָא ברגליה (בְּרַגְלַהּ) רָפְסָה׃ יט  אֲזַי חָפַצְתִּי וַדָּאוּת עַל הַחַיָּה הָרְבִיעִית, שֶׁהָיְתָה שׁוֹנָה מִכֻּלָּן, נוֹרָאָה יוֹתֵר, שִׁנֶּיהָ שֶׁל בַּרְזֶל וְצִפָּרְנֶיהָ שֶׁל נְחֹשֶׁת, אוֹכֶלֶת מְדִקָּה וְהַשְׁאָר בְּרַגְלָהּ רוֹפֶסֶת.
כ  וְעַל-קַרְנַיָּא עֲשַׂר דִּי בְרֵאשַׁהּ וְאָחֳרִי דִּי סִלְקַת ונפלו (וּנְפַלָה) מִן-קדמיה (קֳדָמַהּ) תְּלָת וְקַרְנָא דִכֵּן וְעַיְנִין לַהּ וּפֻם מְמַלִּל רַבְרְבָן וְחֶזְוַהּ רַב מִן-חַבְרָתַהּ׃ כ  וְעַל עֶשֶׂר הַקַּרְנַיִם אֲשֶׁר בְּרֹאשָׁהּ, וְאַחֶרֶת אֲשֶׁר עָלְתָה, וְנָפְלוּ מִפָּנֶיהָ שָׁלֹשׁ, וְהַקֶּרֶן הַהִיא וְעֵינַיִם לָהּ, וּפֶה מְדַבֵּר גְּדוֹלוֹת, וּמַרְאֶהָ גָּדוֹל מֵחַבְרוֹתֶיהָ.
כא  חָזֵה הֲוֵית וְקַרְנָא דִכֵּן עָבְדָא קְרָב עִם-קַדִּישִׁין וְיָכְלָה לְהֹן׃ כא  רוֹאֶה הָיִיתִי, וְהַקֶּרֶן הַהִיא עוֹשָׂה קְרָב עִם קְדוֹשִׁים, וִיכוֹלָה לָהֶם.
כב  עַד דִּי-אֲתָה עַתִּיק יוֹמַיָּא וְדִינָא יְהִב לְקַדִּישֵׁי עֶלְיוֹנִין וְזִמְנָא מְטָה וּמַלְכוּתָא הֶחֱסִנוּ קַדִּישִׁין׃ כב  עַד אֲשֶׁר בָּא עַתִּיק הַיָּמִים, וְדִין נִתַּן לִקְדוֹשֵׁי עֶלְיוֹנִים, וְהַזְּמָן הִגִּיעַ, וְאֶת הַמַּלְכוּת יָרְשׁוּ קְדוֹשִׁים.
כג  כֵּן אֲמַר חֵיוְתָא רְבִיעָיְתָא מַלְכוּ רביעיה (רְבִיעָאָה) תֶּהֱוֵא בְאַרְעָא דִּי תִשְׁנֵא מִן-כָּל-מַלְכְוָתָא וְתֵאכֻל כָּל-אַרְעָא וּתְדוּשִׁנַּהּ וְתַדְּקִנַּהּ׃ כג  כֵּן אָמַר: הַחַיָּה הָרְבִיעִית -- מַלְכוּת רְבִיעִית תִּהְיֶה בָּאָרֶץ, אֲשֶׁר תִּשְׁנֶה מִכָּל הַמַּלְכֻיּוֹת, וְתֹאכַל אֶת כָּל הָאָרֶץ, וּתְדוּשֶׁנָּה וּתְדִקֶּנָּה.
כד  וְקַרְנַיָּא עֲשַׂר מִנַּהּ מַלְכוּתָא עַשְׂרָה מַלְכִין יְקֻמוּן וְאָחֳרָן יְקוּם אַחֲרֵיהֹן וְהוּא יִשְׁנֵא מִן-קַדְמָיֵא וּתְלָתָה מַלְכִין יְהַשְׁפִּל׃ כד  וְעֶשֶׂר הַקַּרְנַיִם מִמֶּנָּה, הַמַּלְכוּת, עֲשָׂרָה מְלָכִים יָקוּמוּ, וְאַחֵר יָקוּם אַחֲרֵיהֶם, וְהוּא יִשְׁנֶה מִן הָרִאשׁוֹנִים, וּשְׁלֹשֶׁת מְלָכִים יַשְׁפִּיל.
כה  וּמִלִּין לְצַד עליא (עִלָּאָה) יְמַלִּל וּלְקַדִּישֵׁי עֶלְיוֹנִין יְבַלֵּא וְיִסְבַּר לְהַשְׁנָיָה זִמְנִין וְדָת וְיִתְיַהֲבוּן בִּידֵהּ עַד-עִדָּן וְעִדָּנִין וּפְלַג עִדָּן׃ כה  וּדְבָרִים לְצַד הָעֶלְיוֹן יְדַבֵּר, וְאֶת קְדוֹשֵׁי הָעֶלְיוֹנִים יְעַנֶּה וְיַחְשֹׁב לְשַׁנּוֹת זְמַנִּים וְדָת, וְיִנָּתְנוּ בְיָדוֹ עַד מוֹעֵד וּמוֹעֲדִים וַחֲצִי מוֹעֵד.
כו  וְדִינָא יִתִּב וְשָׁלְטָנֵהּ יְהַעְדּוֹן לְהַשְׁמָדָה וּלְהוֹבָדָה עַד-סוֹפָא׃ כו  וְהַדִּין יֵשֵׁב, וְשִׁלְטוֹנוֹ יַעֲבִירוּ, לְהַשְׁמִיד וּלְהַאֲבִיד עַד הַסּוֹף.
כז  וּמַלְכוּתָא וְשָׁלְטָנָא וּרְבוּתָא דִּי מַלְכְוָת תְּחוֹת כָּל-שְׁמַיָּא יְהִיבַת לְעַם קַדִּישֵׁי עֶלְיוֹנִין מַלְכוּתֵהּ מַלְכוּת עָלַם וְכֹל שָׁלְטָנַיָּא לֵהּ יִפְלְחוּן וְיִשְׁתַּמְּעוּן׃ כז  וְהַמַּלְכוּת וְהַשִּׁלְטוֹן וְהַגְּדֻלָּה שֶׁל הַמַּלְכֻיּוֹת תַּחַת כָּל הַשָׁמַיִם, נִתְּנוּ לְעַם קְדוֹשֵׁי עֶלְיוֹנִים, מַלְכוּתוֹ מַלְכוּת עוֹלָם, וְכָל הַשִּׁלְטוֹנוֹת לוֹ יַעַבְדוּ וְיִשָׁמֵעוּ.
כח  עַד-כָּה סוֹפָא דִי-מִלְּתָא אֲנָה דָנִיֵּאל שַׂגִּיא רַעְיוֹנַי יְבַהֲלֻנַּנִי וְזִיוַי יִשְׁתַּנּוֹן עֲלַי וּמִלְּתָא בְּלִבִּי נִטְרֵת׃   פ כח  עַד כָּאן סוֹפוֹ שֶׁל הַדָּבָר; אֲנִי דָּנִיֵּאל, מְאֹד רַעְיוֹנַי יְבַהֲלוּנִי וְזִיוִי יִשְׁתַּנֶּה עָלַי, וְאֶת הַדָּבָר בְּלִבִּי שָׁמַרְתִּי.   פ
