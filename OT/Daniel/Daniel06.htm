לפרק הקודם * * * תוכן העניינים

דניאל פרק ו

א  וְדָרְיָוֶשׁ מדיא (מָדָאָה) קַבֵּל מַלְכוּתָא כְּבַר שְׁנִין שִׁתִּין וְתַרְתֵּין׃ א  וְדָרְיָוֶשׁ הַמָּדִי קִבֵּל אֶת הַמַּלְכוּת כְּבֶן שָׁנִים שִׁשִּׁים וּשְׁתַּיִם.
ב  שְׁפַר קֳדָם דָּרְיָוֶשׁ וַהֲקִים עַל-מַלְכוּתָא לַאֲחַשְׁדַּרְפְּנַיָּא מְאָה וְעֶשְׂרִין דִּי לֶהֱוֹן בְּכָל-מַלְכוּתָא׃ ב  טוֹב הָיָה לִפְנֵי דָרְיָוֶשׁ, וְהֵקִים עַל הַמַּלְכוּת אֶת הָאֲחַשְׁדַּרְפְּנִים מֵאָה וְעֶשְׂרִים, אֲשֶׁר יִהְיוּ בְּכָל הַמַּלְכוּת.
ג  וְעֵלָּא מִנְּהוֹן סָרְכִין תְּלָתָה דִּי דָנִיֵּאל חַד-מִנְּהוֹן  דִּי-לֶהֱוֹן אֲחַשְׁדַּרְפְּנַיָּא אִלֵּין יָהֲבִין לְהוֹן טַעְמָא וּמַלְכָּא לָא-לֶהֱוֵא נָזִק׃ ג  וּלְמַעְלָה מֵהֶם וְזִירִים שְׁלֹשָׁה, אֲשֶׁר דָּנִיֵּאל אֶחָד מֵהֶם, אֲשֶׁר יִהְיוּ הָאֲחַשְׁדַּרְפְּנִים הָאֵלֶּה נוֹתְנִים לָהֶם עֵצָה, וְהַמֶּלֶךְ לֹא יִהְיֶה נִזָּק.
ד  אֱדַיִן דָּנִיֵּאל דְּנָה הֲוָא מִתְנַצַּח עַל-סָרְכַיָּא וַאֲחַשְׁדַּרְפְּנַיָּא  כָּל-קֳבֵל דִּי רוּחַ יַתִּירָא בֵּהּ וּמַלְכָּא עֲשִׁית לַהֲקָמוּתֵהּ עַל-כָּל-מַלְכוּתָא׃ ד  אֲזַי דָּנִיֵּאל זֶה הָיָה גּוֹבֵר עַל הַוְּזִירִים וְהָאֲחַשְׁדַּרְפְּנִים, כָּל-עֻמַּת שֶׁרוּחַ יְתֵרָה בּוֹ, וְהַמֶּלֶךְ חָשַׁב לַהֲקִימוֹ עַל כָּל הַמַּלְכוּת.
ה  אֱדַיִן סָרְכַיָּא וַאֲחַשְׁדַּרְפְּנַיָּא הֲווֹ בָעַיִן עִלָּה לְהַשְׁכָּחָה לְדָנִיֵּאל מִצַּד מַלְכוּתָא וְכָל-עִלָּה וּשְׁחִיתָה לָא-יָכְלִין לְהַשְׁכָּחָה כָּל-קֳבֵל דִּי-מְהֵימַן הוּא וְכָל-שָׁלוּ וּשְׁחִיתָה לָא הִשְׁתְּכַחַת עֲלוֹהִי׃ ה  אֲזַי הַוְּזִירִים וְהָאֲחַשְׁדַּרְפְּנִים הָיוּ מְבַקְּשִׁים עֲלִילָה לִמְצֹא לְדָנִיֵּאל מִצַּד הַמַּלְכוּת, וְכָל עֲלִילָה וּמָשְׁחָת אֵינָם יְכוֹלִים לִמְצֹא, כָּל-עֻמַּת שֶׁנֶּאֱמָן הוּא וְכָל מִשְׁגֶּה וּמָשְׁחָת לֹא נִמְצָא אֶצְלוֹ.
ו  אֱדַיִן גֻּבְרַיָּא אִלֵּךְ אָמְרִין דִּי לָא נְהַשְׁכַּח לְדָנִיֵּאל דְּנָה כָּל-עִלָּה לָהֵן הַשְׁכַּחְנָא עֲלוֹהִי בְּדָת אֱלָהֵהּ׃   ס ו  אֲזַי הָאֲנָשִׁים הָאֵלֶּה אוֹמְרִים, שֶׁלֹּא נִמְצָא לְדָנִיֵּאל זֶה כָּל עֲלִילָה, אֶלָּא אִם מָצָאנוּ אֶצְלוֹ בְּדָת אֱלֹהָיו.   ס

ז  אֱדַיִן סָרְכַיָּא וַאֲחַשְׁדַּרְפְּנַיָּא אִלֵּן הַרְגִּשׁוּ עַל-מַלְכָּא וְכֵן אָמְרִין לֵהּ דָּרְיָוֶשׁ מַלְכָּא לְעָלְמִין חֱיִי׃ ז  אֲזַי הַוְּזִירִים וְהָאֲחַשְׁדַּרְפְּנִים הָאֵלֶּה רָגְשׁוּ אֶל הַמֶּלֶךְ, וְכֵן אוֹמְרִים לוֹ: דָּרְיָוֶשׁ הַמֶּלֶךְ, לְעוֹלָמִים חֱיֵה.
ח  אִתְיָעַטוּ כֹּל סָרְכֵי מַלְכוּתָא סִגְנַיָּא וַאֲחַשְׁדַּרְפְּנַיָּא הַדָּבְרַיָּא וּפַחֲוָתָא לְקַיָּמָה קְיָם מַלְכָּא וּלְתַקָּפָה אֱסָר  דִּי כָל-דִּי-יִבְעֵא בָעוּ מִן-כָּל-אֱלָהּ וֶאֱנָשׁ עַד-יוֹמִין תְּלָתִין לָהֵן מִנָּךְ מַלְכָּא יִתְרְמֵא לְגֹב אַרְיָוָתָא׃ ח  הִתְיָעֲצוּ כָּל וְזִירֵי הַמַּלְכוּת, הַסְּגָנִים וְהָאֲחַשְׁדַּרְפְּנִים, הַיּוֹעֲצִים וְהַפַּחוֹת לְהָקִים קִיּוּם הַמֶּלֶךְ וְלָתֵת תֹּקֶף לְאִסָּר, שֶׁכָּל אֲשֶׁר יְבַקֵּשׁ בַּקָּשָׁה מִכָּל אֱלוֹהַּ וְאָדָם עַד יָמִים שְׁלשִׁים חוּץ מִמְּךָ הַמֶּלֶךְ, יָשְׁלַךְ לְגֹב הָאֲרָיוֹת.
ט  כְּעַן מַלְכָּא תְּקִים אֱסָרָא וְתִרְשֻׁם כְּתָבָא  דִּי לָא לְהַשְׁנָיָה כְּדָת-מָדַי וּפָרַס דִּי-לָא תֶעְדֵּא׃ ט  כָּעֵת הַמֶּלֶךְ תָּקִים אֶת הָאִסָּר וְתִרְשֹׁם אֶת הַכְּתָב, שֶׁלֹּא לְשַׁנּוֹת, כְּדָת מָדַי וּפָרַס אֲשֶׁר לֹא תַעֲבֹר.
י  כָּל-קֳבֵל דְּנָה מַלְכָּא דָּרְיָוֶשׁ רְשַׁם כְּתָבָא וֶאֱסָרָא׃ י  כָּל-עֻמַּת זֶה הַמֶּלֶךְ דָּרְיָוֶשׁ רָשַׁם הַכְּתָב וְהָאִסָּר.
יא  וְדָנִיֵּאל כְּדִי יְדַע דִּי-רְשִׁים כְּתָבָא עַל לְבַיְתֵהּ וְכַוִּין פְּתִיחָן לֵהּ בְּעִלִּיתֵהּ נֶגֶד יְרוּשְׁלֶם וְזִמְנִין תְּלָתָה בְיוֹמָא הוּא בָּרֵךְ עַל-בִּרְכוֹהִי וּמְצַלֵּא וּמוֹדֵא קֳדָם אֱלָהֵהּ כָּל-קֳבֵל דִּי-הֲוָא עָבֵד מִן-קַדְמַת דְּנָה׃   ס יא  וְדָנִיֵּאל, כַּאֲשֶׁר יָדַע שֶׁנִּרְשַׁם הַכְּתָב, נִכְנַס לְבֵיתוֹ, וְחַלּוֹנוֹת פְּתוּחִים לוֹ בַּעֲלִיָּתוֹ נֶגֶד יְרוּשָׁלַיִם, וּזְמַנִּים שְׁלשָׁה בַּיּוֹם הוּא כוֹרֵעַ עַל בִּרְכָּיו, וּמִתְפַּלֵּל וּמוֹדֶה לִפְנֵי אֱלֹהָיו, כָּל-עֻמַּת שֶׁהָיָה עוֹשֶׂה מִלִּפְנֵי זֶה.   ס

יב  אֱדַיִן גֻּבְרַיָּא אִלֵּךְ הַרְגִּשׁוּ וְהַשְׁכַּחוּ לְדָנִיֵּאל בָּעֵה וּמִתְחַנַּן קֳדָם אֱלָהֵהּ׃ יב  אֲזַי הָאֲנָשִׁים הָאֵלֶּה רָגְשׁוּ וּמָצְאוּ אֶת דָּנִיֵּאל, מְבַקֵּשׁ וּמִתְחַנֵּן לִפְנֵי אֱלֹהָיו.
יג  בֵּאדַיִן קְרִבוּ וְאָמְרִין קֳדָם-מַלְכָּא עַל-אֱסָר מַלְכָּא הֲלָא אֱסָר רְשַׁמְתָּ דִּי כָל-אֱנָשׁ דִּי-יִבְעֵא מִן-כָּל-אֱלָהּ וֶאֱנָשׁ עַד-יוֹמִין תְּלָתִין לָהֵן מִנָּךְ מַלְכָּא יִתְרְמֵא לְגוֹב אַרְיָוָתָא עָנֵה מַלְכָּא וְאָמַר יַצִּיבָא מִלְּתָא כְּדָת-מָדַי וּפָרַס דִּי-לָא תֶעְדֵּא׃ יג  אֲזַי קָרְבוּ וְאוֹמְרִים לִפְנֵי הַמֶּלֶךְ עַל אִסַּר הַמֶּלֶךְ: הֲלֹא אִסָּר רָשַׁמְתָּ, אֲשֶׁר כָּל אִישׁ שֶׁיְּבַקֵּשׁ מִכָּל אֱלוֹהַּ וְאָדָם עַד יָמִים שְׁלשִׁים, חוּץ מִמְּךָ הַמֶּלֶךְ, יָשְׁלָךְ לְגֹב הָאֲרָיוֹת; עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר: נָכוֹן הַדָּבָר כְּדָת מָדַי וּפָרַס אֲשֶׁר לֹא תַעֲבֹר.
יד  בֵּאדַיִן עֲנוֹ וְאָמְרִין קֳדָם מַלְכָּא דִּי דָנִיֵּאל דִּי מִן-בְּנֵי גָלוּתָא דִּי יְהוּד לָא-שָׂם עליך (עֲלָךְ) מַלְכָּא טְעֵם וְעַל-אֱסָרָא דִּי רְשַׁמְתָּ וְזִמְנִין תְּלָתָה בְּיוֹמָא בָּעֵא בָּעוּתֵהּ׃ יד  אֲזַי עָנוּ וְאוֹמְרִים לִפְנֵי הַמֶּלֶךְ, שֶׁדָּנִיֵּאל אֲשֶׁר מִבְּנֵי גָּלוּת יְהוּדָה, לֹא שָׂם עָלֶיךָ הַמֶּלֶךְ לֵב וְעַל הָאִסָּר אֲשֶׁר רָשַׁמְתָּ, וּזְמַנִּים שְׁלשָׁה בַּיּוֹם מְבַקֵּשׁ בַּקָּשָׁתוֹ.
טו  אֱדַיִן מַלְכָּא כְּדִי מִלְּתָא שְׁמַע שַׂגִּיא בְּאֵשׁ עֲלוֹהִי וְעַל דָּנִיֵּאל שָׂם בָּל לְשֵׁיזָבוּתֵהּ וְעַד מֶעָלֵי שִׁמְשָׁא הֲוָא מִשְׁתַּדַּר לְהַצָּלוּתֵהּ׃ טו  אֲזַי הַמֶּלֶךְ, כַּאֲשֶׁר אֶת הַדָּבָר שָׁמַע, מְאֹד רַע עָלָיו, וְעַל דָּנִיֵּאל שָׂם דַּעַת לְהַצִּילוֹ, וְעַד מְבוֹאֵי הַשֶּמֶשׁ הָיָה מִתְאַמֵּץ לְהַצִּילוֹ.
טז  בֵּאדַיִן גֻּבְרַיָּא אִלֵּךְ הַרְגִּשׁוּ עַל-מַלְכָּא וְאָמְרִין לְמַלְכָּא דַּע מַלְכָּא דִּי-דָת לְמָדַי וּפָרַס דִּי-כָל-אֱסָר וּקְיָם דִּי-מַלְכָּא יְהָקֵים לָא לְהַשְׁנָיָה׃ טז  אֲזַי הָאֲנָשִׁים הָאֵלֶּה רָגְשׁוּ אֶל הַמֶּלֶךְ וְאוֹמְרִים לַמֶּלֶךְ: דַּע, הַמֶּלֶךְ, שֶׁדָּת לְמָדַי וּפָרַס, שֶׁכָּל אִסּוּר וְקִיּוּם שֶׁהַמֶּלֶךְ יָקִים אֵין לְשַׁנּוֹת.
יז  בֵּאדַיִן מַלְכָּא אֲמַר וְהַיְתִיו לְדָנִיֵּאל וּרְמוֹ לְגֻבָּא דִּי אַרְיָוָתָא עָנֵה מַלְכָּא וְאָמַר לְדָנִיֵּאל אֱלָהָךְ דִּי אנתה (אַנְתְּ) פָּלַח-לֵהּ בִּתְדִירָא הוּא יְשֵׁיזְבִנָּךְ׃ יז  אֲזַי הַמֶּלֶךְ צִוָּה וְהֵבִיאוּ אֶת דָּנִיֵּאל, וְהִשְׁלִיכוּ לְגֹב הָאֲרָיוֹת; עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר לְדָנִיֵּאל: אֱלֹהֶיךָ אֲשֶׁר אַתָּה עוֹבֵד אוֹתוֹ תָּמִיד, הוּא יַצִּילֶךָ.
יח  וְהֵיתָיִת אֶבֶן חֲדָה וְשֻׂמַת עַל-פֻּם גֻּבָּא וְחַתְמַהּ מַלְכָּא בְּעִזְקְתֵהּ וּבְעִזְקָת רַבְרְבָנוֹהִי דִּי לָא-תִשְׁנֵא צְבוּ בְּדָנִיֵּאל׃ יח  וְהוּבְאָה אֶבֶן אַחַת וְהוּשְׂמָה עַל פִּי הַגֹּב, וַחֲתָמָהּ הַמֶּלֶךְ בְּטַבַּעְתּוֹ וּבְטַבַּעַת שָׂרָיו, שֶׁלֹּא יְשַׁנֶּה דָּבָר בְּדָנִיֵּאל.
יט  אֱדַיִן אֲזַל מַלְכָּא לְהֵיכְלֵהּ וּבָת טְוָת וְדַחֲוָן לָא-הַנְעֵל קָדָמוֹהִי וְשִׁנְתֵּהּ נַדַּת עֲלוֹהִי׃ יט  אֲזַי הָלַךְ הַמֶּלֶךְ לְהֵיכָלוֹ וְלָן בְּצוֹם, וְסַמִּים לֹא הִכְנִיס לְפָנָיו, וּשְׁנָתוֹ נָדְדָה עָלָיו.
כ  בֵּאדַיִן מַלְכָּא בִּשְׁפַרְפָּרָא יְקוּם בְּנָגְהָא וּבְהִתְבְּהָלָה לְגֻבָּא דִי-אַרְיָוָתָא אֲזַל׃ כ  אֲזַי הַמֶּלֶךְ עִם שַׁחַר יָקוּם בַּבֹּקֶר, וּבְהִתְבַּהֲלוּת אֶל גֹּב הָאֲרָיוֹת הָלַךְ.
כא  וּכְמִקְרְבֵהּ לְגֻבָּא לְדָנִיֵּאל בְּקָל עֲצִיב זְעִק עָנֵה מַלְכָּא וְאָמַר לְדָנִיֵּאל דָּנִיֵּאל עֲבֵד אֱלָהָא חַיָּא אֱלָהָךְ דִּי אנתה (אַנְתְּ) פָּלַח-לֵהּ בִּתְדִירָא הַיְכִל לְשֵׁיזָבוּתָךְ מִן-אַרְיָוָתָא׃ כא  וּבְקָרְבוֹ לַגֹּב אֶל דָּנִיֵּאל, בְּקוֹל עָצוּב זָעַק; עוֹנֶה הַמֶּלֶךְ וְאוֹמֵר לְדָנִיֵּאל: דָּנִיֵּאל עֶבֶד אֵל חַי, אֱלֹהֶיךָ אֲשֶׁר אַתָּה עוֹבֵד אוֹתוֹ תָּמִיד, הֲיָכוֹל לְהַצִּילְךָ מִן הָאֲרָיוֹת?
כב  אֱדַיִן דָּנִיֵּאל עִם-מַלְכָּא מַלִּל  מַלְכָּא לְעָלְמִין חֱיִי׃ כב  אֲזַי דָּנִיֵּאל עִם הַמֶּלֶךְ דִּבֵּר: הַמֶּלֶךְ לְעוֹלָמִים חֱיֵה!
כג  אֱלָהִי שְׁלַח מַלְאֲכֵהּ וּסְגַר פֻּם אַרְיָוָתָא וְלָא חַבְּלוּנִי כָּל-קֳבֵל דִּי קָדָמוֹהִי זָכוּ הִשְׁתְּכַחַת לִי וְאַף קדמיך (קָדָמָךְ) מַלְכָּא חֲבוּלָה לָא עַבְדֵת׃ כג  אֱלֹהַי שָׁלַח מַלְאָכוֹ וְסָגַר אֶת פִּי הָאֲרָיוֹת וְלֹא חִבְּלוּנִי, כָּל-עֻמַּת שֶׁלְּפָנָיו זְכוּת נִמְצְאָה לִי, וְאַף לְפָנֶיךָ הַמֶּלֶךְ חִבּוּל לֹא עָשִׂיתִי.
כד  בֵּאדַיִן מַלְכָּא שַׂגִּיא טְאֵב עֲלוֹהִי וּלְדָנִיֵּאל אֲמַר לְהַנְסָקָה מִן-גֻּבָּא וְהֻסַּק דָּנִיֵּאל מִן-גֻּבָּא וְכָל-חֲבָל לָא-הִשְׁתְּכַח בֵּהּ דִּי הֵימִן בֵּאלָהֵהּ׃ כד  אֲזַי הַמֶּלֶךְ מְאֹד יָטַב עָלָיו, וְאֶת דָּנִיֵּאל צִוָּה לְהַעֲלוֹת מִן הַגֹּב, וְהָעֳלָה דָנִיֵּאל מִן הַגֹּב, וְכָל חַבָּלָה לֹא נִמְצְאָה בוֹ, שֶׁהֶאֱמִין בֵּאלֹהָיו.
כה  וַאֲמַר מַלְכָּא וְהַיְתִיו גֻּבְרַיָּא אִלֵּךְ דִּי-אֲכַלוּ קַרְצוֹהִי דִּי דָנִיֵּאל וּלְגוֹב אַרְיָוָתָא רְמוֹ אִנּוּן בְּנֵיהוֹן וּנְשֵׁיהוֹן וְלָא-מְטוֹ לְאַרְעִית גֻּבָּא עַד דִּי-שְׁלִטוּ בְהוֹן אַרְיָוָתָא וְכָל-גַּרְמֵיהוֹן הַדִּקוּ׃ כה  וְצִוָּה הַמֶּלֶךְ, וְהֵבִיאוּ אֶת הָאֲנָשִׁים הָאֵלֶּה, אֲשֶׁר הִלְשִׁינוּ עַל דָּנִיֵּאל, וּלְגֹב הָאֲרָיוֹת הִשְׁלִיכוּ אוֹתָם, אֶת בְּנֵיהֶם וּנְשֵׁיהֶם, וְלֹא הִגִּיעוּ לְקַרְקֵעַ הַגֹּב, עַד אֲשֶׁר שָׁלְטוּ בָהֶם הָאֲרָיוֹת וְאֶת כָּל עַצְמוֹתֵיהֶם הֵדֵקּוּ.
כו  בֵּאדַיִן דָּרְיָוֶשׁ מַלְכָּא כְּתַב לְכָל-עַמְמַיָּא אֻמַּיָּא וְלִשָּׁנַיָּא דִּי-דארין (דָיְרִין) בְּכָל-אַרְעָא שְׁלָמְכוֹן יִשְׂגֵּא׃ כו  אֲזַי דָּרְיָוֶשׁ הַמֶּלֶךְ כָּתַב לְכָל הָעַמִּים, הָאֻמּוֹת וְהַלְּשׁוֹנוֹת שֶׁדָּרִים בְּכָל הָאָרֶץ: שְׁלוֹמְכֶם יִרְבֶּה!
כז  מִן-קֳדָמַי שִׂים טְעֵם דִּי בְּכָל-שָׁלְטָן מַלְכוּתִי לֶהֱוֹן זאעין (זָיְעִין) וְדָחֲלִין מִן-קֳדָם אֱלָהֵהּ דִּי-דָנִיֵּאל  דִּי-הוּא אֱלָהָא חַיָּא וְקַיָּם לְעָלְמִין וּמַלְכוּתֵהּ דִּי-לָא תִתְחַבַּל וְשָׁלְטָנֵהּ עַד-סוֹפָא׃ כז  מִלְּפָנַי הוּשַׂם צַו, אֲשֶׁר בְּכָל שִׁלְטוֹן מַלְכוּתִי יִהְיוּ זָעִים וִירֵאִים מִלִּפְנֵי אֱלֹהָיו שֶׁל דָּנִיֵּאל, שֶׁהוּא אֵל חַי וְקַיָּם לְעוֹלָמִים, וּמַלְכוּתוֹ לֹא תִּשָׁחֵת, וְשִׁלְטוֹנוֹ עַד הַסּוֹף.
כח  מְשֵׁיזִב וּמַצִּל וְעָבֵד אָתִין וְתִמְהִין בִּשְׁמַיָּא וּבְאַרְעָא  דִּי שֵׁיזִב לְדָנִיֵּאל מִן-יַד אַרְיָוָתָא׃ כח  מוֹשִׁיעַ וּמַצִּיל וְעוֹשֶׂה אוֹתוֹת וְנִפְלָאוֹת בַּשָׁמַיִם וּבָאָרֶץ, אֲשֶׁר הִצִּיל אֶת דָּנִיֵּאל מִיַּד הָאֲרָיוֹת.
כט  וְדָנִיֵּאל דְּנָה הַצְלַח בְּמַלְכוּת דָּרְיָוֶשׁ וּבְמַלְכוּת כּוֹרֶשׁ פרסיא (פָּרְסָאָה)׃   פ כט  וְדָנִיֵּאל זֶה הִצְלִיחַ בְּמַלְכוּת דָּרְיָוֶשׁ וּבְמַלְכוּת כּוֹרֶשׁ הַפַּרְסִי.   פ
