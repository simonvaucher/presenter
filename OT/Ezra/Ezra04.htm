לפרק הקודם * * * תוכן העניינים

עזרא פרק ד

א וַיִּשְׁמְעוּ צָרֵי יְהוּדָה וּבִנְיָמִן  כִּי-בְנֵי הַגּוֹלָה בּוֹנִים הֵיכָל לַיהוָה אֱלֹהֵי יִשְׂרָאֵל׃    ב וַיִּגְּשׁוּ אֶל-זְרֻבָּבֶל וְאֶל-רָאשֵׁי הָאָבוֹת וַיֹּאמְרוּ לָהֶם נִבְנֶה עִמָּכֶם כִּי כָכֶם נִדְרוֹשׁ לֵאלֹהֵיכֶם ולא (וְלוֹ) אֲנַחְנוּ זֹבְחִים מִימֵי אֵסַר חַדֹּן מֶלֶךְ אַשּׁוּר הַמַּעֲלֶה אֹתָנוּ פֹּה׃    ג וַיֹּאמֶר לָהֶם זְרֻבָּבֶל וְיֵשׁוּעַ וּשְׁאָר רָאשֵׁי הָאָבוֹת לְיִשְׂרָאֵל לֹא-לָכֶם וָלָנוּ לִבְנוֹת בַּיִת לֵאלֹהֵינוּ  כִּי אֲנַחְנוּ יַחַד נִבְנֶה לַיהוָה אֱלֹהֵי יִשְׂרָאֵל כַּאֲשֶׁר צִוָּנוּ הַמֶּלֶךְ כּוֹרֶשׁ מֶלֶךְ-פָּרָס׃    ד וַיְהִי עַם-הָאָרֶץ מְרַפִּים יְדֵי עַם-יְהוּדָה ומבלהים (וּמְבַהֲלִים) אוֹתָם לִבְנוֹת׃    ה וְסֹכְרִים עֲלֵיהֶם יוֹעֲצִים לְהָפֵר עֲצָתָם כָּל-יְמֵי כּוֹרֶשׁ מֶלֶךְ פָּרַס וְעַד-מַלְכוּת דָּרְיָוֶשׁ מֶלֶךְ-פָּרָס׃    ו וּבְמַלְכוּת אֲחַשְׁוֵרוֹשׁ בִּתְחִלַּת מַלְכוּתוֹ כָּתְבוּ שִׂטְנָה עַל-יֹשְׁבֵי יְהוּדָה וִירוּשָׁלִָם׃   ס

ז וּבִימֵי אַרְתַּחְשַׁשְׂתָּא כָּתַב בִּשְׁלָם מִתְרְדָת טָבְאֵל וּשְׁאָר כְּנָוֹתָו עַל-אַרְתַּחְשַׁשְׂתְּא מֶלֶךְ פָּרָס וּכְתָב הַנִּשְׁתְּוָן כָּתוּב אֲרָמִית וּמְתֻרְגָּם אֲרָמִית׃   פ

ח  רְחוּם בְּעֵל-טְעֵם וְשִׁמְשַׁי סָפְרָא כְּתַבוּ אִגְּרָה חֲדָה עַל-יְרוּשְׁלֶם לְאַרְתַּחְשַׁשְׂתְּא מַלְכָּא כְּנֵמָא׃ ח  רְחוּם בַּעַל-פְּקֻדָּה וְשִׁמְשַׁי הַסּוֹפֵר, כָּתְבוּ אִגֶּרֶת אַחַת עַל יְרוּשָׁלַיִם, לְאַרְתַּחְשַׁשְׂתְּא הַמֶּלֶךְ, כֵּן.
ט  אֱדַיִן רְחוּם בְּעֵל-טְעֵם וְשִׁמְשַׁי סָפְרָא וּשְׁאָר כְּנָוָתְהוֹן דִּינָיֵא וַאֲפַרְסַתְכָיֵא טַרְפְּלָיֵא אֲפָרְסָיֵא ארכוי (אַרְכְּוָיֵא) בָבְלָיֵא שׁוּשַׁנְכָיֵא דהוא (דֶּהָיֵא) עֵלְמָיֵא׃ ט  אֲזַי רְחוּם בַּעַל-פְּקֻדָּה וְשִׁמְשַׁי הַסּוֹפֵר וּשְׁאָר חַבְרֵיהֶם, הַדִּינִים וְהַפַּרְסִים, הַנִּטְפָּלִים, הַפַּרְסִים, הָאָרִיִּים, הַבַּבְלִים, הַשּׁוּשַׁנִּים, שֶׁהוּא הָעֵלָמִים.
י  וּשְׁאָר אֻמַּיָּא דִּי הַגְלִי אָסְנַפַּר רַבָּא וְיַקִּירָא וְהוֹתֵב הִמּוֹ בְּקִרְיָה דִּי שָׁמְרָיִן וּשְׁאָר עֲבַר-נַהֲרָה וּכְעֶנֶת׃ י  וּשְׁאָר הָאֻמּוֹת, אֲשֶׁר הֶגְלָה אָסְנַפַּר הַגָּדוֹל וְהַנִּכְבָּד, וְהוֹשִׁיב אוֹתָם בְּקִרְיָה שֶׁל שֹׁמְרוֹן, וּשְׁאָר עֵבֶר הַנָּהָר, וְכָעֵת.
יא  דְּנָה פַּרְשֶׁגֶן אִגַּרְתָּא דִּי שְׁלַחוּ עֲלוֹהִי עַל-אַרְתַּחְשַׁשְׂתְּא מַלְכָּא עבדיך (עַבְדָךְ) אֱנָשׁ עֲבַר-נַהֲרָה וּכְעֶנֶת׃   פ יא  זֶה הֶעְתֵּק הָאִגֶּרֶת, אֲשֶׁר שָׁלְחוּ אֵלָיו אֶל אַרְתַּחְשַׁשְׂתְּא הַמֶּלֶךְ: עֲבָדֶיךָ אִישׁ עֵבֶר הַנָּהָר, וְכָעֵת.   פ

יב  יְדִיעַ לֶהֱוֵא לְמַלְכָּא דִּי יְהוּדָיֵא דִּי סְלִקוּ מִן-לְוָתָךְ עֲלֶינָא אֲתוֹ לִירוּשְׁלֶם קִרְיְתָא מָרָדְתָּא וּבִאישְׁתָּא בָּנַיִן ושורי אשכללו (וְשׁוּרַיָּא שַׁכְלִלוּ) וְאֻשַּׁיָּא יַחִיטוּ׃ יב  יָדוּעַ יְהִי לַמֶּלֶךְ, שֶׁהַיְּהוּדִים אֲשֶׁר עָלוּ מֵאִתְּךָ אֵלֵינוּ, בָּאוּ לִירוּשָׁלַיִם, הַקִּרְיָה הַמּוֹרֶדֶת וְהָרָעָה בּוֹנִים, וְהַחוֹמוֹת הִשְׁלִימוּ וְהַיְּסוֹדוֹת יְחַזֵּקוּ.
יג  כְּעַן יְדִיעַ לֶהֱוֵא לְמַלְכָּא דִּי הֵן קִרְיְתָא דָךְ תִּתְבְּנֵא וְשׁוּרַיָּא יִשְׁתַּכְלְלוּן מִנְדָּה-בְלוֹ וַהֲלָךְ לָא יִנְתְּנוּן וְאַפְּתֹם מַלְכִים תְּהַנְזִק׃ יג  כָּעֵת, יָדוּעַ יְהִי לַמֶּלֶךְ, שֶׁאִם קִרְיָה זוֹ תִּבָּנֶה וְהַחוֹמוֹת יֻשְׁלְמוּ, מַתָּת, מִנְחָה וּמַס לֹא יִתְּנוּ, וְלִבְסוֹף מְלָכִים תַּזִּיק.
יד  כְּעַן כָּל-קֳבֵל דִּי-מְלַח הֵיכְלָא מְלַחְנָא וְעַרְוַת מַלְכָּא לָא אֲרִיךְ-לַנָא לְמֶחֱזֵא עַל-דְּנָה שְׁלַחְנָא וְהוֹדַעְנָא לְמַלְכָּא׃ יד  כָּעֵת, כָּל-עֻמַּת שֶׁמֶּלַח הַהֵיכָל מָלַחְנוּ וְעֶרְוַת הַמֶּלֶךְ לֹא יָאֶה לָנוּ לִרְאוֹת, עַל זֹאת שָׁלַחְנוּ וְהוֹדַעְנוּ לַמֶּלֶךְ.
טו  דִּי יְבַקַּר בִּסְפַר-דָּכְרָנַיָּא דִּי אֲבָהָתָךְ וּתְהַשְׁכַּח בִּסְפַר דָּכְרָנַיָּא וְתִנְדַּע דִּי קִרְיְתָא דָךְ קִרְיָא מָרָדָא וּמְהַנְזְקַת מַלְכִין וּמְדִנָן וְאֶשְׁתַּדּוּר עָבְדִין בְּגַוַּהּ מִן-יוֹמָת עָלְמָא עַל-דְּנָה קִרְיְתָא דָךְ הָחָרְבַת׃ טו  שֶׁיְּבַקֵּר בְּסֵפֶר הַזִּכְרוֹנוֹת שֶׁל אֲבוֹתֶיךָ, וְתִמְצָא בְּסֵפֶר הַזִּכְרוֹנוֹת וְתֵדַע, שֶׁקִּרְיָה זוֹ קִרְיָה מוֹרֶדֶת, וּמַזֶּקֶת מְלָכִים וּמְדִינוֹת, וְאַלִּימוּת עוֹשִׂים בְּתוֹכָהּ מִימוֹת עוֹלָם; עַל זֶה קִרְיָה זוֹ הָחֳרָבָה.
טז  מְהוֹדְעִין אֲנַחְנָה לְמַלְכָּא דִּי הֵן קִרְיְתָא דָךְ תִּתְבְּנֵא וְשׁוּרַיָּה יִשְׁתַּכְלְלוּן  לָקֳבֵל דְּנָה חֲלָק בַּעֲבַר נַהֲרָא לָא אִיתַי לָךְ׃   פ טז  מוֹדִיעִים אֲנַחְנוּ לַמֶּלֶךְ, שֶׁאִם קִרְיָה זוֹ תִּבָּנֶה וְהַחוֹמוֹת יֻשְׁלְמוּ, לְעֻמַּת זֶה חֵלֶק בְּעֵבֶר הַנָּהָר אֵין לָךְ.   פ

יז  פִּתְגָמָא שְׁלַח מַלְכָּא עַל-רְחוּם בְּעֵל-טְעֵם וְשִׁמְשַׁי סָפְרָא וּשְׁאָר כְּנָוָתְהוֹן דִּי יָתְבִין בְּשָׁמְרָיִן וּשְׁאָר עֲבַר-נַהֲרָה שְׁלָם וּכְעֶת׃   ס יז  אֶת הַצַּו שָׁלַח הַמֶּלֶךְ אֶל רְחוּם בַּעַל-פְּקֻדָּה וְשִׁמְשַׁי הַסּוֹפֵר וּשְׁאָר חַבְרֵיהֶם, שֶׁיּוֹשְׁבִים בְּשֹׁמְרוֹן וּשְׁאָר עֵבֶר הַנָּהָר: שָׁלוֹם, וְכָעֵת.   ס

יח  נִשְׁתְּוָנָא דִּי שְׁלַחְתּוּן עֲלֶינָא מְפָרַשׁ קֱרִי קָדָמָי׃ יח  הָאִגֶּרֶת אֲשֶׁר שְׁלַחְתֶּם אֵלֵינוּ, מְפֹרָשׁ נִקְרְאָה לְפָנַי.
יט  וּמִנִּי שִׂים טְעֵם וּבַקַּרוּ וְהַשְׁכַּחוּ דִּי קִרְיְתָא דָךְ מִן-יוֹמָת עָלְמָא עַל-מַלְכִין מִתְנַשְּׂאָה וּמְרַד וְאֶשְׁתַּדּוּר מִתְעֲבֶד-בַּהּ׃ יט  וּמִמֶּנִּי הוּשַׂם צַו, וּבִקְּרוּ וּמָצְאוּ, שֶׁקִּרְיָה זוֹ מִימוֹת עוֹלָם עַל מְלָכִים מִתְנַשֵּׂאת, וּמֶרֶד וְאַלִּימוּת נַעֲשֶׂה בָּהּ.
כ  וּמַלְכִין תַּקִּיפִין הֲווֹ עַל-יְרוּשְׁלֶם וְשַׁלִּיטִין בְּכֹל עֲבַר נַהֲרָה וּמִדָּה בְלוֹ וַהֲלָךְ מִתְיְהֵב לְהוֹן׃ כ  וּמְלָכִים חֲזָקִים הָיוּ עַל יְרוּשָׁלַיִם, וְשׁוֹלְטִים בְּכָל עֵבֶר הַנָּהָר, וּמַתָּת, מִנְחָה וּמַס נִתָּן לָהֶם.
כא  כְּעַן שִׂימוּ טְּעֵם לְבַטָּלָא גֻּבְרַיָּא אִלֵּךְ וְקִרְיְתָא דָךְ לָא תִתְבְּנֵא עַד-מִנִּי טַעְמָא יִתְּשָׂם׃ כא  כָּעֵת, שִׂימוּ צַו לְבַטֵּל הָאֲנָשִׁים הָאֵלֶּה, וְקִרְיָה זוֹ לֹא תִּבָּנֶה, עַד מִמֶּנִּי צַו יוּשּׂם.
כב  וּזְהִירִין הֱווֹ שָׁלוּ לְמֶעְבַּד עַל-דְּנָה לְמָה יִשְׂגֵּא חֲבָלָא לְהַנְזָקַת מַלְכִין׃   ס כב  וּזְהִירִים הֱיוּ מֵעֲשׂוֹת מִשְׁגֶּה עַל זֶה, שֶׁמָּא יִרְבֶּה הַהֶפְסֵד לְהַזָּקַת מְלָכִים.   ס

כג  אֱדַיִן מִן-דִּי פַּרְשֶׁגֶן נִשְׁתְּוָנָא דִּי אַרְתַּחְשַׁשְׂתְּא מַלְכָּא קֱרִי קֳדָם-רְחוּם וְשִׁמְשַׁי סָפְרָא וּכְנָוָתְהוֹן אֲזַלוּ בִבְהִילוּ לִירוּשְׁלֶם עַל-יְהוּדָיֵא וּבַטִּלוּ הִמּוֹ בְּאֶדְרָע וְחָיִל׃   ס כג  אֲזַי מִשּׁהֶעְתֵּק הָאִגֶּרֶת שֶׁל אַרְתַּחְשַׁשְׂתְּא הַמֶּלֶךְ נִקְרָא לִפְנֵי רְחוּם וְשִׁמְשַׁי הַסּוֹפֵר וְחַבְרֵיהֶם, הָלְכוּ בִבְהִילוּת לִירוּשָׁלַיִם אֶל הַיְּהוּדִים, וּבִטְּלוּ אוֹתָם בִּזְרוֹעַ וָחָיִל.   ס

כד  בֵּאדַיִן בְּטֵלַת עֲבִידַת בֵּית-אֱלָהָא דִּי בִּירוּשְׁלֶם וַהֲוָת בָּטְלָא עַד שְׁנַת תַּרְתֵּין לְמַלְכוּת דָּרְיָוֶשׁ מֶלֶךְ-פָּרָס׃   פ כד  אֲזַי בָּטְלָה עֲבוֹדַת בֵּית הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלַיִם, וְהָיְתָה בְּטֵלָה עַד שְׁנַת שְׁתַּיִם לְמַלְכוּת דָּרְיָוֶשׁ מֶלֶךְ פָּרָס.   פ
