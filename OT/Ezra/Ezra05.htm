לפרק הקודם * * * תוכן העניינים

עזרא פרק ה

א  וְהִתְנַבִּי חַגַּי נְבִיָּאה וּזְכַרְיָה בַר-עִדּוֹא נְבִיַּאיָּא עַל-יְהוּדָיֵא דִּי בִיהוּד וּבִירוּשְׁלֶם בְּשֻׁם אֱלָהּ יִשְׂרָאֵל עֲלֵיהוֹן׃   ס א  וְנִבָּא חַגַּי הַנָּבִיא וּזְכַרְיָה בֶּן עִדּוֹא הַנְּבִיאִים, עַל הַיְּהוּדִים אֲשֶׁר בִּיהוּדָה וּבִירוּשָׁלַיִם, בְּשֵׁם אֱלֹהֵי יִשְׂרָאֵל אֲלֵיהֶם.   ס

ב  בֵּאדַיִן קָמוּ זְרֻבָּבֶל בַּר-שְׁאַלְתִּיאֵל וְיֵשׁוּעַ בַּר-יוֹצָדָק וְשָׁרִיו לְמִבְנֵא בֵּית אֱלָהָא דִּי בִירוּשְׁלֶם וְעִמְּהוֹן נְבִיַּאיָּא דִי-אֱלָהָא מְסָעֲדִין לְהוֹן׃   פ ב  אֲזַי קָמוּ זְרֻבָּבֶל בֶּן שְׁאַלְתִּיאֵל וְיֵשׁוּעַ בֶּן יְהוֹצָדָק, וְהֵחֵלּוּ לִבְנוֹת אֶת בֵּית הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלַיִם, וְעִמָּהֶם נְבִיאֵי הָאֱלֹהִים תּוֹמְכִים אוֹתָם.   פ

ג  בֵּהּ-זִמְנָא אֲתָה עֲלֵיהוֹן תַּתְּנַי פַּחַת עֲבַר-נַהֲרָה וּשְׁתַר בּוֹזְנַי וּכְנָוָתְהוֹן וְכֵן אָמְרִין לְהֹם מַן-שָׂם לְכֹם טְעֵם בַּיְתָא דְנָה לִבְּנֵא וְאֻשַּׁרְנָא דְנָה לְשַׁכְלָלָה׃   ס ג  בּוֹ בַּזְּמַן, בָּא אֲלֵיהֶם תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר, וּשְׁתַר בּוֹזְנַי וְחַבְרֵיהֶם, וְכֵן אוֹמְרִים לָהֶם: מִי שּׂם לָכֶם צַו, בַּיִת זֶה לִבְנוֹת וְחוֹמָה זוֹ לְהַשְׁלִים?   ס

ג  בֵּהּ-זִמְנָא אֲתָה עֲלֵיהוֹן תַּתְּנַי פַּחַת עֲבַר-נַהֲרָה וּשְׁתַר בּוֹזְנַי וּכְנָוָתְהוֹן וְכֵן אָמְרִין לְהֹם מַן-שָׂם לְכֹם טְעֵם בַּיְתָא דְנָה לִבְּנֵא וְאֻשַּׁרְנָא דְנָה לְשַׁכְלָלָה׃ ג  בּוֹ בַּזְּמַן, בָּא אֲלֵיהֶם תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר, וּשְׁתַר בּוֹזְנַי וְחַבְרֵיהֶם, וְכֵן אוֹמְרִים לָהֶם: מִי שּׂם לָכֶם צַו, בַּיִת זֶה לִבְנוֹת וְחוֹמָה זוֹ לְהַשְׁלִים?
ד  אֱדַיִן כְּנֵמָא אֲמַרְנָא לְּהֹם  מַן-אִנּוּן שְׁמָהָת גֻּבְרַיָּא דִּי-דְנָה בִנְיָנָא בָּנַיִן׃   פ ד  אֲזַי כֵּן אָמַרְנוּ לָהֶם, מִי הֵם שְׁמוֹת אֲנָשִׁים, שֶׁבּוֹנִים בִּנְיָן זֶה.   פ

ה  וְעֵין אֱלָהֲהֹם הֲוָת עַל-שָׂבֵי יְהוּדָיֵא וְלָא-בַטִּלוּ הִמּוֹ עַד-טַעְמָא לְדָרְיָוֶשׁ יְהָךְ וֶאֱדַיִן יְתִיבוּן נִשְׁתְּוָנָא עַל-דְּנָה׃   פ ה  וְעֵין אֱלֹהֵיהֶם הָיְתָה עַל זִקְנֵי הַיְּהוּדִים, וְלֹא בִּטְּלוּ אוֹתָם עַד שֶׁהַפְּקֻדָּה אֶל דָּרְיָוֶשׁ תֵּלֵךְ, וַאֲזַי יָשִׁיבוּ אִגֶּרֶת עַל זֶה.   פ

ו  פַּרְשֶׁגֶן אִגַּרְתָּא דִּי-שְׁלַח תַּתְּנַי פַּחַת עֲבַר-נַהֲרָה וּשְׁתַר בּוֹזְנַי וּכְנָוָתֵהּ אֲפַרְסְכָיֵא דִּי בַּעֲבַר נַהֲרָה עַל-דָּרְיָוֶשׁ מַלְכָּא׃ ו  הֶעְתֵּק הָאִגֶּרֶת, אֲשֶׁר שָׁלַח תַּתְּנַי פַּחַת עֵבֶר הַנָּהָר וּשְׁתַר בּוֹזְנַי וַחֲבֵרָיו, הַפַּרְסִים אֲשֶׁר בְּעֵבֶר הַנָּהָר, אֶל דָּרְיָוֶשׁ הַמֶּלֶךְ.
ז  אֱדַיִן כְּנֵמָא אֲמַרְנָא לְּהֹם  מַן-אִנּוּן שְׁמָהָת גֻּבְרַיָּא דִּי-דְנָה בִנְיָנָא בָּנַיִן׃   פ ז  הַדָּבָר שָׁלְחוּ אֵלָיו, וְכָזֹאת כָּתוּב בְּתוֹכוֹ: לְדָרְיָוֶשׁ הַמֶּלֶךְ כָּל הַשָׁלוֹם.   פ

ח  יְדִיעַ לֶהֱוֵא לְמַלְכָּא דִּי-אֲזַלְנָא לִיהוּד מְדִינְתָּא לְבֵית אֱלָהָא רַבָּא וְהוּא מִתְבְּנֵא אֶבֶן גְּלָל וְאָע מִתְּשָׂם בְּכֻתְלַיָּא וַעֲבִידְתָּא דָךְ אָסְפַּרְנָא מִתְעַבְדָא וּמַצְלַח בְּיֶדְהֹם׃   ס ח  יָדוּעַ יְהִי לַמֶּלֶךְ, שֶׁהָלַכְנוּ לִיהוּדָה הַמְּדִינָה, לְבֵית הָאֵל הַגָּדוֹל, וְהוּא נִבְנֶה אֶבֶן גָּלָל וְעֵץ מוּשּׂם בַּכְּתָלִים, וַעֲבוֹדָה זוֹ בְּמֶרֶץ נַעֲשֵׂית וּמַצְלִיחָה בְּיָדָם.   ס

ט  אֱדַיִן שְׁאֵלְנָא לְשָׂבַיָּא אִלֵּךְ כְּנֵמָא אֲמַרְנָא לְּהֹם  מַן-שָׂם לְכֹם טְעֵם בַּיְתָא דְנָה לְמִבְנְיָה וְאֻשַּׁרְנָא דְנָה לְשַׁכְלָלָה׃ ט  אֲזַי שָׁאַלְנוּ לַזְּקֵנִים אֵלֶּה, כֵּן אָמַרְנוּ לָהֶם: מִי שּׂם לָכֶם צַו, בַּיִת זֶה לִבְנוֹת וְחוֹמָה זוֹ לְהַשְׁלִים?
י  וְאַף שְׁמָהָתְהֹם שְׁאֵלְנָא לְהֹם לְהוֹדָעוּתָךְ דִּי נִכְתֻּב שֻׁם-גֻּבְרַיָּא דִּי בְרָאשֵׁיהֹם׃   ס י  וְאַף שְׁמוֹתֵיהֶם שָׁאַלְנוּ אוֹתָם לְהוֹדִיעֲךָ, שֶׁנִּכְתֹּב שֵׁם הָאֲנָשִׁים אֲשֶׁר בְּרָאשֵׁיהֶם.   ס

יא  וּכְנֵמָא פִתְגָמָא הֲתִיבוּנָא לְמֵמַר  אֲנַחְנָא הִמּוֹ עַבְדוֹהִי דִי-אֱלָהּ שְׁמַיָּא וְאַרְעָא וּבָנַיִן בַּיְתָא דִּי-הֲוָא בְנֵה מִקַּדְמַת דְּנָה שְׁנִין שַׂגִּיאָן וּמֶלֶךְ לְיִשְׂרָאֵל רַב בְּנָהִי וְשַׁכְלְלֵהּ׃ יא  וְכֵן הַדָּבָר הֱשִׁיבוּנוּ לֵאמֹר: אֲנַחְנוּ הֵם עֲבָדָיו שֶׁל אֱלֹהֵי הַשָׁמַיִם וְהָאָרֶץ, וּבוֹנִים בַּיִת אֲשֶׁר הָיָה בָנוּי מִלִּפְנֵי זֶה שָׁנִים רַבּוֹת, וּמֶלֶךְ גָּדוֹל לְיִשְׂרָאֵל בָּנָהוּ וְהִשְׁלִימוֹ.
יב  לָהֵן מִן-דִּי הַרְגִּזוּ אֲבָהָתַנָא לֶאֱלָהּ שְׁמַיָּא יְהַב הִמּוֹ בְּיַד נְבוּכַדְנֶצַּר מֶלֶךְ-בָּבֶל כסדיא (כַּסְדָּאָה) וּבַיְתָה דְנָה סַתְרֵהּ וְעַמָּה הַגְלִי לְבָבֶל׃   ס יב  אֶלָּא מִשּׁהִכְעִיסוּ אֲבוֹתֵינוּ אֶת אֱלֹהֵי הַשָׁמַיִם, נָתַן אוֹתָם בְּיַד נבוכדנצר מֶלֶךְ בָּבֶל הַכַּשְׂדִּי, וּבַיִת זֶה הֶחֱרִיבוֹ וְאֶת הָעָם הֶגְלָה לְבָבֶל.   ס

יג  בְּרַם בִּשְׁנַת חֲדָה לְכוֹרֶשׁ מַלְכָּא דִּי בָבֶל כּוֹרֶשׁ מַלְכָּא שָׂם טְעֵם בֵּית-אֱלָהָא דְנָה לִבְּנֵא׃ יג  אוּלָם בִּשְׁנַת אַחַת לְכוֹרֶשׁ מֶלֶךְ בָּבֶל, כּוֹרֶשׁ הַמֶּלֶךְ שּׂם צַו, בֵּית אֱלֹהִים זֶה לִבְנוֹת.
יד  וְאַף מָאנַיָּא דִי-בֵית-אֱלָהָא דִּי דַהֲבָה וְכַסְפָּא דִּי נְבוּכַדְנֶצַּר הַנְפֵּק מִן-הֵיכְלָא דִּי בִירוּשְׁלֶם וְהֵיבֵל הִמּוֹ לְהֵיכְלָא דִּי בָבֶל הַנְפֵּק הִמּוֹ כּוֹרֶשׁ מַלְכָּא מִן-הֵיכְלָא דִּי בָבֶל וִיהִיבוּ לְשֵׁשְׁבַּצַּר שְׁמֵהּ דִּי פֶחָה שָׂמֵהּ׃ יד  וְאַף הַכֵּלִים שֶׁל בֵּית הָאֱלֹהִים שֶׁל זָהָב וָכֶסֶף, אֲשֶׁר נְבוּכַדְנֶצַּר הוֹצִיא מִן הַהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם וְהוֹבִיל אוֹתָם לַהֵיכָל שֶׁל בָּבֶל, הוֹצִיא אוֹתָם כּוֹרֶשׁ הַמֶּלֶךְ מִן הַהֵיכָל שֶׁל בָּבֶל, וְנִתְּנוּ לְשֵׁשְׁבַּצַּר שְׁמוֹ, אֲשֶׁר פֶּחָה שָׁמוֹ.
טו  וַאֲמַר-לֵהּ אלה (אֵל) מָאנַיָּא שֵׂא אֵזֶל-אֲחֵת הִמּוֹ בְּהֵיכְלָא דִּי בִירוּשְׁלֶם וּבֵית אֱלָהָא יִתְבְּנֵא עַל-אַתְרֵהּ׃   ס טו  וְאָמַר לוֹ: אֵלֶּה הַכֵּלִים שּׂא, לֵךְ הוֹרֵד אוֹתָם בַּהֵיכָל אֲשֶׁר בִּירוּשָׁלַיִם, וּבֵית הָאֱלֹהִים יִבָּנֶה עַל מְקוֹמוֹ.nbsp;  ס

טז  אֱדַיִן שֵׁשְׁבַּצַּר דֵּךְ אֲתָא יְהַב אֻשַּׁיָּא דִּי-בֵית אֱלָהָא דִּי בִירוּשְׁלֶם וּמִן-אֱדַיִן וְעַד-כְּעַן מִתְבְּנֵא וְלָא שְׁלִם׃ טז  אֲזַי שֵׁשְׁבַּצַּר זֶה בָּא, נָתַן יְסוֹדוֹת שֶׁל בֵּית הָאֱלֹהִים אֲשֶׁר בִּירוּשָׁלַיִם, וּמֵאָז וְעַד כָּעֵת הוּא נִבְנֶה וְלֹא נִשְׁלָם.
יז  וּכְעַן הֵן עַל-מַלְכָּא טָב יִתְבַּקַּר בְּבֵית גִּנְזַיָּא דִּי-מַלְכָּא תַמָּה דִּי בְּבָבֶל הֵן אִיתַי דִּי-מִן-כּוֹרֶשׁ מַלְכָּא שִׂים טְעֵם לְמִבְנֵא בֵּית-אֱלָהָא דֵךְ בִּירוּשְׁלֶם וּרְעוּת מַלְכָּא עַל-דְּנָה יִשְׁלַח עֲלֶינָא׃   ס יז  וְכָעֵת, אִם עַל הַמֶּלֶךְ טוֹב, יְבֻקַּר בְּבֵית הַגְּנָזִים שֶׁל הַמֶּלֶךְ שָׁם אֲשֶׁר בְּבָבֶל, אִם יֵשׁ שֶׁמִּכּוֹרֶשׁ הַמֶּלֶךְ הוּשַׂם צַו לִבְנוֹת בֵּית אֱלֹהִים זֶה בִּירוּשָׁלַיִם, וּרְצוֹן הַמֶּלֶךְ עַל זֶה יִשְׁלַח אֵלֵינוּ.   ס
