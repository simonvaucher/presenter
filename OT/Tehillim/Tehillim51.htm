לפרק הקודם * * * תוכן העניינים

תהילים פרק נא

א   לַמְנַצֵּחַ מִזְמוֹר לְדָוִד׃   
ב   בְּבוֹא-אֵלָיו נָתָן הַנָּבִיא     כַּאֲשֶׁר-בָּא אֶל-בַּת-שָׁבַע׃   
ג   חָנֵּנִי אֱלֹהִים כְּחַסְדֶּךָ    כְּרֹב רַחֲמֶיךָ מְחֵה פְשָׁעָי׃   
ד   הרבה (הֶרֶב) כַּבְּסֵנִי מֵעֲוֹנִי    וּמֵחַטָּאתִי טַהֲרֵנִי׃   
ה   כִּי-פְשָׁעַי אֲנִי אֵדָע    וְחַטָּאתִי נֶגְדִּי תָמִיד׃   
ו   לְךָ לְבַדְּךָ חָטָאתִי    וְהָרַע בְּעֵינֶיךָ עָשִׂיתִי
לְמַעַן תִּצְדַּק בְּדָבְרֶךָ     תִּזְכֶּה בְשָׁפְטֶךָ׃   
ז   הֵן-בְּעָווֹן חוֹלָלְתִּי    וּבְחֵטְא יֶחֱמַתְנִי אִמִּי׃   
ח   הֵן-אֱמֶת חָפַצְתָּ בַטֻּחוֹת    וּבְסָתֻם חָכְמָה תוֹדִיעֵנִי׃   
ט   תְּחַטְּאֵנִי בְאֵזוֹב וְאֶטְהָר    תְּכַבְּסֵנִי וּמִשֶּׁלֶג אַלְבִּין׃   
י   תַּשְׁמִיעֵנִי שָׂשׂוֹן וְשִׂמְחָה    תָּגֵלְנָה עֲצָמוֹת דִּכִּיתָ׃   
יא   הַסְתֵּר פָּנֶיךָ מֵחֲטָאָי    וְכָל-עֲוֹנֹתַי מְחֵה׃   
יב   לֵב טָהוֹר בְּרָא-לִי אֱלֹהִים    וְרוּחַ נָכוֹן חַדֵּשׁ בְּקִרְבִּי׃   
יג   אַל-תַּשְׁלִיכֵנִי מִלְּפָנֶיךָ    וְרוּחַ קָדְשְׁךָ אַל-תִּקַּח מִמֶּנִּי׃   
יד   הָשִׁיבָה לִּי שְׂשׂוֹן יִשְׁעֶךָ    וְרוּחַ נְדִיבָה תִסְמְכֵנִי׃   
טו   אֲלַמְּדָה פֹשְׁעִים דְּרָכֶיךָ    וְחַטָּאִים אֵלֶיךָ יָשׁוּבוּ׃   
טז   הַצִּילֵנִי מִדָּמִים אֱלֹהִים     אֱלֹהֵי תְּשׁוּעָתִי
תְּרַנֵּן לְשׁוֹנִי    צִדְקָתֶךָ׃   
יז   אֲדֹנָי שְׂפָתַי תִּפְתָּח    וּפִי יַגִּיד תְּהִלָּתֶךָ׃   
יח   כִּי לֹא-תַחְפֹּץ זֶבַח וְאֶתֵּנָה    עוֹלָה לֹא תִרְצֶה׃   
יט   זִבְחֵי אֱלֹהִים    רוּחַ נִשְׁבָּרָה
לֵב-נִשְׁבָּר וְנִדְכֶּה     אֱלֹהִים לֹא תִבְזֶה׃   
כ   הֵיטִיבָה בִרְצוֹנְךָ אֶת-צִיּוֹן    תִּבְנֶה חוֹמוֹת יְרוּשָׁלִָם׃   
כא   אָז תַּחְפֹּץ זִבְחֵי-צֶדֶק    עוֹלָה וְכָלִיל
אָז יַעֲלוּ עַל-מִזְבַּחֲךָ    פָרִים׃   

לפרק הבא * * * תוכן העניינים
