פרק כא

מתנתה של אלמנה ענייה

(מרקוס יב' 44-41)

1 הוּא הִתְבּוֹנֵן וְרָאָה אֶת הָעֲשִׁירִים שָׂמִים נִדְבוֹתֵיהֶם בְּתֵבַת הָאוֹצָר.(מלכים ב', יב' י') 2 כִּרְאוֹתוֹ אַלְמָנָה עֲנִיָּה שָׂמָה בּוֹ שְׁתֵּי פְּרוּטוֹת, 3 אָמַר: "אֱמֶת אוֹמֵר אֲנִי לָכֶם, הָאַלְמָנָה הָעֲנִיָּה הַזֹּאת נָתְנָה יוֹתֵר מִכֻּלָּם. 4 כִּי כָּל אֵלֶּה נָתְנוּ אֶת נִדְבוֹתֵיהֶם מִן הָעֹדֶף שֶׁלָּהֶם, אַךְ הִיא מִמַּחְסוֹרָהּ נָתְנָה אֶת כָּל אֲשֶׁר הָיָה לָהּ לְמִחְיָה."

ישוע מנבא חורבן וצרות

(מתי כד' 28-1; מרקוס יג' 23-1)

5 כֵּיוָן שֶׁכַּמָּה מֵהֶם דִּבְּרוּ עַל הַמִּקְדָּשׁ, שֶׁהוּא מְהֻדָּר בַּאֲבָנִים יָפוֹת וּבְמַתָּנוֹת, אָמַר: 6 "הַדְּבָרִים הָאֵלֶּה אֲשֶׁר אַתֶּם רוֹאִים, הִנֵּה יָמִים בָּאִים וְלֹא תִּשָּׁאֵר אֶבֶן עַל אֶבֶן אֲשֶׁר לֹא תֻּפַּל אַרְצָה."(ישעיהו ו' יא'; מיכה ג' יב') 7 שְׁאָלוּהוּ: "רַבִּי, אִם כֵּן מָתַי יִהְיֶה הַדָּבָר הַזֶּה? וּמַהוּ הָאוֹת לִפְנֵי שֶׁזֶּה יִקְרֶה?" 8 הֵשִׁיב וְאָמַר: "הִזָּהֲרוּ שֶׁלֹּא יַתְעוּ אֶתְכֶם, כִּי רַבִּים יָבוֹאוּ בִּשְׁמִי וְיֹאמְרוּ, 'אֲנִי הוּא!' וְ'הָעֵת קְרוֹבָה!' אַל תֵּלְכוּ אַחֲרֵיהֶם. 9 וּבְשָׁמְעֲכֶם מִלְחָמוֹת וּמְהוּמוֹת אַל תִּבָּהֲלוּ, כִּי אֵלֶּה צְרִיכִים לִהְיוֹת בָּרִאשׁוֹנָה, אַךְ לֹא מִיָּד יָבוֹא הַקֵּץ." 10 הוֹסִיף וְאָמַר לָהֶם: "גּוֹי יָקוּם עַל גּוֹי וּמַמְלָכָה עַל מַמְלָכָה,(דברי הימים ב', טו' ו'; ישעיהו יט' ב') 11 וְתִהְיֶינָה רְעִידוֹת אֲדָמָה גְּדוֹלוֹת וּבִמְקוֹמוֹת רַבִּים רָעָב וְדֶבֶר וְגַם מוֹרָאִים וְאוֹתוֹת גְּדוֹלִים מִן הַשָּׁמַיִם. 12 אַךְ לִפְנֵי כָּל אֵלֶּה יִשְׁלְחוּ בָּכֶם אֶת יְדֵיהֶם וְיִרְדְּפוּ וְיִמְסְרוּ אֶתְכֶם לְבָתֵּי כְּנֶסֶת וּלְבָתֵּי כֶּלֶא, וְתוּבְאוּ לִפְנֵי מְלָכִים וּמוֹשְׁלִים לְמַעַן שְׁמִי. 13 זֶה יְבִיאֲכֶם לְמַתַּן עֵדוּת. 14 וְעַתָּה שִׂימוּ לִבְּכֶם שֶׁלֹּא לְהָכִין מֵרֹאשׁ דְּבָרִים לַהֲגַנַּתְכֶם, 15 כִּי אֲנִי אֶתֵּן לָכֶם פֶּה וְחָכְמָה אֲשֶׁר כָּל מִתְנַגְּדֵיכֶם לֹא יוּכְלוּ לַעֲמֹד נֶגְדָּהּ אוֹ לְסָתְרָהּ.(שמות ד' יב') 16 וְגַם תֻּסְגְּרוּ עַל-יְדֵי הוֹרִים וְאַחִים, קְרוֹבִים וַחֲבֵרִים, וְיָמִיתוּ מִכֶּם;(מיכה ז' ו') 17 וְתִהְיוּ שְׂנוּאִים עַל הַכֹּל בִּגְלַל שְׁמִי, 18 אַךְ שַׂעֲרָה מִשְּׂעַר רֹאשְׁכֶם לֹא תֹּאבַד. 19 עִמְדוּ בּתוֹחַלְתְּכֶם וְתִקְנוּ אֶת נַפְשׁוֹתֵיכֶם. 20 וְכַאֲשֶׁר תִּרְאוּ אֶת יְרוּשָׁלַיִם מֻקֶּפֶת מַחֲנוֹת, דְּעוּ כִּי קָרֵב חֻרְבָּנָהּ.(ישעיהו כט' ג'; ירמיהו ו' ו') 21 אֲזַי הַנִּמְצָאִים בִּיהוּדָה -- שֶׁיָּנוּסוּ אֶל הֶהָרִים; הַנִּמְצָאִים בָּעִיר, שֶׁיֵּצְאוּ מִמֶּנָּה, וְאֵלֶּה אֲשֶׁר בַּשָּׂדוֹת אַל יִכָּנְסוּ אֵלֶיהָ; 22 כִּי יְמֵי נָקָם הֵם לְמַלֵּא כָּל הַכָּתוּב.(הושע ט' ז'; דברים לב' לה') 23 אוֹי לֶהָרוֹת וְלַמֵּינִיקוֹת בַּיָּמִים הָהֵם, כִּי צָרָה גְּדוֹלָה תִּהְיֶה עַל הָאָרֶץ וְקֶצֶף עַל הָעָם הַזֶּה. 24 בְּחֶרֶב יִפְּלוּ וְיֻגְלוּ אֶל כָּל הַגּוֹיִים,(דברים כח' סד') וִירוּשָׁלַיִם תִּהְיֶה מִרְמָס לְגוֹיִים עַד אֲשֶׁר יִמָּלְאוּ עִתּוֹת הַגּוֹיִים."(תהילים עט' א'; ישעיהו סג' יח'; זכריה יב' ג')

בואו של בן-האדם

(מתי כד' 44-29; מרקוס יג' 37-24)

25 "וְיִהְיוּ אוֹתוֹת בַּשֶּׁמֶשׁ וּבַיָּרֵחַ וּבַכּוֹכָבִים, וְעַל הָאָרֶץ מְצוּקָה לַגּוֹיִים וּמְבוּכָה מִגַּעַשׁ הַיָּם וְגַלָּיו.(יואל ג' ג') 26 בְּנֵי אָדָם יִתְעַלְּפוּ מֵאֵימָה וּמִצִּפִּיָּה לַבָּאוֹת עַל הָעוֹלָם, כִּי כֹּחוֹת הַשָּׁמַיִם יִזְדַּעְזְעוּ.(ישעיהו לד' ד') 27 אָז יִרְאוּ אֶת בֶּן-הָאָדָם בָּא בֶּעָנָן בִּגְבוּרָה וּבְכָבוֹד רַב.(דניאל ז' יג'-יד') 28 כַּאֲשֶׁר יַתְחִילוּ לִקְרוֹת הַדְּבָרִים הָאֵלֶּה הִתְעוֹדְדוּ וְהָרִימוּ רָאשֵׁיכֶם, כִּי קְרֵבָה גְּאֻלַּתְכֶם." 29 הוּא סִפֵּר לָהֶם מָשָׁל: "רְאוּ אֶת הַתְּאֵנָה וְאֶת כָּל הָעֵצִים. 30 כְּשֶׁאַתֶּם רוֹאִים אוֹתָם מְלַבְלְבִים, יוֹדְעִים אַתֶּם בְּעַצְמְכֶם כִּי קָרוֹב הַקַּיִץ. 31 כֵּן גַּם כְּשֶׁתִּרְאוּ אֶת הַדְּבָרִים הָאֵלֶּה מִתְקַיְּמִים, דְּעוּ כִּי קְרוֹבָה מַלְכוּת הָאֱלֹהִים. 32 אָמֵן. אוֹמֵר אֲנִי לָכֶם, לֹא יַעֲבֹר הַדּוֹר הַזֶּה עַד אֲשֶׁר יִהְיוּ כָּל אֵלֶּה. 33 הַשָּׁמַיִם וְהָאָרֶץ יַעַבְרוּ וּדְבָרַי לֹא יַעַבְרוּ. 34 הִשָּׁמְרוּ לָכֶם שֶׁלֹּא יִכְבַּד לְבַבְכֶם בִּשְׁתִיָּה וּבְשִׁכָּרוֹן וּבְדַאֲגוֹת הַחַיִּים וְהַיּוֹם הַהוּא יָבוֹא עֲלֵיכֶם פִּתְאוֹם;(ישעיהו ה' יא'-יב') 35 כִּי כְּמוֹ פַּח יָבוֹא עַל כָּל הַיּוֹשְׁבִים עַל פְּנֵי כָּל הָאָרֶץ. 36 לָכֵן עִמְדוּ עַל הַמִּשְׁמָר בְּכָל עֵת וְהִתְפַּלְּלוּ שֶׁיִּהְיֶה בְּכֹחֲכֶם לְהִמָּלֵט מִכָּל הָעֲתִידוֹת הָאֵלֶּה וּלְהִתְיַצֵּב לִפְנֵי בֶּן-הָאָדָם." 37 בִּשְׁעוֹת הַיּוֹם הָיָה מְלַמֵּד בְּבֵית הַמִּקְדָּשׁ וּבַלֵּילוֹת הָיָה יוֹצֵא לָלוּן בָּהָר הַנִּקְרָא הַר הַזֵּיתִים. 38 וְכָל הָעָם הִשְׁכִּימוּ לָבוֹא אֵלָיו לְבֵית הַמִּקְדָּשׁ לִשְׁמֹעַ אוֹתוֹ.
