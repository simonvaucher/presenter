לפרק הקודם

הבשורה על-פי לוקס
פרק ב

הולדת ישוע

(מתי א' 25-18)

1 בַּיָּמִים הָהֵם יָצָא צַו מֵאֵת הַקֵּיסָר אוֹגוּסְטוֹס לַעֲרֹךְ רִשּׁוּם תּוֹשָׁבִים בְּכָל הָאֲרָצוֹת. 2 הַמִּפְקָד הַזֶּה, הָרִאשׁוֹן, נֶעֱרַךְ בִּהְיוֹת קִירֶנְיוֹס נְצִיב סוּרְיָה. 3 הַכֹּל הָלְכוּ לְהִתְפַּקֵּד, אִישׁ אִישׁ לְעִירוֹ. 4 וְגַם יוֹסֵף, שֶׁהָיָה מִבֵּית דָּוִד וּמִמִּשְׁפַּחְתּוֹ, עָלָה מִן הַגָּלִיל אֶל יְהוּדָה, מֵהָעִיר נָצְרַת לְעִיר דָּוִד הַנִּקְרֵאת בֵּית לֶחֶם, 5 לְהִתְפַּקֵּד עִם מִרְיָם אֲרוּסָתוֹ וְהִיא הָרָה. 6 כַּאֲשֶׁר הָיוּ שָׁם מָלְאוּ יָמֶיהָ לָלֶדֶת 7 וְהִיא יָלְדָה אֶת בְּנָהּ הַבְּכוֹר.(ישעיהו ט' ה') לְאַחַר שֶׁחִתְּלָה אוֹתוֹ הִשְׁכִּיבָה אוֹתוֹ בְּאֵבוּס, כִּי לֹא הָיָה לָהֶם מָקוֹם בַּמָּלוֹן. 8 בְּאוֹתוֹ חֶבֶל אֶרֶץ הָיוּ רוֹעִים שֶׁנָּהֲגוּ לָגוּר בַּשָּׂדֶה וְלִשְׁמֹר עַל עֶדְרָם בְּאַשְׁמוּרוֹת הַלַּיְלָה. 9 לְפֶתַע נִצַּב עֲלֵיהֶם מַלְאַךְ יהוה וּכְבוֹד יהוה נָגַהּ סְבִיבָם. הֵם נִבְהֲלוּ עַד מְאֹד, 10 אוּלָם הַמַּלְאָךְ אָמַר לָהֶם: "אַל תִּפְחֲדוּ, כִּי הִנְנִי מְבַשֵּׂר לָכֶם שִׂמְחָה גְּדוֹלָה אֲשֶׁר תִּהְיֶה לְכָל הָעָם.(ישעיהו ט' ב') 11 הַיּוֹם נוֹלַד לָכֶם מוֹשִׁיעַ בְּעִיר דָּוִד, הוּא הַמָּשִׁיחַ הָאָדוֹן. 12 וְזֶה לָכֶם הָאוֹת: תִּמְצְאוּ תִּינוֹק מְחֻתָּל שׁוֹכֵב בְּאֵבוּס." 13 פִּתְאוֹם הָיוּ לְיַד הַמַּלְאָךְ הֲמוֹן צְבָא הַשָּׁמַיִם וְהֵמָּה מְהַלְלִים אֶת הָאֱלֹהִים וְאוֹמְרִים:(דניאל ז' י'; תהילים קג' כא') 14 "כָּבוֹד לֵאלֹהִים בַּמְּרוֹמִים, וּבָאָרֶץ שָׁלוֹם עַל בְּנֵי אָדָם רְצוֹנוֹ."(ישעיהו נז' יט') 15 לְאַחַר שֶׁעָלוּ הַמַּלְאָכִים מֵעֲלֵיהֶם הַשָּׁמַיְמָה, אָמְרוּ הָרוֹעִים אִישׁ אֶל רֵעֵהוּ: "בּוֹאוּ נֵלֵךְ עַד בֵּית לֶחֶם וְנִרְאֶה אֶת הַדָּבָר הַזֶּה אֲשֶׁר קָרָה, מַה שֶּׁאֱלֹהִים הוֹדִיעַ לָנוּ." 16 הֵם הָלְכוּ מַהֵר וּמָצְאוּ אֶת מִרְיָם וְיוֹסֵף וְאֶת הַתִּינוֹק הַשּׁוֹכֵב בָּאֵבוּס. 17 כַּאֲשֶׁר רָאוּ אוֹתוֹ הִשְׁמִיעוּ אֶת הַדָּבָר שֶׁנֶּאֱמַר לָהֶם עַל-אוֹדוֹת הַיֶּלֶד הַזֶּה. 18 כָּל הַשּׁוֹמְעִים תָּמְהוּ עַל הַדְּבָרִים שֶׁאָמְרוּ לָהֶם הָרוֹעִים, 19 אַךְ מִרְיָם שָׁמְרָה אֶת הַדְּבָרִים הָאֵלֶּה וְשָׁקְלָה אוֹתָם בְּלִבָּהּ.(בראשית לז' יא') 20 לְאַחַר מִכֵּן חָזְרוּ הָרוֹעִים כְּשֶׁהֵם מְהַלְלִים וּמְשַׁבְּחִים אֶת הָאֱלֹהִים עַל כָּל אֲשֶׁר שָׁמְעוּ וְרָאוּ כְּפִי שֶּׁנֶּאֱמַר לָהֶם. 21 כְּשֶׁמָּלְאוּ שְׁמוֹנָה יָמִים וְהִגִּיעַ הַזְּמַן לָמוּל אוֹתוֹ קָרְאוּ אֶת שְׁמוֹ יֵשׁוּעַ,(ויקרא יב' ג'; בראשית יז' יב') כַּשֵּׁם שֶׁקָּרָא לוֹ הַמַּלְאָךְ לִפְנֵי שֶׁהוֹרָה בַּבֶּטֶן.

פדיון הבן בבית המקדש ושיבה לנצרת

22 כַּאֲשֶׁר מָלְאוּ יְמֵי טָהֳרָתָם לְפִי תּוֹרַת מֹשֶׁה,(ויקרא יב' ב'-ח') הֶעֱלוּהוּ לִירוּשָׁלַיִם לְהַצִּיגוֹ לִפְנֵי אֱלֹהִים 23 כַּכָּתוּב בְּתוֹרַת יהוה, שֶׁכָּל בְּכוֹר פֶּטֶר רֶחֶם קָדוֹשׁ לַיהוה יִקָּרֵא;(שמות יג' ב'; יג' יב'; במדבר ג' יג') 24 וּלְהַקְרִיב קָרְבָּן כָּאָמוּר בְּתוֹרַת יהוה, שְׁתֵּי-תֹרִים אוֹ שְׁנֵי בְּנֵי יוֹנָה.(ויקרא יב' ח') 25 אִישׁ הָיָה בִּירוּשָׁלַיִם בְּאוֹתָהּ עֵת, שִׁמְעוֹן שְׁמוֹ; וְהָאִישׁ צַדִּיק וְחָסִיד וּמְחַכֶּה לְנֶחָמַת יִשְׂרָאֵל,(ישעיהו מט' יג'; נב' ט') וְרוּחַ הַקֹּדֶשׁ הָיְתָה עָלָיו.(ישעיהו סא' א') 26 וּבְרוּחַ הַקֹּדֶשׁ נִגְלָה לוֹ שֶׁלֹּא יִרְאֶה מָוֶת בְּטֶרֶם יִרְאֶה אֶת מְשִׁיחַ יהוה, 27 וּבְהַדְרָכַת הָרוּחַ הוּא נִכְנַס אֶל הַמִּקְדָּשׁ. כַּאֲשֶׁר הֵבִיאוּ הַהוֹרִים אֶת הַיֶּלֶד יֵשׁוּעַ, לַעֲשׂוֹת בּוֹ כַּנָּהוּג עַל-פִּי הַתּוֹרָה, 28 לָקַח אוֹתוֹ בִּזְרוֹעוֹתָיו וּבֵרַךְ אֶת אֱלֹהִים וְאָמַר: 29 "וְעַתָּה, אֲדֹנָי, פְּטֹר נָא אֶת עַבְדְּךָ בְּשָׁלוֹם, כַּאֲשֶׁר דִּבַּרְתָּ, 30 כִּי רָאוּ עֵינַי אֶת יְשׁוּעָתְךָ (ישעיהו מ' ה'; תהילים יח' ג') 31 אֲשֶׁר הֲכִינוֹתָ לְעֵינֵי כָּל הָעַמִּים:(ישעיהו נב' י') 32 אוֹר לְהָאִיר לַגּוֹיִים (ישעיהו מב' ו'; מט' ו') וְתִפְאֶרֶת יִשְׂרָאֵל עַמֶּךָ."(ישעיהו מו' יג') 33 אָבִיו וְאִמּוֹ תָּמְהוּ עַל הַדְּבָרִים שֶׁנֶּאֱמְרוּ עָלָיו. 34 בֵּרְכָם שִׁמְעוֹן וְאָמַר אֶל מִרְיָם אִמּוֹ: "הִנֵּה זֶה נוֹסַד לְמִכְשׁוֹל וְלִתְקוּמָה לְרַבִּים בְּיִשְׂרָאֵל וּלְאוֹת אֲשֶׁר יִתְנַגְּדוּ לוֹ --(ישעיהו ח' יד'-טו') 35 וְגַם בְּלִבֵּךְ אַתְּ תַּעֲבֹר חֶרֶב -- לְמַעַן תִּגָּלֶינָה מַחְשְׁבוֹת לֵב רַבִּים." 36 גַּם אִשָּׁה נְבִיאָה הָיְתָה שָׁם, חַנָּה בַּת פְּנוּאֵל מִשֵּׁבֶט אָשֵׁר, וְהִיא בָּאָה בַּיָּמִים. שֶׁבַע שָׁנִים חָיְתָה עִם בַּעֲלָהּ מֵעֵת בְּתוּלֶיהָ; 37 וְהִיא אַלְמָנָה בַּת שְׁמוֹנִים וְאַרְבַּע שָׁנִים וְלֹא עָזְבָה אֶת הַמִּקְדָּשׁ בְּעָבְדָהּ אֶת אֱלֹהִים בְּצוֹם וְתַחֲנוּנִים יוֹמָם וָלַיְלָה. 38 בְּאוֹתָהּ שָׁעָה בָּאָה וְהוֹדְתָה לֵאלֹהִים וְדִבְּרָה עַל הַיֶּלֶד בְּאָזְנֵי כָּל הַמְחַכִּים לַגְּאֻלָּה בִּירוּשָׁלַיִם. 39 לְאַחַר שֶׁהִשְׁלִימוּ אֶת הַכֹּל עַל-פִּי תּוֹרַת יהוה, חָזְרוּ לַגָּלִיל, לְנָצְרַת עִירָם. 40 וְהַיֶּלֶד גָּדַל וְהִתְחַזֵּק, מָלֵא חָכְמָה,(ישעיהו יא' ב') וְחֶסֶד אֱלֹהִים הָיָה עָלָיו.

הנער ישוע בבית המקדש

41 מִדֵּי שָׁנָה בְּשָׁנָה עָלוּ הוֹרָיו לִירוּשָׁלַיִם בְּחַג הַפֶּסַח. 42 וּבִמְלֹאת לוֹ שְׁתֵּים-עֶשְׂרֵה שָׁנִים עָלוּ לִירוּשָׁלַיִם כְּמִנְהַג הֶחָג.(שמות כג' יד'-יז') 43 כְּשֶׁמָּלְאוּ הַיָּמִים (שמות יב' יח') שָׁבוּ, אַךְ הַנַּעַר יֵשׁוּעַ נִשְׁאַר בִּירוּשָׁלַיִם וְהוֹרָיו לֹא יָדְעוּ. 44 הוֹאִיל וְחָשְׁבוּ שֶׁהוּא נִמְצָא בַּשַּׁיָּרָה הָלְכוּ כִּבְרַת דֶּרֶךְ יוֹם אֶחָד וְחִפְּשׂוּהוּ בֵּין הַקְּרוֹבִים וְהַמַּכָּרִים, 45 אַךְ כֵּיוָן שֶׁלֹּא מְצָאוּהוּ חָזְרוּ לִירוּשָׁלַיִם לְחַפֵּשׂ אוֹתוֹ. 46 לְאַחַר שְׁלוֹשָׁה יָמִים מְצָאוּהוּ בַּמִּקְדָּשׁ וְהוּא יוֹשֵׁב בֵּין הַמּוֹרִים, שׁוֹמֵעַ אוֹתָם וְשׁוֹאֵל אוֹתָם; 47 וְכָל שׁוֹמְעָיו הִתְפַּלְּאוּ עַל שִׂכְלוֹ וְעַל תְּשׁוּבוֹתָיו.(תהילים קיט' צט') 48 כַּאֲשֶׁר רָאוּ אוֹתוֹ הוֹרָיו הִשְׁתּוֹמְמוּ. אָמְרָה לוֹ אִמּוֹ: "בְּנִי, לָמָּה עָשִׂיתָ לָנוּ כָּכָה? הִנֵּה אָבִיךָ וַאֲנִי חִפַּשְׂנוּ אוֹתְךָ בִּדְאָגָה רַבָּה." 49 הֵשִׁיב לָהֶם: "לָמָּה חִפַּשְׂתֶּם אוֹתִי? הַאִם לֹא יְדַעְתֶּם כִּי עָלַי לִהְיוֹת בַּאֲשֶׁר לְאָבִי?"(תהילים כו' ח') 50 אַךְ הֵם לֹא הֵבִינוּ אֶת הַדָּבָר שֶׁדִּבֵּר אֲלֵיהֶם. 51 הוּא יָרַד אִתָּם וּבָא לְנָצְרַת וְהָיָה נִכְנָע לְמָרוּתָם. אֶת כָּל הַדְּבָרִים הָאֵלֶּה שָׁמְרָה אִמּוֹ בְּלִבָּהּ. 52 וְיֵשׁוּעַ הוֹסִיף לִגְדֹּל בְּחָכְמָה וּבְקוֹמָה וּבְחֵן לִפְנֵי אֱלֹהִים וּבְנֵי אָדָם.(שמואל א', ב' כו')
