לאגרת יעקב

אגרת פטרוס הראשונה
פרק א

1 מֵאֵת פֶּטְרוֹס, שְׁלִיחַ יֵשׁוּעַ הַמָּשִׁיחַ, אֶל תּוֹשָׁבֵי תְּפוּצוֹת פּוֹנְטוֹס, גָּלַטְיָה, קַפָּדוֹקְיָה, אַסְיָה וְבִּיתִינְיָה, הַנִּבְחָרִים 2 עַל-פִּי יְדִיעָתוֹ מִקֶּדֶם שֶׁל אֱלֹהִים הָאָב, בְּקִדּוּשׁ הָרוּחַ, אֶל צִיּוּת יֵשׁוּעַ הַמָּשִׁיחַ וְהַזָּיַת דָּמוֹ: חֶסֶד וְשָׁלוֹם יִשְׁפְּעוּ עֲלֵיכֶם לְמַכְבִּיר.

תקווה חיה

3 בָּרוּךְ הָאֱלֹהִים אֲבִי אֲדוֹנֵנוּ יֵשׁוּעַ הַמָּשִׁיחַ אֲשֶׁר בְּרֹב רַחֲמָיו הוֹלִיד אוֹתָנוּ מֵחָדָשׁ לְתִקְוָה חַיָּה בִּתְחִיַּת יֵשׁוּעַ הַמָּשִׁיחַ מִן הַמֵּתִים,(תהילים קג' ח') 4 לְנַחֲלָה אֲשֶׁר לֹא תִּשָּׁחֵת, לֹא תִּטָּמֵא וְלֹא תִּבֹּל, הַצְּפוּנָה לָכֶם בַּשָּׁמַיִם, 5 לָכֶם הַשְּׁמוּרִים בִּגְבוּרַת אֱלֹהִים, בִּזְכוּת אֱמוּנָה, אֱלֵי יְשׁוּעָה הָעֲתִידָה לְהִגָּלוֹת בְּעֵת קֵץ.(תהילים לז' כח') 6 וּבָהּ שְׂמֵחִים אַתֶּם גַּם אִם מִן הַצֹּרֶךְ הוּא שֶׁתִּתְעַצְּבוּ זְמַן מָה בְּכָל מִינֵי מַסּוֹת 7 כְּדֵי שֶׁאֱמוּנַתְכֶם הַצְּרוּפָה וְהַיְקָרָה הַרְבֵּה יוֹתֵר מִזָּהָב, הַנִּשְׁחָת עַל אַף הִצָּרְפוֹ בָּאֵשׁ,(מלאכי ג' ג') תֵּצֵא לִתְהִלָּה וּלְכָבוֹד וּלְתִפְאֶרֶת בְּהִתְגַּלּוֹת יֵשׁוּעַ הַמָּשִׁיחַ; 8 הוּא אֲשֶׁר אַתֶּם אוֹהֲבִים אוֹתוֹ אַף כִּי לֹא רְאִיתֶם אוֹתוֹ, וַאֲשֶׁר כָּעֵת אֵינְכֶם רוֹאִים אוֹתוֹ וּבְכָל זֹאת מַאֲמִינִים אַתֶּם בּוֹ וְשָׂשִׂים בְּשִׂמְחָה מְפֹאָרָה וַעֲצוּמָה מִסַּפֵּר, 9 בְּהַשִּׂיגְכֶם אֶת תַּכְלִית אֱמוּנַתְכֶם, אֶת יְשׁוּעַת נַפְשׁוֹתֵיכֶם. 10 אֶת הַיְשׁוּעָה הַזֹּאת חָקְרוּ וְדָרְשׁוּ הַנְּבִיאִים שֶׁנִּבְּאוּ עַל הַחֶסֶד הַמְיֻעָד לָכֶם. 11 הֵם חָקְרוּ מָה הָעֵת וְהַנְּסִבּוֹת שֶׁהוֹדִיעָה רוּחַ הַמָּשִׁיחַ אֲשֶׁר בְּקִרְבָּם,(דניאל ט' כה'-כו') כְּשֶׁהֵעִידָה מֵרֹאשׁ עַל סִבְלוֹת הַמָּשִׁיחַ וְעַל הַכָּבוֹד וְהַתִּפְאֶרֶת שֶׁיָּבוֹאוּ אַחֲרֵיהֶם.(תהילים כב' ב'-כב'; ישעיהו ס' א'-ז') 12 וְנִגְלָה לָהֶם כִּי לֹא לְמַעַן עַצְמָם אֶלָּא לְמַעַנְכֶם שֵׁרְתוּ בַּדְּבָרִים הָהֵם, אֲשֶׁר כָּעֵת הֻשְׁמְעוּ לָכֶם מִפִּי הַמְבַשְּׂרִים אֶת הַבְּשׂוֹרָה בְּרוּחַ הַקֹּדֶשׁ הַשְּׁלוּחָה מִן הַשָּׁמַיִם, דְּבָרִים אֲשֶׁר מַלְאָכִים נִכְסָפִים לְהַשְׁקִיף אֶל תּוֹכָם.

חיי קדוּשה

13 לָכֵן חִגְרוּ מָתְנֵי שִׂכְלֵיכֶם, הֱיוּ עֵרָנִים וְקַוּוּ בְּכָל לְבַבְכֶם לַחֶסֶד הַבָּא עֲלֵיכֶם בְּהִתְגַּלּוֹת יֵשׁוּעַ הַמָּשִׁיחַ. 14 כְּבָנִים מְצַיְּתִים, אַל תִּתְנַהֲגוּ לְפִי הַתַּאֲווֹת שֶׁהָיוּ לָכֶם בֶּעָבָר כְּשֶׁהֱיִיתֶם חַסְרֵי דַעַת, 15 אֶלָּא הֱיוּ גַּם אַתֶּם קְדוֹשִׁים בְּכָל דַּרְכֵיכֶם כְּמוֹ שֶׁקָּדוֹשׁ הוּא אֲשֶׁר קָרָא אֶתְכֶם,(ישעיהו ו' ג') 16 שֶׁהֲרֵי כָּתוּב, "קְדֹשִׁים תִּהְיוּ כִּי קָדוֹשׁ אֲנִי."(ויקרא יא' מד'; יט' ב') 17 וְאִם קוֹרְאִים אַתֶּם אֶל הָאָב, הַשּׁוֹפֵט בְּלִי מַשּׂוֹא פָּנִים לְפִי מַעֲשֵׂי אִישׁ וָאִישׁ,(דברי הימים ב', יט' ז'; תהילים סב' יג') אֲזַי הִתְהַלְּכוּ בְּיִרְאָה בִּימֵי מְגוּרֵיכֶם עֲלֵי אֲדָמוֹת;(תהילים ב' יא') 18 שֶׁהֲרֵי יוֹדְעִים אַתֶּם כִּי לֹא בְּדָבָר נִשְׁחָת, לֹא בְּכֶסֶף וְלֹא בְּזָהָב, נִפְדֵּיתֶם מִדַּרְכְּכֶם הַתְּפֵלָה שֶׁנְּחַלְתֶּם מֵאֲבוֹתֵיכֶם,(ישעיהו נב' ג') 19 כִּי אִם בְּדָם יָקָר שֶׁל שֶׂה תָּמִים שֶׁאֵין בּוֹ מוּם, בְּדָמוֹ שֶׁל הַמָּשִׁיחַ (ישעיהו נג' ז') 20 אֲשֶׁר נוֹדַע מִקֶּדֶם, לִפְנֵי הִוָּסֵד תֵּבֵל, וְנִגְלָה בְּאַחֲרִית הַיָּמִים לְמַעַנְכֶם, 21 אַתֶּם אֲשֶׁר הוֹדוֹת לוֹ מַאֲמִינִים בֵּאלֹהִים שֶׁהֱקִימוֹ מִן הַמֵּתִים וְנָתַן לוֹ כָּבוֹד לְמַעַן תִּהְיֶה אֱמוּנַתְכֶם וְתִקְוַתְכֶם בֵּאלֹהִים. 22 וְכֵיוָן שֶׁטִּהַרְתֶּם אֶת נַפְשׁוֹתֵיכֶם בְּצַיֶּתְכֶם לָאֱמֶת וְהִגַּעְתֶּם אֱלֵי אַחֲוָה שֶׁאֵין בָּהּ חֲנֻפָּה, אֶהֱבוּ זֶה אֶת זֶה בְּכָל לְבַבְכֶם וּמְאֹדְכֶם; 23 שֶׁהֲרֵי נוֹלַדְתֶּם מֵחָדָשׁ לֹא מִזֶּרַע נִשְׁחָת כִּי אִם מִבִּלְתִּי נִשְׁחָת, בִּדְבַר אֱלֹהִים הַחַי וְקַיָּם. 24 הֵן "כָּל-הַבָּשָׂר חָצִיר וְכָל-חַסְדּוֹ כְּצִיץ הַשָּׂדֶה. יָבֵשׁ חָצִיר, נָבֵל צִיץ, 25 וּדְבַר-אֱלֹהֵינוּ יָקוּם לְעוֹלָם."(ישעיהו מ' ו'-ח') וְהוּא הַדָּבָר אֲשֶׁר בִּשְּׂרוּ לָכֶם.
