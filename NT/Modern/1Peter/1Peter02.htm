לפרק הקודם

אגרת פטרוס הראשונה
פרק ב

אבנים חיות ועם קדוש

1 וְעַתָּה הָסִירוּ מֵעֲלֵיכֶם כָּל רֶשַׁע וְכָל מִרְמָה, אֶת הַצְּבִיעוּת וְהַקִּנְאָה וְכָל לָשׁוֹן הָרָע, 2 וּכְעוֹלָלִים אֲשֶׁר זֶה מִקָּרוֹב נוֹלְדוּ הִתְאַוּוּ לֶחָלָב הַזַּךְ שֶׁל הַדָּבָר, לְמַעַן תִּגְדְּלוּ בְּאֶמְצָעוּתוֹ לִישׁוּעָה, 3 אִם אָמְנָם טְעַמְתֶּם כִּי טוֹב הָאָדוֹן (תהילים לד' ט') 4 אֲשֶׁר אַתֶּם נִגָּשִׁים אֵלָיו -- אֶבֶן חַיָּה שֶׁמְּאָסוּהָ בְּנֵי אָדָם, אַךְ נִבְחָרָה וִיקָרָה לֵאלֹהִים.(תהילים קיח' כב') 5 וְגַם אַתֶּם, כַּאֲבָנִים חַיּוֹת, נִבְנִים לְבַיִת רוּחָנִי, לִכְהֻנַּת קֹדֶשׁ, כְּדֵי לְהַעֲלוֹת זִבְחֵי רוּחַ רְצוּיִים לֵאלֹהִים בִּזְכוּת יֵשׁוּעַ הַמָּשִׁיחַ.(ישעיהו סא' ו') 6 מִשּׁוּם כָּךְ אוֹמֵר הַכָּתוּב: "הִנְנִי יִסַּד בְּצִיּוֹן אֶבֶן פִּנָּה נִבְחָרָה וִיקָרָה, וְהַמַּאֲמִין בָּהּ לֹא יֵבוֹשׁ."(ישעיהו כח' טז') 7 עַל כֵּן לָכֶם הַמַּאֲמִינִים הִיא אֶבֶן הַיְּקָר, אֲבָל לַסּוֹרְרִים -- "אֶבֶן מָאֲסוּ הַבּוֹנִים הָיְתָה לְרֹאשׁ פִּנָּה",(תהילים קיח' כב') 8 וְגַם "אֶבֶן נֶגֶף וְצוּר מִכְשׁוֹל."(ישעיהו ח' יד') וְאָמְנָם הֵם נִכְשָׁלִים בַּדָּבָר בִּגְלַל סַרְבָנוּתָם, וּלְכָךְ גַּם נוֹעֲדוּ. 9 אֲבָל אַתֶּם עַם נִבְחָר, מַמְלֶכֶת כֹּהֲנִים וְגוֹי קָדוֹש,(שמות יט' ו') עַם סְגֻלָּה, לְמַעַן תְּסַפְּרוּ תְּהִלּוֹתָיו (ישעיהו מג' כא') שֶׁל הַקּוֹרֵא אֶתְכֶם מֵחֹשֶׁךְ אֶל אוֹרוֹ הַנִּפְלָא; 10 אַתֶּם אֲשֶׁר לְפָנִים לֹא עַם הֱיִיתֶם, אַךְ כָּעֵת עַם אֱלֹהִים אַתֶּם; אֲשֶׁר לְפָנִים לֹא רֻחֲמוּ, אַךְ כָּעֵת מְרֻחָמִים.(הושע א' ו'; א' ט'; ב' כה')

התנהגו כעבדי אלהים

11 אֲהוּבַי, שֶׁבִּבְחִינַת זָרִים וְגוֹלִים אַתֶּם,(תהילים לט' יג') מַפְצִיר אֲנִי בָּכֶם לְהִנָּזֵר מִתַּאֲווֹת בְּשָׂרִים הַנִּלְחָמוֹת נֶגֶד הַנֶּפֶשׁ. 12 הֵיטִיבוּ אֶת דַּרְכֵיכֶם בַּגּוֹיִים, כְּדֵי שֶׁיִּתְבּוֹנְנוּ בְּמַעֲשֵׂיכֶם הַטּוֹבִים וִיהַלְּלוּ אֶת אֱלֹהִים בְּיוֹם פְּקֻדָּה (ישעיהו י' ג') עַל אוֹתָם הַדְּבָרִים שֶׁבִּגְלָלָם הִשְׁמִיצוּ אֶתְכֶם כְּעוֹשֵׂי רָעָה. 13 הִכָּנְעוּ לְכָל מוֹסָד אֱנוֹשִׁי לְמַעַן הָאָדוֹן: אִם לַמֶּלֶךְ, בִּהְיוֹתוֹ הָרֹאשׁ;(משלי כד' כא') 14 אִם לַמּוֹשְׁלִים, בִּהְיוֹתָם שְׁלוּחִים מִטַּעְמוֹ לִנְקָמָה בְּעוֹשֵׂי הָרַע, אַךְ לְמַתַּן שֶׁבַח לְעוֹשֵׂי הַטּוֹב. 15 הֵן זֶהוּ רְצוֹן אֱלֹהִים, שֶׁתַּעֲשׂוּ הַטּוֹב וְכָךְ תָּשִׂימוּ מַחְסוֹם לְאִוֶּלֶת הָאֲנָשִׁים אֲשֶׁר אֵין בָּהֶם דַּעַת. 16 הִתְנַהֲגוּ כַּאֲנָשִׁים חָפְשִׁיִּים, לֹא כַּאֲנָשִׁים הַמַּחֲזִיקִים בַּחֹפֶשׁ כִּכְסוּת לָרֶשַׁע, אֶלָּא כְּעַבְדֵי אֱלֹהִים. 17 נַהֲגוּ כָּבוֹד בְּכָל אָדָם; אֶהֱבוּ אֶת הָאַחִים; יִרְאוּ אֶת אֱלֹהִים; כַּבְּדוּ אֶת הַמֶּלֶךְ.(קוהלת יב' יג')

המופת של סבלות המשיח

18 הָעֲבָדִים, הִכָּנְעוּ לִפְנֵי אֲדוֹנֵיכֶם בְּכָל יִרְאָה, לֹא רַק לִפְנֵי הַטּוֹבִים וְהַנּוֹחִים לַבְּרִיּוֹת, אֶלָּא גַּם לִפְנֵי הָעִקְשִׁים. 19 הֵן בִּרְכַּת חֶסֶד בְּכָךְ אִם בִּגְלַל הַכָּרָתוֹ בֵּאלֹהִים יִשָּׂא אָדָם מַכְאוֹב וְיִסְבֹּל שֶׁלֹּא בְּצֶדֶק; 20 שֶׁהֲרֵי מַה תְּהִלָּה בְּכָךְ אִם תַּעַמְדוּ בְּסֵבֶל שֶׁל מַכּוֹת עַל חֲטָאֵיכֶם? אֲבָל אִם בַּעֲשׂוֹתְכֶם אֶת הַטּוֹב תִּסְבְּלוּ וְתַעַמְדוּ בַּסֵּבֶל, בִּרְכַּת חֶסֶד הִיא מִלִּפְנֵי אֱלֹהִים. 21 אָכֵן לָזֹאת נִקְרֵאתֶם, כִּי גַּם הַמָּשִׁיחַ סָבַל בַּעַדְכֶם וְהִשְׁאִיר לָכֶם מוֹפֵת כְּדֵי שֶׁתֵּלְכוּ בְּעִקְּבוֹתָיו -- 22 הוּא אֲשֶׁר חֵטְא לֹא עָשָׂה וְלֹא נִמְצְאָה מִרְמָה בְּפִיו;(ישעיהו נג' ט') 23 אֲשֶׁר חֵרְפוּהוּ וְלֹא הֵשִׁיב חֵרוּף, סָבַל וְלֹא אִיֵּם, כִּי אִם מָסַר דִּינוֹ לְשׁוֹפֵט הַצֶּדֶק; 24 הוּא אֲשֶׁר אֶת חֲטָאֵינוּ נָשָׂא בְּגוּפוֹ עַל הָעֵץ, כְּדֵי שֶׁנָּמוּת לְגַבֵּי הַחֵטְא וְנִחְיֶה לַצְּדָקָה; אֲשֶׁר בְּחַבּוּרָתוֹ נִרְפָּא לָכֶם,(ישעיהו נג' ה') 25 כִּי הֱיִיתֶם כְּצֹאן אוֹבְדוֹת,(ישעיהו נג' ו') אַךְ עַתָּה שַׁבְתֶּם אֶל הָרוֹעֶה הַשּׁוֹמֵר אֶת נַפְשׁוֹתֵיכֶם.(ישעיהו מ' יא'; יחזקאל לד' כג')
