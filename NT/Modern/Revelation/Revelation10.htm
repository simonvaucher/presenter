פרק י

המלאך והספר הקטן

1 רָאִיתִי מַלְאָךְ אַחֵר רַב-כֹּחַ יוֹרֵד מִן הַשָּׁמַיִם וְהוּא לָבוּשׁ עָנָן;(תהילים קג' כ') קֶשֶׁת עַל רֹאשׁוֹ, פָּנָיו כַּשֶּׁמֶשׁ וְרַגְלָיו כְּעַמּוּדֵי אֵשׁ,(דניאל י' ו'; יחזקאל א' כז'-כח') 2 וּבְיָדוֹ סֵפֶר קָטָן פָּתוּחַ. הוּא שָׂם אֶת רַגְלוֹ הַיְמָנִית עַל הַיָּם וְאֶת הַשְּׂמָאלִית עַל הָאָרֶץ,(יחזקאל ב' ט') 3 וְקָרָא בְּקוֹל גָּדוֹל כְּאַרְיֵה שׁוֹאֵג. כַּאֲשֶׁר קָרָא דִּבְּרוּ שִׁבְעַת הָרְעָמִים בְּקוֹלוֹתֵיהֶם;(הושע יא' י') 4 וּכְשֶׁדִּבְּרוּ שִׁבְעַת הָרְעָמִים הִתְכַּוַּנְתִּי לִכְתֹּב, אַךְ שָׁמַעְתִּי קוֹל מִן הַשָּׁמַיִם אוֹמֵר: "חֲתֹם אֶת אֲשֶׁר דִּבְּרוּ שִׁבְעַת הָרְעָמִים וְאַל תִּכְתֹּב זֹאת."(דניאל ח' כו'; דניאל יב' ד'; יב' ט') 5 וְהַמַּלְאָךְ שֶׁרְאִיתִיו עוֹמֵד עַל הַיָּם וְעַל הָאָרֶץ הֵרִים אֶת יַד יְמִינוֹ הַשָּׁמַיְמָה 6 וְנִשְׁבַּע בַּחַי לְעוֹלְמֵי עוֹלָמִים,(דניאל יב' ז') אֲשֶׁר בָּרָא אֶת הַשָּׁמַיִם וְאֶת אֲשֶׁר בָּם וְאֶת הָאָרֶץ וְאֶת אֲשֶׁר בָּהּ וְאֶת הַיָּם וְאֶת אֲשֶׁר בּוֹ -- שֶׁלֹּא תִּהְיֶה עוֹד שָׁהוּת.(תהילים קמו' ו') 7 אוּלָם בִּימֵי קוֹל הַמַּלְאָךְ הַשְּׁבִיעִי, בְּעָמְדוֹ לִתְקֹעַ בַּשּׁוֹפָר, גַּם יֻשְׁלַם רָז הָאֱלֹהִים, כְּמוֹ שֶׁבִּשֵּׂר לַעֲבָדָיו הַנְּבִיאִים.(עמוס ג' ז') 8 וְהַקּוֹל אֲשֶׁר שָׁמַעְתִּי מִן הַשָּׁמַיִם דִּבֵּר אֵלַי שׁוּב וְאָמַר: "לֵךְ קַח אֶת הַסֵּפֶר הַפָּתוּחַ שֶׁבְּיַד הַמַּלְאָךְ הָעוֹמֵד עַל הַיָּם וְעַל הָאָרֶץ." 9 הָלַכְתִּי אֶל הַמַּלְאָךְ וְאָמַרְתִּי לוֹ שֶׁיִּתֵּן לִי אֶת הַסֵּפֶר הַקָּטָן. אָמַר אֵלַי: "קַח וֶאֱכֹל אוֹתוֹ, הוּא יֵמַר בְּבִטְנְךָ, אַךְ בְּפִיךָ יִהְיֶה מָתוֹק כִּדְבַשׁ."(יחזקאל ב' ח'; ג' א'-ג') 10 לָקַחְתִּי אֶת הַסֵּפֶר הַקָּטָן מִיָּדוֹ שֶׁל הַמַּלְאָךְ וַאֲכַלְתִּיו; בְּפִי הָיָה מָתוֹק כִּדְבַשׁ, אֲבָל אַחֲרֵי אָכְלִי אוֹתוֹ הֵמַר לְבִטְנִי. 11 אָז אָמְרוּ לִי: "עָלֶיךָ עוֹד לְהִנָּבֵא עַל עַמִּים וְאֻמּוֹת וּלְשׁוֹנוֹת וּמְלָכִים רַבִּים."
