לפרק הקודם

התגלות
פרק כ

אלף השנים

1 רָאִיתִי מַלְאָךְ יוֹרֵד מִן הַשָּׁמַיִם וּבְיָדוֹ מַפְתֵּחַ הַתְּהוֹם וְשַׁרְשֶׁרֶת גְּדוֹלָה. 2 הוּא תָּפַשׂ אֶת הַתַּנִּין, הַנָּחָשׁ הַקַּדְמוֹנִי אֲשֶׁר הוּא הַמַּלְשִׁין וְהַשָּׂטָן,(בראשית ג' א'; ג' יד') וְקָשַׁר אוֹתוֹ לְאֶלֶף שָׁנִים. 3 הִשְׁלִיךְ אוֹתוֹ אֶל הַתְּהוֹם וְסָגַר וְחָתַם עָלָיו כְּדֵי שֶׁלֹּא יַתְעֶה עוֹד אֶת הַגּוֹיִים עַד תֹּם אֶלֶף הַשָּׁנִים; אַחֲרֵי כֵן צָרִיךְ שֶׁיֻּתַּר לִזְמַן מְעַט. 4 רָאִיתִי כִּסְאוֹת; הִתְיַשְּׁבוּ עֲלֵיהֶם וּמִשְׁפָּט נִתַּן בְּיָדָם.(דניאל ז' ט'; ז' כב') וְנַפְשׁוֹת הַנֶּעֱרָפִים עַל עֵדוּת יֵשׁוּעַ וְעַל דְּבַר אֱלֹהִים, וְאֵלֶּה אֲשֶׁר לֹא הִשְׁתַּחֲווּ לַחַיָּה וּלְצַלְמָהּ וְלֹא קִבְּלוּ אֶת הַתָּו עַל מִצְחָם וְעַל יָדָם -- חָיוּ וּמָלְכוּ עִם הַמָּשִׁיחַ אֶלֶף שָׁנִים. 5 שְׁאָר הַמֵּתִים לֹא חָיוּ עַד תֹּם אֶלֶף הַשָּׁנִים. זֹאת הַתְּחִיָּה הָרִאשׁוֹנָה. 6 מְאֻשָּׁר וְקָדוֹשׁ מִי שֶׁחֶלְקוֹ בַּתְּחִיָּה הָרִאשׁוֹנָה; עַל אֵלֶּה אֵין לַמָּוֶת הַשֵּׁנִי שִׁלְטוֹן, כִּי יִהְיוּ כֹּהֲנִים לֵאלֹהִים וְלַמָּשִׁיחַ וְיִמְלְכוּ אִתּוֹ אֶלֶף שָׁנִים.(ישעיהו סא' ו')

מפלת השטן

7 אַחֲרֵי כְּלוֹת אֶלֶף הַשָּׁנִים יֻתַּר הַשָּׂטָן מִכִּלְאוֹ 8 וְיֵצֵא לְהַתְעוֹת אֶת הַגּוֹיִים בְּאַרְבַּע קְצוֹת הָאָרֶץ, אֶת גּוֹג וּמָגוֹג,(יחזקאל לח' ב'; בראשית י' ב') לֶאֱסֹף אוֹתָם לַמִּלְחָמָה; וּמִסְפָּרָם כְּחוֹל הַיָּם. 9 הֵם עָלוּ עַל מֶרְחַב הָאָרֶץ וְהִקִּיפוּ אֶת מַחֲנֵה הַקְּדוֹשִׁים וְאֶת הָעִיר הָאֲהוּבָה;(דניאל ז' כא') אֵשׁ יָרְדָה מִן הַשָּׁמַיִם מֵאֵת אֱלֹהִים וְאָכְלָה אוֹתָם.(יחזקאל לח' כב'; לט' ו') 10 וְהַשָּׂטָן אֲשֶׁר הִדִּיחָם הֻשְׁלַךְ אֶל אֲגַם הָאֵשׁ וְהַגָּפְרִית, אֲשֶׁר שָׁם גַּם הַחַיָּה וּנְבִיא הַשֶּׁקֶר, וְהֵם יְעֻנּוּ יוֹמָם וָלַיְלָה לְעוֹלְמֵי עוֹלָמִים.(ישעיהו לד' י')

המשפט האחרון

11 רָאִיתִי כִּסֵּא לָבָן גָּדוֹל וְאֶת הַיּוֹשֵׁב עָלָיו אֲשֶׁר נָסוּ מִפָּנָיו הָאָרֶץ וְהַשָּׁמַיִם, וּמָקוֹם לֹא נִמְצָא לָהֶם.(דניאל ז' ט'; ישעיהו ו' א'; לד' ד'; נא' ו') 12 וְרָאִיתִי אֶת הַמֵּתִים, הַקְּטַנִּים וְהַגְּדוֹלִים, עוֹמְדִים לִפְנֵי הַכִּסֵּא וּסְפָרִים נִפְתְּחוּ. גַּם סֵפֶר אַחֵר נִפְתַּח, סֵפֶר הַחַיִּים.(דניאל ז' י'; יב' א') וְהַמֵּתִים נִשְׁפְּטוּ מִתּוֹךְ הַדְּבָרִים הַכְּתוּבִים בַּסְּפָרִים, לְפִי מַעֲשֵׂיהֶם.(משלי כד' יב'; קוהלת יב' יד') 13 הַיָּם נָתַן אֶת הַמֵּתִים אֲשֶׁר בּוֹ, מָוֶת וּשְׁאוֹל נָתְנוּ אֶת מֵתֵיהֶם, וְאִישׁ אִישׁ נִשְׁפְּטוּ לְפִי מַעֲשֵׂיהֶם. 14 הַמָּוֶת וְהַשְּׁאוֹל הֻשְׁלְכוּ אֶל אֲגַם הָאֵשׁ. זֶהוּ הַמָּוֶת הַשֵּׁנִי -- אֲגַם הָאֵשׁ. 15 וּמִי שֶׁלֹּא נִמְצָא כָּתוּב בְּסֵפֶר הַחַיִּים, הֻשְׁלַךְ אֶל אֲגַם הָאֵשׁ.(דניאל יב' א'; ז' יא')
