לפרק הקודם

התגלות
פרק ז

החתומים מבני ישראל

1 אַחֲרֵי כֵן רָאִיתִי אַרְבָּעָה מַלְאָכִים עוֹמְדִים בְּאַרְבַּע פִּנּוֹת הָאָרֶץ, אוֹחֲזִים אֶת אַרְבַּע רוּחוֹת הָאָרֶץ כְּדֵי שֶׁלֹּא תִּשֹּׁב רוּחַ לֹא עַל הָאָרֶץ וְלֹא עַל הַיָּם וְלֹא עַל כָּל עֵץ.(יחזקאל ז' ב') 2 רָאִיתִי מַלְאָךְ אַחֵר עוֹלֶה מִמִּזְרַח שֶׁמֶשׁ וּבִרְשׁוּתוֹ חוֹתַם אֱלֹהִים חַיִּים. הוּא קָרָא בְּקוֹל גָּדוֹל אֶל אַרְבַּעַת הַמַּלְאָכִים אֲשֶׁר הֻתַּר לָהֶם לְחַבֵּל בָּאָרֶץ וּבַיָּם, 3 בְּאָמְרוֹ: "אַל תְּחַבְּלוּ בָּאָרֶץ וּבַיָּם וּבָעֵצִים עַד אֲשֶׁר נַחְתֹּם אֶת עַבְדֵי אֱלֹהֵינוּ עַל מִצְחוֹתֵיהֶם.(יחזקאל ט' ד'; ו') 4 וְשָׁמַעְתִּי אֶת מִסְפַּר הַחֲתוּמִים -- מֵאָה וְאַרְבָּעִים וְאַרְבָּעָה אֶלֶף חֲתוּמִים מִכָּל שִׁבְטֵי בְּנֵי יִשְׂרָאֵל: 5 מִשֵּׁבֶט יְהוּדָה שְׁנֵים-עָשָׂר אֶלֶף חֲתוּמִים, מִשֵּׁבֶט רְאוּבֵן שְׁנֵים-עָשָׂר אֶלֶף, מִשֵּׁבֶט גָּד שְׁנֵים-עָשָׂר אֶלֶף, 6 מִשֵּׁבֶט אָשֵׁר שְׁנֵים-עָשָׂר אֶלֶף, מִשֵּׁבֶט נַפְתָּלִי שְׁנֵים-עָשָׂר אֶלֶף, מִשֵּׁבֶט מְנַשֶּׁה שְׁנֵים-עָשָׂר אֶלֶף, 7 מִשֵּׁבֶט שִׁמְעוֹן שְׁנֵים-עָשָׂר אֶלֶף, מִשֵּׁבֶט לֵוִי שְׁנֵים-עָשָׂר אֶלֶף, מִשֵּׁבֶט יִשָּׂשכָר שְׁנֵים-עָשָׂר אֶלֶף, 8 מִשֵּׁבֶט זְבוּלוּן שְׁנֵים-עָשָׂר אֶלֶף, מִשֵּׁבֶט יוֹסֵף שְׁנֵים-עָשָׂר אֶלֶף, מִשֵּׁבֶט בִּנְיָמִין שְׁנֵים-עָשָׂר אֶלֶף חֲתוּמִים.

ההמון הרב

9 אַחֲרֵי כֵן רָאִיתִי וְהִנֵּה הָמוֹן רַב, אֲשֶׁר לֹא יָכוֹל אִישׁ לִמְנוֹתוֹ, מִכָּל הָאֻמּוֹת וְהַשְּׁבָטִים וְהָעַמִּים וְהַלְּשׁוֹנוֹת -- עוֹמְדִים לִפְנֵי הַכִּסֵּא וְלִפְנֵי הַשֶּׂה וְהֵם לְבוּשִׁים גְּלִימוֹת לְבָנוֹת וְכַפּוֹת תְּמָרִים בִּידֵיהֶם, 10 וְקוֹרְאִים בְּקוֹל גָּדוֹל: "הַיְשׁוּעָה לֵאלֹהֵינוּ הַיּוֹשֵׁב עַל הַכִּסֵּא וְלַשֶּׂה." 11 כָּל הַמַּלְאָכִים עָמְדוּ סָבִיב לַכִּסֵּא וְלַזְּקֵנִים וּלְאַרְבַּע הַחַיּוֹת. הֵם נָפְלוּ עַל פְּנֵיהֶם לִפְנֵי הַכִּסֵּא וְהִשְׁתַּחֲווּ לֵאלֹהִים 12 בְּאָמְרָם: "אָמֵן. הַבְּרָכָה וְהַכָּבוֹד, הַחָכְמָה וְהַתּוֹדָה, הַיְקָר וְהַגְּבוּרָה וְהָעֹז לֵאלֹהֵינוּ לְעוֹלְמֵי עוֹלָמִים. אָמֵן."(תהילים כט' א'-ב'; מא' יד') 13 דִּבֵּר אַחַד הַזְּקֵנִים וְאָמַר אֵלַי: "אֵלֶּה הַלְּבוּשִׁים גְּלִימוֹת לְבָנוֹת מִי הֵם וּמֵאַיִן בָּאוּ?" 14 הֵשַׁבְתִּי לוֹ: "אֲדוֹנִי, אַתָּה יוֹדֵעַ." אָמַר אֵלַי: "אֵלֶּה הֵם הַבָּאִים מִן הַצָּרָה הַגְּדוֹלָה. הֵם כִּבְּסוּ אֶת גְּלִימוֹתֵיהֶם וְהִלְבִּינוּ אוֹתָן בְּדַם הַשֶּׂה.(ירמיהו ל' ז') 15 לָכֵן הֵם נִמְצָאִים לִפְנֵי כִּסֵּא אֱלֹהִים וְעוֹבְדִים אוֹתוֹ יוֹמָם וָלַיְלָה בְּהֵיכָלוֹ, וְהַיּוֹשֵׁב עַל הַכִּסֵּא יִפְרֹשׂ מִשְׁכָּנוֹ עֲלֵיהֶם. 16 לֹא יִרְעֲבוּ עוֹד וְלֹא יִצְמְאוּ, אַף לֹא תַּכֶּה עֲלֵיהֶם הַשֶּׁמֶשׁ וְכָל שָׁרָב,(ישעיהו מט' י') 17 כִּי הַשֶּׂה אֲשֶׁר בְּאֶמְצַע הַכִּסֵּא יִרְעֶה אוֹתָם וְיַנְהִיגֵם אֶל מַבּוּעֵי מַיִם חַיִּים;(יחזקאל לד' כג'; תהילים כג' ב') וֵאלֹהִים יִמְחֶה כָּל דִּמְעָה מֵעֵינֵיהֶם."(ישעיהו כה' ח')
