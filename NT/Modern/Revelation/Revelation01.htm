לאגרת יהודה

התגלות
פרק א

1 הִתְגַּלּוּת יֵשׁוּעַ הַמָּשִׁיחַ שֶׁנָּתַן לוֹ הָאֱלֹהִים כְּדֵי לְהַרְאוֹת לַעֲבָדָיו אֶת אֲשֶׁר צָרִיךְ לִהְיוֹת בִּמְהֵרָה. וְהוּא שָׁלַח בְּיַד מַלְאָכוֹ וְהוֹדִיעַ לְעַבְדּוֹ יוֹחָנָן (דניאל ב' כח'; ח' טז') 2 שֶׁהֵעִיד אֶת דְּבַר אֱלֹהִים וְאֶת עֵדוּת יֵשׁוּעַ הַמָּשִׁיחַ, אֶת כָּל אֲשֶׁר רָאָה. 3 אַשְׁרֵי הַקּוֹרֵא וְאַשְׁרֵי הַשּׁוֹמְעִים אֶת דִּבְרֵי הַנְּבוּאָה וְשׁוֹמְרִים אֶת הַכָּתוּב בָּהּ, כִּי קְרוֹבָה הָעֵת.(תהילים קיט' ב') 4 יוֹחָנָן אֶל שֶׁבַע הַקְּהִלּוֹת אֲשֶׁר בְּאַסְיָה: חֶסֶד וְשָׁלוֹם לָכֶם מֵאֵת הַהֹוֶה וְהָיָה וְיָבוֹא, וּמֵאֵת שֶׁבַע הָרוּחוֹת אֲשֶׁר לִפְנֵי כִּסְאוֹ, 5 וּמֵאֵת יֵשׁוּעַ הַמָּשִׁיחַ הָעֵד הַנֶּאֱמָן,(ישעיהו נה' ד') בְּכוֹר הַמֵּתִים וְעֶלְיוֹן לְמַלְכֵי הָאָרֶץ.(תהילים עב' יא') לָאוֹהֵב אוֹתָנוּ אֲשֶׁר בְּדָמוֹ שִׁחְרֵר אוֹתָנוּ מֵחֲטָאֵינוּ 6 וְעָשָׂה אוֹתָנוּ מַמְלֶכֶת כֹּהֲנִים לֵאלֹהִים אָבִיו, לוֹ הַכָּבוֹד וְהַגְּבוּרָה לְעוֹלְמֵי עוֹלָמִים. אָמֵן.(ישעיהו סא' ו') 7 הִנֵּה הוּא בָּא עִם הָעֲנָנִים.(דניאל ז' יג'-יד') כָּל עַיִן תִּרְאֶה אוֹתוֹ, גַּם אֵלֶּה שֶׁדְּקָרוּהוּ, וְיִסְפְּדוּ עָלָיו כָּל מִשְׁפְּחוֹת הָאָרֶץ. כֵּן; אָמֵן.(זכריה יב' י'-יד') 8 "אֲנִי הָאָלֶף אַף אֲנִי הַתָּו", נְאֻם יהוה אֱלֹהִים, "הַהֹוֶה וְהָיָה וְיָבוֹא, אֱלֹהֵי צְבָאוֹת."(ישעיהו מא' ד'; מד' ו'; מח' יב'; שמות ג' יד')

התגלות הקול המדבר

9 אֲנִי יוֹחָנָן אֲחִיכֶם וְשֻׁתָּף בַּצָּרָה וּבַמַּלְכוּת וּבַסַּבְלָנוּת בְּיֵשׁוּעַ, הָיִיתִי בָּאִי הַקָּרוּי פַּטְמוֹס בַּעֲבוּר דְּבַר אֱלֹהִים וְעֵדוּת יֵשׁוּעַ. 10 הָיִיתִי בְּהַשְׁרָאַת הָרוּחַ בְּיוֹם הָאָדוֹן וְשָׁמַעְתִּי מֵאַחֲרַי קוֹל גָּדוֹל כְּקוֹל שׁוֹפָר (שמות יט' יט'; תהילים כט' ד') 11 אוֹמֵר: "אֶת אֲשֶׁר אַתָּה רוֹאֶה כְּתֹב בְּסֵפֶר וּשְׁלַח אֶל שֶׁבַע הַקְּהִלּוֹת: אֶל אֶפֶסוֹס, אֶל זְמִירְנָה, אֶל פֶּרְגָּמוֹס, אֶל תִּיאָטִירָה, אֶל סַרְדִּיס, אֶל פִילָדֶלְפִיָה וְאֶל לָאוֹדִיקֵאָה." 12 פָּנִיתִי לִרְאוֹת אֶת הַקּוֹל הַמְדַבֵּר אֵלַי, וּכְשֶׁפָּנִיתִי רָאִיתִי שֶׁבַע מְנוֹרוֹת זָהָב,(זכריה ד' ב') 13 וּבֵין שֶׁבַע הַמְּנוֹרוֹת כְּמַרְאֵה בֶּן-אָדָם (יחזקאל א' כו'; דניאל ז' יג') עוֹטֶה מְעִיל עַד מַרְגְּלוֹתָיו וְחָגוּר חֲגוֹרַת זָהָב עַל חֲזֵהוּ. 14 רֹאשׁוֹ וּשְׂעָרוֹ לְבָנִים כְּצֶמֶר לָבָן כַּשֶּׁלֶג, וְעֵינָיו כְּשַׁלְהֶבֶת אֵשׁ; 15 רַגְלָיו כְּמוֹ נְחֹשֶׁת נוֹצֶצֶת שֶׁנִּצְרְפָה בַּכּוּר,(דניאל י' ו'; יחזקאל א' כז') וְקוֹלוֹ כְּקוֹל מַיִם רַבִּים. 16 בְּיַד יְמִינוֹ שִׁבְעָה כּוֹכָבִים; מִפִּיו יוֹצֵאת חֶרֶב פִּיפִיּוֹת חַדָּה, וּפָנָיו כַּשֶּׁמֶשׁ הַמְּאִירָה בִּגְבוּרָתָהּ. 17 כַּאֲשֶׁר רְאִיתִיו נָפַלְתִּי לְרַגְלָיו כַּמֵּת,(דניאל ח' יח'; י' ח') וְהוּא שָׂם אֶת יַד יְמִינוֹ עָלַי וְאָמַר: "אַל תִּירָא. אֲנִי הָרִאשׁוֹן וְהָאַחֲרוֹן וְהַחַי.(***) 18 הָיִיתִי מֵת וְהִנֵּה חַי אֲנִי לְעוֹלְמֵי עוֹלָמִים, וְלִי מַפְתְּחוֹת שְׁאוֹל וָמָוֶת. 19 כְּתֹב אֶת אֲשֶׁר רָאִיתָ וְאֶת אֲשֶׁר הֹוֶה וְאֶת אֲשֶׁר עָתִיד לִהְיוֹת אַחֲרֵי כֵן, 20 אֶת סוֹד שִׁבְעַת הַכּוֹכָבִים אֲשֶׁר רָאִיתָ בִּימִינִי וְשֶׁבַע מְנוֹרוֹת הַזָּהָב: שִׁבְעַת הַכּוֹכָבִים הֵם מַלְאֲכֵי שֶׁבַע הַקְּהִלּוֹת, וְשֶׁבַע הַמְּנוֹרוֹת הֵן שֶׁבַע קְהִלּוֹת."
