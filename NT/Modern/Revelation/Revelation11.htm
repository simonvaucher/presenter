לפרק הקודם

התגלות
פרק יא

שני העדים

1 נִתַּן לִי קָנֶה דּוֹמֶה לְמַטֶּה, וְנֶאֱמַר לִי: "קוּם וּמְדֹד אֶת הֵיכַל אֱלֹהִים וְאֶת הַמִּזְבֵּחַ וְאֶת הַמִּשְׁתַּחֲוִים בּוֹ.(יחזקאל מב' טו') 2 אֲבָל אֶת הֶחָצֵר אֲשֶׁר מִחוּץ לַהֵיכָל הַשְׁאֵר בַּחוּץ וְאַל תִּמְדֹּד אוֹתָהּ, כִּי נִתְּנָה לַגּוֹיִים וְיִרְמְסוּ אֶת עִיר הַקֹּדֶשׁ אַרְבָּעִים וּשְׁנַיִם חֳדָשִׁים.(תהילים עט' א'; ישעיהו סג' יח'; דניאל יב' ז') 3 וְאֶתֵּן לִשְׁנֵי עֵדַי וִינַבְּאוּ אֶלֶף וּמָאתַיִם וְשִׁשִּׁים יָמִים וְהֵם לְבוּשֵׁי שַׂקִּים."(דברים יט' טו') 4 אֵלֶּה הֵם שְׁנֵי הַזֵּיתִים וּשְׁתֵּי הַמְּנוֹרוֹת הָעוֹמְדִים לִפְנֵי אֲדוֹן הָאָרֶץ.(זכריה ד' ג'; יא'-יד') 5 וְאִישׁ אִם יִרְצֶה לְהָרַע לָהֶם, תֵּצֵא אֵשׁ מִפִּיהֶם וְתֹאכַל אֶת אוֹיְבֵיהֶם;(שמואל ב', כב' ט') מִי שֶׁיִּרְצֶה לְהָרַע לָהֶם, בְּדֶרֶךְ זֹאת מוֹת יוּמַת. 6 יֵשׁ לָהֶם הַסַּמְכוּת לַעֲצֹר אֶת הַשָּׁמַיִם כְּדֵי שֶׁלֹּא יֵרֵד גֶּשֶׁם בִּימֵי נְבוּאָתָם,(מלכים א', יז' א') וְסַמְכוּת לָהֶם עַל הַמַּיִם לַהֲפֹךְ אוֹתָם לְדָם וּלְהַכּוֹת אֶת הָאָרֶץ בְּכָל מַכָּה שֶׁיִּרְצוּ.(שמות ז' כ'-כא') 7 כַּאֲשֶׁר יִגְמְרוּ אֶת עֵדוּתָם, הַחַיָּה הָעוֹלָה מִן הַתְּהוֹם תַּעֲשֶׂה עִמָּהֶם מִלְחָמָה, תְּנַצֵּחַ אוֹתָם וְתַהַרְגֵם.(דניאל ז' ז'; ז' כא') 8 גְּוִיּוֹתֵיהֶם תִּהְיֶינָה בִּרְחוֹב הָעִיר הַגְּדוֹלָה הַנִּקְרֵאת בְּאֹפֶן רוּחָנִי סְדוֹם וּמִצְרַיִם, אֲשֶׁר שָׁם גַּם נִצְלַב אֲדוֹנָם. 9 וּמִן הָעַמִּים וְהַשְּׁבָטִים וְהַלְּשׁוֹנוֹת וְהָאֻמּוֹת יִרְאוּ אֶת גְּוִיּוֹתֵיהֶם בְּמֶשֶׁךְ שְׁלוֹשָׁה יָמִים וָחֵצִי, וְלֹא יַרְשׁוּ לְהַנִּיחַ אֶת גְּוִיּוֹתֵיהֶם בְּקֶבֶר.(תהילים עט' ג') 10 יוֹשְׁבֵי הָאָרֶץ יִשְׂמְחוּ עֲלֵיהֶם וְיַעַלְזוּ, וְיִשְׁלְחוּ מָנוֹת אִישׁ לְרֵעֵהוּ, שֶׁכֵּן שְׁנֵי הַנְּבִיאִים הָאֵלֶּה הִכְאִיבוּ לְיוֹשְׁבֵי הָאָרֶץ. 11 אַחֲרֵי שְׁלוֹשָׁה יָמִים וָחֵצִי בָּאָה בָּהֶם רוּחַ חַיִּים מֵאֵת הָאֱלֹהִים; הֵם קָמוּ עַל רַגְלֵיהֶם וּפַחַד גָּדוֹל נָפַל עַל רוֹאֵיהֶם.(יחזקאל לז' י') 12 אָז שָׁמְעוּ קוֹל גָּדוֹל מִן הַשָּׁמַיִם אוֹמֵר לָהֶם: "עֲלוּ הֵנָּה!", וְהֵם עָלוּ בֶּעָנָן הַשָּׁמַיְמָה לְעֵינֵי אוֹיְבֵיהֶם. 13 אוֹתָהּ שָׁעָה הִתְחוֹלְלָה רְעִידַת אֲדָמָה גְּדוֹלָה; עֲשִׂירִית הָעִיר נָפְלָה וְשִׁבְעַת אֲלָפִים אֲנָשִׁים, לִשְׁמוֹתֵיהֶם, נֶהֶרְגוּ בִּרְעִידַת הָאֲדָמָה; וְהַנִּשְׁאָרִים נֶחֶרְדוּ וְנָתְנוּ כָּבוֹד לֵאלֹהֵי הַשָּׁמַיִם. 14 אוֹי שֵׁנִי חָלַף; הִנֵּה אוֹי שְׁלִישִׁי בָּא מַהֵר.

השופר השביעי

15 הַמַּלְאָךְ הַשְּׁבִיעִי תָּקַע בַּשּׁוֹפָר וְקוֹלוֹת גְּדוֹלִים נִשְׁמְעוּ בַּשָּׁמַיִם -- אוֹמְרִים: "הָיְתָה מַמְלֶכֶת תֵּבֵל לְמַמְלַכְתּוֹ שֶׁל אֲדוֹנֵנוּ וְשֶׁל מְשִׁיחוֹ וְהוּא יִמְלֹךְ לְעוֹלְמֵי עוֹלָמִים."(דניאל ב' מד') 16 עֶשְׂרִים וְאַרְבָּעָה הַזְּקֵנִים, הַיּוֹשְׁבִים עַל כִּסְאוֹתֵיהֶם לִפְנֵי הָאֱלֹהִים, נָפְלוּ עַל פְּנֵיהֶם וְהִשְׁתַּחֲווּ לֵאלֹהִים 17 בְּאָמְרָם: "מוֹדִים אֲנַחְנוּ לְךָ, יהוה אֱלֹהֵי צְבָאוֹת אֲשֶׁר הֹוֶה וְהָיָה, כִּי לָבַשְׁתָּ עֻזְּךָ הַגָּדוֹל וַתִּמְלֹךְ.(תהילים קמה' יג' ) 18 הַגּוֹיִים קָצְפוּ; וּבָא קִצְפְּךָ (תהילים ב' א'-ב'; ב' ה') וְעֵת לִשְׁפֹּט הַמֵּתִים וְלָתֵת שָׂכָר לַעֲבָדֶיךָ הַנְּבִיאִים וְלַקְּדוֹשִׁים וּלְיִרְאֵי שִׁמְךָ, לַקְּטַנִּים וְלַגְּדוֹלִים, וּלְהַשְׁחִית אֶת מַשְׁחִיתֵי הָאָרֶץ."(ישעיהו מ' י'; סא' ח') 19 אָז נִפְתַּח הֵיכַל אֱלֹהִים בַּשָּׁמַיִם וַאֲרוֹן בְּרִיתוֹ נִרְאָה בְּהֵיכָלוֹ;(שמות כה' ט') בְּרָקִים וְקוֹלוֹת וּרְעָמִים הִתְחוֹלְלוּ, וּרְעִידַת אֲדָמָה וּבָרָד כָּבֵד.(שמות יט' טז')
