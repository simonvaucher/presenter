להאגרת אל העברים

אגרת יעקב
פרק א

1 יַעֲקֹב, עֶבֶד אֱלֹהִים וַאֲדוֹנֵנוּ יֵשׁוּעַ הַמָּשִׁיחַ, דּוֹרֵשׁ בִּשְׁלוֹם שְׁנֵים-עָשָׂר הַשְּׁבָטִים שֶׁבַּגּוֹלָה.

אמונה וחכמה

2 לְשִׂמְחָה גְּדוֹלָה חִשְׁבוּ זֹאת, אַחַי, כַּאֲשֶׁר אַתֶּם בָּאִים בְּכָל מִינֵי נִסְיוֹנוֹת, 3 שֶׁהֲרֵי יוֹדְעִים אַתֶּם כִּי בְּחִינַת אֱמוּנַתְכֶם מְבִיאָה לִידֵי סַבְלָנוּת. 4 אֲבָל שֶׁתְּהֵא הַסַּבְלָנוּת שְׁלֵמָה בְּפָעֳלָהּ, לְמַעַן תִּהְיוּ שְׁלֵמִים וּבְלֹא דֹּפִי וְלֹא יֶחְסַר לָכֶם דָּבָר. 5 אִישׁ מִכֶּם אִם יֶחְסַר חָכְמָה, יְבַקֵּשׁ מֵאֱלֹהִים הַנּוֹתֵן לַכֹּל בִּנְדִיבוּת וּבְלֹא גְּעָרָה, וְתִנָּתֵן לוֹ.(משלי ב' ו') 6 אַךְ יְבַקֵּשׁ בֶּאֱמוּנָה וּבְלִי סָפֵק, כִּי בַּעַל-סָפֵק דּוֹמֶה לְגַלֵּי הַיָּם הַנִּדְחָפִים וְסוֹעֲרִים מִפְּנֵי הָרוּחַ. 7 אוֹתוֹ הָאִישׁ אַל יַחְשֹׁב כִּי יְקַבֵּל מַשֶּׁהוּ מֵאֵת יהוה, 8 בִּהְיוֹתוֹ אִישׁ הַפּוֹסֵחַ עַל שְׁתֵּי הַסְּעִפִּים הֲפַכְפַּךְ בְּכָל דְּרָכָיו.

עוני ועושר

9 יִתְהַלֵּל הָאָח הַדַּל בְּרוֹמְמוּתוֹ, 10 וְהֶעָשִׁיר שֶׁיִּתְהַלֵּל בְּשִׁפְלוֹ, כִּי כְּצִיץ חָצִיר הוּא יַחֲלֹף.(ירמיהו ט' כב'; תהילים צ' ה'-ו') 11 הֵן בַּעֲלוֹת הַשֶּׁמֶשׁ הַלּוֹהֶטֶת יִיבַשׁ הֶחָצִיר וְיִפֹּל צִיצוֹ וְיֹאבַד חֵן מַרְאֵהוּ;(ישעיהו מ' ו'-ז') כֵּן גַּם הֶעָשִׁיר יִבֹּל בַּהֲלִיכוֹתָיו.

עמידה בניסיון

12 אַשְׁרֵי הָאִישׁ הַמַּחֲזִיק מַעֲמָד בְּנִסָּיוֹן, כִּי לְאַחַר עָמְדוֹ בַּנִּסָּיוֹן יְקַבֵּל אֶת עֲטֶרֶת הַחַיִּים אֲשֶׁר הִבְטִיחַ הָאָדוֹן לְאוֹהֲבָיו. 13 מִי שֶׁמִּתְנַסֶּה אַל יֹאמַר, "אֱלֹהִים מְנַסֶּה אוֹתִי", שֶׁכֵּן אֱלֹהִים לֹא יְנֻסֶּה בָּרַע וְאֵין הוּא מְנַסֶּה אִישׁ. 14 אֶלָּא שֶׁכָּל אִישׁ מִתְנַסֶּה כַּאֲשֶׁר הוּא נִמְשָׁךְ וּמִתְפַּתֶּה בְּתַאֲוָתוֹ שֶׁלּוֹ. 15 אַחֲרֵי כֵן תַּהֲרֶה הַתַּאֲוָה וְתֵלֵד חֵטְא; וְהַחֵטְא כְּשֶׁיִּשְׁלַם יוֹלִיד מָוֶת.(בראשית ג' ו'; יהושוע ז' כא') 16 אַל נָא תִּטְעוּ, אַחַי אֲהוּבַי. 17 כָּל מַתָּנָה טוֹבָה וְכָל מַתָּנָה שְׁלֵמָה יוֹרֶדֶת מִמַּעַל מֵאֵת אֲבִי הָאוֹרוֹת אֲשֶׁר כָּל שִׁנּוּי וְכָל צֵל חִלּוּף אֵין בּוֹ.(מלאכי ג' ו') 18 הוּא כִּרְצוֹנוֹ הוֹלִיד אוֹתָנוּ בִּדְבַר הָאֱמֶת, לְמַעַן נִהְיֶה בִּכּוּרֵי יְצִירָיו.

לשמוע ולעשות

19 עַל כֵּן, אַחַי אֲהוּבַי, יְהֵא כָּל אִישׁ מָהִיר לִשְׁמֹעַ, בִּלְתִּי נֶחְפָּז לְדַבֵּר וְקָשֶׁה לִכְעֹס,(משלי י' יט') 20 שֶׁהֲרֵי כַּעַס אָדָם לֹא יִפְעַל צִדְקַת אֱלֹהִים.(משלי כט' כב') 21 לָכֵן הָסִירוּ מֵעֲלֵיכֶם כָּל טִנּוּף וְרֹב רֶשַׁע, וְקַבְּלוּ בַּעֲנָוָה אֶת הַדָּבָר הַנָּטוּעַ אֲשֶׁר יָכוֹל לְהוֹשִׁיעַ אֶת נַפְשׁוֹתֵיכֶם. 22 הֱיוּ עוֹשֵׂי הַדָּבָר וְלֹא רַק שׁוֹמְעִים, פֶּן תְּרַמּוּ אֶת עַצְמְכֶם; 23 כִּי מִי שֶׁשּׁוֹמֵעַ אֶת הַדָּבָר וְאֵינוֹ עוֹשֶׂה, כָּמוֹהוּ כְּאִישׁ הַמַּבִּיט עַל תֹּאַר פָּנָיו בַּמַּרְאָה; 24 הוּא הִתְבּוֹנֵן בְּעַצְמוֹ וְהָלַךְ לוֹ, וּמִיָּד שָׁכַח מַה צּוּרָתוֹ. 25 אֲבָל הַמַּשְׁקִיף בַּתּוֹרָה הַשְּׁלֵמָה, תּוֹרַת הַחֵרוּת, וְעוֹמֵד בָּהּ מִבְּלִי לִהְיוֹת שׁוֹמֵעַ-וְשׁוֹכֵחַ, אֶלָּא עוֹשֶׂה בְּפֹעַל -- אִישׁ זֶה מְבֹרָךְ יִהְיֶה בְּמַעֲשֵׂהוּ.(תהילים יט' יב') 26 מִי שֶׁסָּבוּר שֶׁהוּא עוֹבֵד אֱלֹהִים, וְאֵינֶנּוּ שָׂם רֶסֶן לִלְשׁוֹנוֹ כִּי אִם מַתְעֶה אֶת לִבּוֹ, עֲבוֹדָתוֹ אֵינָהּ אֶלָּא הֶבֶל.(תהילים לד' יד') 27 זֹאת הִיא עֲבוֹדַת אֱלֹהִים טְהוֹרָה וּתְמִימָה לִפְנֵי אֱלֹהִים אָבִינוּ: לִפְקֹד אֶת הַיְתוֹמִים וְהָאַלְמָנוֹת בְּצָרָתָם, וּלְהִשָּׁמֵר נָקִי מִטֻּמְאַת הָעוֹלָם.(שמות כב' כא'-כב')
