לפרק הקודם

הבשורה על-פי מרקוס
פרק טז

תחיית ישוע

(מתי כח' 8-1; לוקס כד' 12-1; יוחנן כ' 10-1)

1 כְּשֶׁעָבַר יוֹם הַשַּׁבָּת, מִרְיָם הַמַּגְדָּלִית וּמִרְיָם אֵם יַעֲקֹב וּשְׁלוֹמִית קָנוּ בְּשָׂמִים, כִּי הִתְכַּוְּנוּ לָבוֹא וְלִמְשֹׁחַ אוֹתוֹ. 2 בָּרִאשׁוֹן בַּשָּׁבוּעַ, בַּבֹּקֶר הַשְׁכֵּם, בָּאוּ אֶל הַקֶּבֶר עִם זְרִיחַת הַשֶּׁמֶשׁ. 3 אָמְרוּ זוֹ אֶל זוֹ: "מִי יָגֹל לָנוּ אֶת הָאֶבֶן מֵעַל פֶּתַח הַקֶּבֶר?" 4 כַּאֲשֶׁר הִבִּיטוּ רָאוּ כִּי הָאֶבֶן כְּבָר נָגֹלָּה, וְהִיא גְּדוֹלָה מְאֹד. 5 נִכְנְסוּ לְתוֹךְ הַקֶּבֶר וְרָאוּ בָּחוּר אֶחָד יוֹשֵׁב מִיָּמִין וְהוּא לָבוּשׁ גְּלִימָה לְבָנָה. הֵן נִדְהֲמוּ. 6 אָמַר לָהֶן: "אַל תִּבָּהַלְנָה. אֶת יֵשׁוּעַ מִנָּצְרַת אֲשֶׁר נִצְלַב אַתֶּן מְחַפְּשׂוֹת? הוּא קָם לִתְחִיָּה וְאֵינֶנּוּ פֹּה עוֹד. הִנֵּה הַמָּקוֹם שֶׁהִנִּיחוּהוּ בּוֹ. 7 וְעַתָּה לֵכְנָה וְהַגֵּדְנָה לְתַלְמִידָיו וּלְכֵיפָא, 'הוּא הוֹלֵךְ לִפְנֵיכֶם לַגָּלִיל. שָׁם תִּרְאוּ אוֹתוֹ כְּפִי שֶׁאָמַר לָכֶם.'" 8 הֵן יָצְאוּ וּבָרְחוּ מִן הַקֶּבֶר, כִּי רְעָדָה וְתַדְהֵמָה אָחֲזוּ בָּהֶן, וְלֹא סִפְּרוּ דָּבָר לְאִישׁ מִפְּנֵי שֶׁפָּחֲדוּ.

התגלותו ועלייתו למרום

[ 9 כַּאֲשֶׁר קָם בַּבֹּקֶר בָּרִאשׁוֹן בַּשָּׁבוּעַ, נִרְאָה בָּרִאשׁוֹנָה לְמִרְיָם הַמַּגְדָּלִית אֲשֶׁר מִמֶּנָּה גֵּרֵשׁ שִׁבְעָה שֵׁדִים. 10 הִיא הָלְכָה וְסִפְּרָה לָאֲנָשִׁים שֶׁהָיוּ אִתּוֹ לִפְנֵי כֵן וְהֵם מִתְאַבְּלִים וּבוֹכִים. 11 כַּאֲשֶׁר שָׁמְעוּ שֶׁהוּא חַי וְנִרְאָה אֵלֶיהָ לֹא הֶאֱמִינוּ. 12 אַחֲרֵי כֵן נִרְאָה בִּדְמוּת אַחֶרֶת לִשְׁנַיִם מֵהֶם בְּעֵת שֶׁהָלְכוּ בְּדַרְכָּם לְאַחַד הַכְּפָרִים. 13 הַלָּלוּ הָלְכוּ וְסִפְּרוּ לַאֲחֵרִים וְגַם לָהֶם לֹא הֶאֱמִינוּ. 14 אַחַר כָּךְ נִרְאָה אֶל הָאַחַד-עָשָׂר בִּהְיוֹתָם מְסֻבִּים לֶאֱכֹל. הוּא גָּעַר בָּהֶם עַל חֹסֶר אֱמוּנָתָם וְקֵהוּת לְבָבָם, מִפְּנֵי שֶׁלֹּא הֶאֱמִינוּ לְאֵלֶּה שֶׁרָאוּ אוֹתוֹ אַחֲרֵי קוּמוֹ לִתְחִיָּה. 15 אָמַר לָהֶם: "לְכוּ אֶל כָּל הָעוֹלָם וְהַכְרִיזוּ אֶת הַבְּשׂוֹרָה לְכָל הַבְּרִיאָה. 16 הַמַּאֲמִין וְנִטְבָּל יִוָּשַׁע, וּמִי שֶׁלֹּא יַאֲמִין יֶאְשַׁם. 17 וְאֵלֶּה הָאוֹתוֹת אֲשֶׁר יִלָּווּ אֶל הַמַּאֲמִינִים: יְגָרְשׁוּ שֵׁדִים בִּשְׁמִי; בִּלְשׁוֹנוֹת חֲדָשׁוֹת יְדַבְּרוּ; 18 יָרִימוּ נְחָשִׁים, וְאִם יִשְׁתּוּ רַעַל מֵמִית לֹא יִנָּזְקוּ; יְדֵיהֶם יִסְמְכוּ עַל חוֹלִים וְיִיטַב לָהֶם." 19 לְאַחַר שֶׁדִּבֵּר אֲלֵיהֶם נִשָּׂא הָאָדוֹן הַשָּׁמַיְמָה (מלכים ב', ב' יא') וְיָשַׁב לִימִין אֱלֹהִים.(תהילים קי' א') 20 הֵם יָצְאוּ וְהִכְרִיזוּ אֶת הַבְּשׂוֹרָה בְּכָל מָקוֹם, וְהָאָדוֹן פָּעַל עִמָּהֶם וְאִשֵּׁר אֶת הַדָּבָר בָּאוֹתוֹת שֶׁנִּלְווּ אֲלֵיהֶם.]
