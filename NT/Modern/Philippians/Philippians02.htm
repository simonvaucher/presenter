לפרק הקודם

אגרת שאול אל הפיליפים
פרק ב

התנהגות המשיח מופת למאמינים

1 לָכֵן אִם יֵשׁ אֵיזֶה עִדּוּד בַּמָּשִׁיחַ, אִם אֵיזוֹ נֶחָמָה שֶׁל אַהֲבָה, אִם אֵיזוֹ שֻׁתָּפוּת שֶׁל רוּחַ, אִם אֵילוּ רַחֲמִים וְחֶמְלָה -- 2 אֲזַי מַלְּאוּ נָא אֶת שִׂמְחָתִי בָּזֶה שֶׁתִּהְיוּ תְּמִימֵי דֵעִים, חֲדוּרֵי אַהֲבָה אַחַת, בַּעֲלֵי כַּוָּנָה אַחַת וּמַחֲשָׁבָה אַחַת 3 וְאַל תַּעֲשׂוּ דָּבָר מִתּוֹךְ תַּחֲרוּת, אַף לֹא מִתּוֹךְ כְּבוֹד שָׁוְא, אֶלָּא בִּנְמִיכוּת רוּחַ יַחְשֹׁב אִישׁ אִישׁ אֶת רֵעֵהוּ לְנִכְבָּד מִמֶּנּוּ. 4 כָּל אֶחָד אַל יִדְאַג רַק לְעִנְיָנָיו, אֶלָּא גַּם לְעִנְיָנָיו שֶׁל זוּלָתוֹ. 5 יְהֵא בָּכֶם הֲלָךְ רוּחַ זֶה אֲשֶׁר הָיָה בַּמָּשִׁיחַ יֵשׁוּעַ: 6 הוּא אֲשֶׁר הָיָה קַיָּם בִּדְמוּת אֱלֹהִים לֹא חָשַׁב לְשָׁלָל הֱיוֹת שָׁוֶה לֵאלֹהִים, 7 אֶלָּא הֵרִיק אֶת עַצְמוֹ, נָטַל דְּמוּת עֶבֶד (ישעיהו נג' ג') וְנִהְיָה כִּבְנֵי אָדָם; וְכַאֲשֶׁר הָיָה בְּצוּרָתוֹ כְּאָדָם,(ישעיהו מב' א') 8 הִשְׁפִּיל עַצְמוֹ וְצִיֵּת עַד מָוֶת, עַד מָוֶת בַּצְּלָב.(תהילים מ' ט') 9 עַל כֵּן הִגְבִּיהוֹ אֱלֹהִים מְאֹד וְנָתַן לוֹ אֶת הַשֵּׁם הַנַּעֲלֶה עַל כָּל שֵׁם, 10 לְמַעַן תִּכְרַע בְּשֵׁם יֵשׁוּעַ כָּל בֶּרֶךְ, בַּשָּׁמַיִם וּבָאָרֶץ וּמִתַּחַת לָאָרֶץ,(תהילים כב' ל'; ישעיהו מה' כג') 11 וְכָל לָשׁוֹן תּוֹדֶה כִּי יֵשׁוּעַ הַמָּשִׁיחַ הוּא הָאָדוֹן, לְתִפְאֶרֶת אֱלֹהִים הָאָב. 12 לָכֵן אֲהוּבַי, כְּשֵׁם שֶׁצִּיַּתֶּם תָּמִיד, לֹא רַק כְּמוֹ בְּעֵת נוֹכְחוּתִי, אֶלָּא עַל אַחַת כַּמָּה וְכַמָּה כָּעֵת, בְּהֵעָדְרִי -- הִשָּׁמְעוּ עַתָּה, וּבְיִרְאָה וּבִרְתֵת פַּעֲלוּ לְמִמּוּשׁ תְּשׁוּעַתְכֶם;(תהילים ב' יא') 13 וְאָמְנָם אֱלֹהִים הוּא הַפּוֹעֵל בָּכֶם גַּם שֶׁתִּרְצוּ וְגַם שֶׁתִּפְעֲלוּ כַּטּוֹב בְּעֵינָיו. 14 עֲשׂוּ כָּל דָּבָר בְּלִי לְהִתְלוֹנֵן וּבְלִי לְהִתְוַכֵּחַ, 15 לְמַעַן תִּהְיוּ נְקִיִּים מֵאָשָׁם וּטְהוֹרִים, בָּנִים לֵאלֹהִים אֵין דֹּפִי בָּהֶם, בְּתוֹךְ דּוֹר עִקֵּשׁ וּפְתַלְתֹּל אֲשֶׁר תּוֹפִיעוּ בּוֹ כִּמְאוֹרוֹת בָּעוֹלָם,(דניאל יב' ג') 16 בְּהַחֲזִיקְכֶם אֶת דְּבַר הַחַיִּים; זֹאת תִּהְיֶה לִי לִתְהִלָּה בְּיוֹם הַמָּשִׁיחַ, שֶׁלֹּא לַשָּׁוְא רַצְתִּי וְלֹא לָרִיק עָמַלְתִּי.(ישעיהו סה' כג') 17 אֲבָל גַּם אִם אֶשָּׁפֵךְ כְּנֶסֶךְ עַל קָרְבַּן עֲבוֹדַת הַקֹּדֶשׁ שֶׁל אֱמוּנַתְכֶם,(במדבר טו' ה'-י') שָׂמֵחַ אֲנִי וְשָׂשׂ עִם כֻּלְּכֶם. 18 כָּךְ שִׂמְחוּ גַּם אַתֶּם וְשִׂישׂוּ עִמִּי.

טימוטיאוס ואפפרודיטוס

19 תִּקְוָתִי בָּאָדוֹן יֵשׁוּעַ שֶׁאֶשְׁלַח אֲלֵיכֶם בִּמְהֵרָה אֶת טִימוֹתֵיאוֹס, לְמַעַן אֶתְעוֹדֵד גַּם אֲנִי בְּהִוָּדַע לִי מַה מַּצַּבְכֶם. 20 אֵין לִי אִישׁ תְּמִים דֵּעִים עִמִּי כָּמוֹהוּ, אֲשֶׁר בְּלֵב שָׁלֵם יִדְאַג לְעִנְיְנֵיכֶם. 21 הַכֹּל דּוֹאֲגִים לְעִנְיְנֵיהֶם, לֹא לַדְּבָרִים אֲשֶׁר לַמָּשִׁיחַ יֵשׁוּעַ. 22 אֲבָל יוֹדְעִים אַתֶּם אֶת אָפְיוֹ הַבָּחוּן; כְּמוֹ בֵּן הָעוֹבֵד עִם אָבִיו הוּא שֵׁרֵת עִמִּי בַּעֲבוֹדַת הַבְּשׂוֹרָה. 23 לָכֵן אוֹתוֹ מְקַוֶּה אֲנִי לִשְׁלֹחַ מִיָּד לְאַחַר שֶׁאֶרְאֶה מַה יַּעֲשׂוּ בִּי. 24 וּבוֹטֵחַ אֲנִי בָּאָדוֹן שֶׁגַּם אֲנִי אָבוֹא בִּמְהֵרָה. 25 אֲבָל חָשַׁבְתִּי לְנָחוּץ לִשְׁלֹחַ אֲלֵיכֶם אֶת הָאָח אֶפַּפְרוֹדִיטוֹס, שֻׁתָּפִי לַעֲבוֹדָה וְלַמַּעֲרָכָה וְגַם שְׁלִיחֲכֶם הַמְשָׁרֵת בְּמִלּוּי מַחְסוֹרִי; 26 שֶׁכֵּן הִתְגַּעְגֵּעַ אֶל כֻּלְּכֶם וְאַף נֶעֱצַב מְאֹד עַל שֶׁנּוֹדַע לָכֶם כִּי חָלָה. 27 אָמְנָם חָלָה וְנָטָה לָמוּת, אַךְ אֱלֹהִים רִחֵם עָלָיו וְלֹא רַק עָלָיו, אֶלָּא גַּם עָלַי, לְמַעַן לֹא יָבוֹא עָלַי יָגוֹן עַל יָגוֹן. 28 לָכֵן מִהַרְתִּי לְשָׁלְחוֹ אֲלֵיכֶם כְּדֵי שֶׁתִּשְׂמְחוּ בִּרְאוֹתְכֶם אוֹתוֹ שׁוּב, וְאָז יִמְעַט יְגוֹנִי. 29 עַל כֵּן קַבְּלוּהוּ בָּאָדוֹן בִּמְלֹא שִׂמְחָה וְהוֹקִירוּ אֲנָשִׁים כָּאֵלֶּה, 30 כִּי בִּגְלַל עֲבוֹדַת הַמָּשִׁיחַ הִגִּיעַ עַד סַף הַמָּוֶת, בְּסַכְּנוֹ אֶת נַפְשׁוֹ כְּדֵי לְמַלֵּא אֶת מַה שֶּׁנִּבְצַר מִכֶּם לַעֲשׂוֹת בְּשֵׁרוּתְכֶם לְמַעֲנִי.
