לפרק הקודם

אגרת אל העברים
פרק יא

אמונה

1 הָאֱמוּנָה הִיא בִּטָּחוֹן בְּמַמָּשׁוּת הַדְּבָרִים הַמְקֻוִּים, הוֹכָחַת דְּבָרִים שֶׁאֵינָם נִרְאִים. 2 בִּגְלָלָהּ הוּעַד עַל הַקַּדְמוֹנִים. 3 בֶּאֱמוּנָה נָבִין כִּי הָעוֹלָמִים הוּכְנוּ בִּדְבַר אֱלֹהִים, בְּאֹפֶן שֶׁהַנִּרְאֶה נִתְהַוָּה מִן הַבִּלְתִּי נִרְאֶה.(תהילים לג' ו',ט') 4 מִתּוֹךְ אֱמוּנָה הֵבִיא הֶבֶל לֵאלֹהִים קָרְבָּן טוֹב מִזֶּה שֶׁהֵבִיא קַיִן; בִּגְלָלָהּ הוּעַד עָלָיו כִּי צַדִּיק הוּא, שֶׁכֵּן אֱלֹהִים הֵעִיד עַל מַתְּנוֹתָיו, וּבִגְלָלָהּ הוּא מְדַבֵּר עֲדַיִן גַּם לְאַחַר שֶׁמֵּת.(בראשית ד' ד'-ה') 5 בִּזְכוּת אֱמוּנָה לֻקַּח חֲנוֹךְ בְּלִי לִרְאוֹת מָוֶת, "וְאֵינֶנּוּ, כִּי-לָקַח אֹתוֹ אֱלֹהִים".(בראשית ה' כד') וְהוּעַד עָלָיו לִפְנֵי הִלָּקְחוֹ כִּי "אֶת הָאֱלֹהִים הִתְהַלֵּךְ."(***) 6 וּבְלִי אֱמוּנָה אִי אֶפְשָׁר לִהְיוֹת רָצוּי לֵאלֹהִים, כִּי כָּל הַקָּרֵב אֶל אֱלֹהִים צָרִיךְ לְהַאֲמִין שֶׁהוּא קַיָּם וְהוּא נוֹתֵן גְּמוּל לְדוֹרְשָׁיו.(ישעיהו מ' י') 7 מִתּוֹךְ אֱמוּנָה בָּנָה נֹחַ תֵּבָה לְהַצָּלַת בֵּיתוֹ, בִּהְיוֹתוֹ יָרֵא אַחֲרֵי שֶׁהֻזְהַר עַל דְּבָרִים אֲשֶׁר עוֹד לֹא נִרְאוּ, וּבֶאֱמוּנָתוֹ הִרְשִׁיעַ אֶת הָעוֹלָם וְהָיָה לְיוֹרֵשׁ הַצְּדָקָה הַמְיֻסֶּדֶת עַל אֱמוּנָה.(בראשית ו' כב') 8 מִתּוֹךְ אֱמוּנָה צִיֵּת אַבְרָהָם בְּהִקָּרְאוֹ לָצֵאת אֶל מָקוֹם שֶׁהָיָה עָתִיד לְקַבֵּל לְנַחֲלָה, וְהוּא יָצָא מִבְּלִי לָדַעַת לְאָן יֵלֵךְ.(בראשית יב' א') 9 מִתּוֹךְ אֱמוּנָה הִתְגּוֹרֵר בְּאֶרֶץ הַהַבְטָחָה כְּגֵר בְּאֶרֶץ זָרָה, בְּשִׁבְתּוֹ בְּאֹהָלִים עִם יִצְחָק וְיַעֲקֹב אֲשֶׁר יָרְשׁוּ עִמּוֹ אֶת אוֹתָהּ הַהַבְטָחָה;(בראשית יז' ח') 10 כִּי חִכָּה לָעִיר שֶׁיֵּשׁ לָהּ יְסוֹדוֹת, שֶׁאַדְרִיכָלָהּ וּמְקִימָהּ הוּא הָאֱלֹהִים. 11 מִתּוֹךְ אֱמוּנָה קִבְּלָה גַּם שָׂרָה כֹּחַ לַהֲרוֹת אַף לְאַחַר שֶׁהִזְדַּקְּנָה, שֶׁכֵּן חָשְׁבָה אֶת הַמַּבְטִיחַ לְנֶאֱמָן.(בראשית כא' ב') 12 עַל כֵּן גַּם מֵאֶחָד, שֶׁהָיָה בִּבְחִינַת מֵת, יָצְאוּ כְּכוֹכְבֵי הַשָּׁמַיִם לָרֹב וְכַחוֹל עַל שְׂפַת הַיָּם אֲשֶׁר לֹא יִסָּפֵר.(בראשית כב' יז'; דברים א' י') 13 בְּעוֹדָם מַחֲזִיקִים בֶּאֱמוּנָה מֵתוּ כָּל אֵלֶּה בְּלִי שֶׁהִשִּׂיגוּ אֶת הַדְּבָרִים הַמֻּבְטָחִים, וְאוּלָם רָאוּ אוֹתָם מֵרָחוֹק וְקִדְּמוּ אוֹתָם בִּבְרָכָה כְּשֶׁהֵם מוֹדִים כִּי זָרִים וְגוֹלִים הֵם בָּאָרֶץ.(תהילים לט' יג') 14 הֵן הַמְדַבְּרִים כָּךְ מַצְהִירִים שֶׁהֵם מְבַקְשִׁים לָהֶם אֶרֶץ מוֹשָׁב. 15 וְאִלּוּ כִּוְּנוּ דַּעְתָּם עַל הָאָרֶץ הַהִיא אֲשֶׁר יָצְאוּ מִמֶּנָּה, הָיָה לָהֶם זְמַן לָשׁוּב אֵלֶיהָ. 16 אַךְ עַתָּה נִכְסָפִים הֵם לְטוֹבָה מִמֶּנָּה, לְאֶרֶץ מוֹשָׁב שְׁמֵימִית. לָכֵן לֹא בּוֹשׁ בָּהֶם הָאֱלֹהִים לְהִקָּרֵא אֱלֹהֵיהֶם, שֶׁהֲרֵי הֵכִין לָהֶם עִיר.(שמות ג' ו') 17 מִתּוֹךְ אֱמוּנָה הִקְרִיב אַבְרָהָם אֶת יִצְחָק בְּעֵת שֶׁנֻּסָּה; הוּא שֶׁקִּבֵּל אֶת הַהַבְטָחוֹת הִקְרִיב אֶת יְחִידוֹ, 18 הוּא שֶׁנֶּאֱמַר לוֹ, "כִּי בְיִצְחָק יִקָּרֵא לְךָ זָרַע",(בראשית כא' יב') 19 בְּחָשְׁבוֹ בְּלִבּוֹ כִּי הָאֱלֹהִים יָכוֹל לְהָקִים אֲפִלּוּ מֵעִם הַמֵּתִים; מִשָּׁם, בִּבְחִינַת מָשָׁל, אַף קִבֵּל אוֹתוֹ. 20 מִתּוֹךְ אֱמוּנָה בֵּרֵךְ יִצְחָק אֶת יַעֲקֹב וְאֶת עֵשָׂו בְּנוֹגֵעַ לִדְבָרִים עֲתִידִים לָבוֹא. 21 מִתּוֹךְ אֱמוּנָה בֵּרֵךְ יַעֲקֹב לִפְנֵי מוֹתוֹ אֶת שְׁנֵי בְּנֵי יוֹסֵף, וְקַד מֵעַל רֹאשׁ הַמַּטֶּה.(בראשית מז' לא') 22 מִתּוֹךְ אֱמוּנָה הִזְכִּיר יוֹסֵף לִפְנֵי מוֹתוֹ אֶת יְצִיאַת בְּנֵי יִשְׂרָאֵל וְצִוָּה עַל-אוֹדוֹת עַצְמוֹתָיו.(בראשית נ' כה') 23 כְּשֶׁנּוֹלַד מֹשֶׁה, מִתּוֹךְ אֱמוּנָה הִסְתִּירוּהוּ הוֹרָיו בְּמֶשֶׁךְ שְׁלוֹשָה חֳדָשִׁים; כִּי רָאוּ שֶׁנָּאֶה הַיֶּלֶד, וְלֹא פָּחֲדוּ מִגְּזֵרַת הַמֶּלֶך.(שמות ב' ב') 24 כַּאֲשֶׁר גָּדַל מֹשֶׁה, מִתּוֹךְ אֱמוּנָה סֵרֵב לְהִקָּרֵא בֵּן לְבַת פַּרְעֹה 25 וּבָחַר לְהִתְעַנּוֹת עִם עַם אֱלֹהִים מִלְּהִתְעַנֵּג לְשָׁעָה בְּתַעֲנוּגֵי הַחֵטְא. 26 הוּא חָשַׁב אֶת חֶרְפַּת הַמָּשִׁיחַ לְעֹשֶׁר גָּדוֹל מֵאוֹצְרוֹת מִצְרַיִם, כִּי נָשָׂא עֵינָיו אֶל הַגְּמוּל. 27 בֶּאֱמוּנָה עָזַב אֶת מִצְרַיִם וְלֹא פָּחַד מִזַּעַם הַמֶּלֶך, כִּי עָמַד בְּעֹז כְּרוֹאֶה אֶת מִי שֶׁאֵינֶנּוּ נִרְאֶה. 28 מִתּוֹךְ אֱמוּנָה עָשָׂה אֶת הַפֶּסַח וְאֶת נְתִינַת הַדָּם כְּדֵי שֶׁלֹּא יִגַּע בָּהֶם מַכֵּה הַבְּכוֹרִים.(שמות יב' יב'-יג') 29 בֶּאֱמוּנָה עָבְרוּ בְּיַם סוּף כְּמוֹ בַּיַּבָּשָׁה, דָּבָר שֶׁנִּסּוּ הַמִּצְרִים וְטָבְעוּ.(שמות יד' כב'; כח') 30 בִּגְלַל אֱמוּנָה נָפְלוּ חוֹמוֹת יְרִיחוֹ אַחֲרֵי שֶׁהִקִּיפוּ אוֹתָן שִׁבְעַת יָמִים. 31 בִּזְכוּת אֱמוּנָה לֹא נִסְפְּתָה רָחָב הַזּוֹנָה עִם הַסּוֹרְרִים, שֶׁכֵּן קִבְּלָה אֶת הַמְרַגְּלִים בְּשָׁלוֹם.(יהושוע ו' כה') 32 וּמָה עוֹד אֹמַר? הֵן תִּקְצַר הָעֵת מִלְּסַפֵּר עַל גִּדְעוֹן וּבָרָק וְשִׁמְשׁוֹן וְיִפְתָּח וְדָוִד וּשְׁמוּאֵל וְהַנְּבִיאִים,(שופטים ו' יא'; ד' ו'; יג' כד'; יא' א'; שמואל א', טז' יג'; א' כ') 33 אֲשֶׁר עַל-יְדֵי אֱמוּנָה הִכְנִיעוּ מַמְלָכוֹת, פָּעֲלוּ צֶדֶק, הִשִּׂיגוּ הַבְטָחוֹת, סָגְרוּ פִּי אֲרָיוֹת,(שמואל ב', ח' טו'; דניאל ו' כג') 34 כִּבּוּ לַהֲבוֹת אֵשׁ, נִמְלְטוּ מִפִּי חֶרֶב, הִתְחַזְּקוּ מֵחֻלְשָׁה, עָשׂוּ חַיִל בַּמִּלְחָמָה וְהֵבִיסוּ צִבְאוֹת זָרִים.(דניאל ג' כה') 35 נָשִׁים קִבְּלוּ אֶת מֵתֵיהֶן שֶׁקָּמוּ לִתְחִיָּה. אֲחֵרִים עֻנּוּ עַד מָוֶת וְלֹא הִסְכִּימוּ לְהִנָּצֵל, לְמַעַן יַשִּׂיגוּ תְּחִיָּה טוֹבָה יוֹתֵר.(מלכים א', יז' כב') 36 אֲחֵרִים הִתְנַסּוּ בְּחֶרְפַּת לַעַג וְהַלְקָאוֹת שׁוֹטִים וְגַם בִּכְבָלִים וְכֶלֶא;(תהילים קכג' ד'; ירמיהו לז' טו') 37 נִסְקְלוּ, נֻסּוּ בְּעִנּוּיִים, נֻסְּרוּ בְּמַשּׂוֹר, הוּמְתוּ בְּחֶרֶב, נָעוּ וְנָדוּ עֲטוּפֵי עוֹרוֹת כְּבָשִׂים וְעִזִּים, בְּסָבְלָם מַחְסוֹר, צָרוֹת וְהִתְעַלְּלוּת --(מלכים א', יט' י') 38 אָכֵן הָעוֹלָם לֹא הָיָה רָאוּי לַאֲנָשִׁים אֲשֶׁר כָּאֵלֶּה -- וְתוֹעִים הָיוּ בְּמִדְבָּרִיּוֹת, בְּהָרִים, בִּמְעָרוֹת וּבִנְקִיקֵי הָאָרֶץ.(מלכים א', יח' יג') 39 כָּל אֵלֶּה, אַף כִּי אֱמוּנָתָם הֵעִידָה עֲלֵיהֶם, לֹא רָאוּ בְּקִיּוּם הַהַבְטָחָה. 40 וְאוּלָם אֱלֹהִים הֵכִין לָנוּ דָּבָר טוֹב יוֹתֵר, כְּדֵי שֶׁלֹּא יוּבְאוּ לִידֵי שְׁלֵמוּת בִּלְעָדֵינוּ.
