לפרק הקודם

אגרת אל העברים
פרק יג

התנהגות רצויה בעיני אלהים

1 תְּהֵא אַהֲבַת אַחִים קַיֶּמֶת בֵּינֵיכֶם. 2 הַכְנָסַת אוֹרְחִים אַל תִּשְׁכְּחוּ, כִּי יֵשׁ אֲשֶׁר הִכְנִיסוּ אֶל בֵּיתָם מַלְאָכִים מִבְּלִי לָדַעַת זֹאת.(בראשית יח' ב') 3 זִכְרוּ אֶת הָאֲסוּרִים כְּאִלּוּ הֱיִיתֶם בַּמַּאֲסָר עִמָּהֶם; זִכְרוּ אֶת הַמְעֻנִּים, כְּשֵׁם שֶׁגַּם אַתֶּם חַיִּים בַּגּוּף. 4 חַיֵּי אִישׁוּת יְכֻבְּדוּ אֵצֶל הַכֹּל וִיצוּעֲכֶם אַל יְחֻלַּל. אֶת הַזּוֹנִים וְהַנּוֹאֲפִים יִשְׁפֹּט אֱלֹהִים. 5 רַחֲקוּ מֵאַהֲבַת כֶּסֶף וְשִׂמְחוּ בְּחֶלְקְכֶם, כִּי הוּא אָמַר, "לֹא אַרְפְּךָ וְלֹא אֶעֶזְבֶךָּ."(דברים לא' ו'; ח'; יהושע א' ה') 6 עַל כֵּן נִבְטַח וְנֹאמַר, "יהוה לִי בְּעֹזְרָי, לֹא אִירָא מַה-יַּעֲשֶׂה לִי אָדָם."(תהילים קיח' ו') 7 זִכְרוּ אֶת מַנְהִיגֵיכֶם אֲשֶׁר דִּבְּרוּ אֲלֵיכֶם אֶת דְּבַר אֱלֹהִים; הַבִּיטוּ אֶל אַחֲרִית דַּרְכָּם וּלְכוּ בְּעִקְּבוֹת אֱמוּנָתָם. 8 יֵשׁוּעַ הַמָּשִׁיחַ הוּא הוּא -- אֶתְמוֹל, הַיּוֹם וּלְעוֹלָמִים.(תהילים קב' כח') 9 אַל תִּסָּחֲפוּ עַל-יְדֵי תּוֹרוֹת שׁוֹנוֹת וְזָרוֹת; הֵן טוֹב לִסְעֹד אֶת הַלֵּב בְּחֶסֶד וְלֹא בְּדִבְרֵי מַאֲכָל אֲשֶׁר לֹא הוֹעִילוּ לְמִי שֶׁחַי עַל-פִּיהֶם. 10 יֵשׁ לָנוּ מִזְבֵּחַ אֲשֶׁר אֵין רְשׁוּת לִמְשָׁרְתֵי הַמִּשְׁכָּן לֶאֱכֹל מֵעָלָיו. 11 הַבְּהֵמוֹת אֲשֶׁר דָּמָן מוּבָא אֶל הַקֹּדֶשׁ בְּיַד הַכֹּהֵן הַגָּדוֹל לְכַפָּרַת הַחֵטְא, גּוֹפוֹתֵיהֶן נִשְׂרָפוֹת מִחוּץ לַמַּחֲנֶה.(ויקרא טז' כז') 2 לָכֵן גַּם יֵשׁוּעַ, כְּדֵי לְקַדֵּשׁ אֶת הָעָם בְּדָמוֹ, סָבַל מִחוּץ לַשַּׁעַר. 13 עַל כֵּן נֵצֵא נָא אֵלָיו אֶל מִחוּץ לַמַּחֲנֶה וְנִשָּׂא אֶת חֶרְפָּתוֹ;(***) 14 כִּי אֵין לָנוּ פֹּה עִיר קֶבַע, אֶלָּא אֶת זוֹ שֶׁלֶּעָתִיד לָבוֹא מְבַקְשִׁים אָנוּ. 15 לָכֵן בְּכָל עֵת נַקְרִיבָה בְּתִוּוּכוֹ זֶבַח תּוֹדָה לֵאלֹהִים, כְּלוֹמַר, פְּרִי שְׂפָתַיִם הַמּוֹדוֹת לִשְׁמוֹ.(תהילים כז' ו'; ישעיהו נז' יט') 16 וְאַל תִּשְׁכְּחוּ לִגְמֹל חֶסֶד וּלְשַׁתֵּף אֶת הַזּוּלַת בְּמַה שֶּׁיֵּשׁ לָכֶם, כִּי זְבָחִים כָּאֵלֶּה יֶעֶרְבוּ לֵאלֹהִים. 17 שִׁמְעוּ בְּקוֹל מַנְהִיגֵיכֶם וְהִכָּנְעוּ לָהֶם, כִּי מַשְׁגִּיחִים הֵם עַל נַפְשׁוֹתֵיכֶם כַּעֲתִידִים לָתֵת דִּין וְחֶשְׁבּוֹן; נַהֲגוּ כָּךְ, לְמַעַן יַעֲשׂוּ זֹאת בְּשִׂמְחָה וְלֹא בַּאֲנָחָה, שֶׁאִם לֹא כֵן אֵין זֶה לְתוֹעֶלֶת לָכֶם.(יחזקאל ג' יח';  כ') 18 הִתְפַּלְּלוּ בַּעֲדֵנוּ. בְּטוּחִים אָנוּ כִּי שָׁלֵם מַצְפּוּנֵנוּ; רְצוֹנֵנוּ לְהִתְנַהֵג בְּיֹשֶׁר בְּכָל דָּבָר. 19 מַפְצִיר אֲנִי בָּכֶם מְאֹד לַעֲשׂוֹת זֹאת, לְמַעַן יִנָּתֵן לִי לָשׁוּב אֲלֵיכֶם בִּמְהֵרָה.

אחרית דבר

20 וֵאלֹהֵי הַשָּׁלוֹם אֲשֶׁר בְּדַם בְּרִית עוֹלָם הֶעֱלָה מִן הַמֵּתִים אֶת רוֹעֵה הַצֹּאן הַגָּדוֹל, אֶת אֲדוֹנֵנוּ יֵשׁוּעַ,(יחזקאל לז' כו') 21 הוּא יַכְשִׁיר אֶתְכֶם לְכָל דָּבָר טוֹב כְּדֵי שֶׁתַּעֲשׂוּ אֶת רְצוֹנוֹ, בְּפָעֳלוֹ בָּכֶם אֶת הָרָצוּי לְפָנָיו בְּיַד יֵשׁוּעַ הַמָּשִׁיחַ, אֲשֶׁר לוֹ הַכָּבוֹד לְעוֹלְמֵי עוֹלָמִים. אָמֵן. 22 וַאֲנִי מְבַקֵּשׁ מִכֶּם, אַחַי, לְקַבֵּל בְּסַבְלָנוּת אֶת דְּבַר הַהַטָּפָה, שֶׁהֲרֵי בִּקְצָרָה כָּתַבְתִּי אֲלֵיכֶם. 23 דְּעוּ כִּי טִימוֹתֵיאוֹס אָחִינוּ שֻׁלַּח לַחָפְשִׁי. אִם יַקְדִּים לָבוֹא, אָבוֹא יַחַד אִתּוֹ לִרְאוֹת אֶתְכֶם. 24 דִּרְשׁוּ בִּשְׁלוֹם כָּל מַנְהִיגֵיכֶם וְכָל הַקְּדוֹשִׁים. בְּנֵי אֶרֶץ אִיטַלְיָה דּוֹרְשִׁים בִּשְׁלוֹמְכֶם. 25 הַחֶסֶד עִם כֻּלְּכֶם. אָמֵן.
