לפרק הקודם

אגרת שאול השניה אל התסלוניקים
פרק ב

התגלות איש הרשע

1 אֲשֶׁר לְבוֹא אֲדוֹנֵנוּ יֵשׁוּעַ הַמָּשִׁיחַ וַאֲסִיפָתֵנוּ אֵלָיו, אָנוּ מְבַקְשִׁים מִכֶּם, אַחַי: 2 אַל תְּמַהֲרוּ לְאַבֵּד אֶת עֶשׁתּוֹנוֹתֵיכֶם וְאַל תִּבָּהֲלוּ, לֹא בִּגְלַל אֵיזוֹ הִתְבַּטְּאוּת שֶׁל רוּחַ, לֹא בִּגְלַל אֵיזֶה דִּבּוּר, וְלֹא בִּגְלַל אֵיזוֹ אִגֶּרֶת אֲשֶׁר כִּבְיָכוֹל נִשְׁלְחָה מֵאִתָּנוּ -- כְּאִלּוּ הִגִּיעַ יוֹם יהוה. 3 אַל יַטְעֶה אֶתְכֶם אִישׁ בְּאֵיזֶה אֹפֶן שֶׁהוּא, שֶׁכֵּן לֹא יַגִּיעַ אִם לֹא תִּהְיֶה בָּרִאשׁוֹנָה הָעֲזִיבָה וְיִתְגַּלֶּה אִישׁ הָרֶשַׁע, בֶּן הָאֲבַדּוֹן, 4 הַמִּתְקוֹמֵם וּמְרוֹמֵם עַצְמוֹ עַל כָּל הַנִּקְרָא אֱלוֹהַּ אוֹ קֹדֶשׁ, עַד כִּי יֵשֵׁב בְּהֵיכַל הָאֱלֹהִים בְּהַצְהִירוֹ עַל עַצְמוֹ שֶׁהוּא אֱלֹהִים.(ישעיהו יד' יד'; יחזקאל כח' ב') 5 הַאִם אֵינְכֶם זוֹכְרִים כִּי עוֹד בִּהְיוֹתִי אֶצְלְכֶם אָמַרְתִּי לָכֶם אֶת הַדְּבָרִים הָאֵלֶּה? 6 אַתֶּם יוֹדְעִים מַה מְעַכֵּב אוֹתוֹ כָּעֵת כְּדֵי שֶׁיִּתְגַּלֶּה בְּעִתּוֹ. 7 הֵן סוֹד הָרֶשַׁע כְּבָר פּוֹעֵל; רַק שֶׁכָּעֵת הַמְעַכֵּב נִמְצָא עַד שֶׁיּוּצָא, 8 וְאָז יִתְגַּלֶּה הָרָשָׁע אֲשֶׁר הָאָדוֹן יָמִית אוֹתוֹ בְּרוּחַ פִּיו (איוב ד' ט'; ישעיהו יא' ד') וִיכַלֵּהוּ בְּהוֹפָעַת בּוֹאוֹ -- 9 אֶת הָרָשָׁע אֲשֶׁר בּוֹאוֹ הוּא בְּהֶתְאֵם לִפְעֻלַּת הַשָּׂטָן, מְלֻוֶּה בְּכָל גְּבוּרָה, בְּאוֹתוֹת וּבְמוֹפְתֵי שֶׁקֶר (שמות ז' כב') 10 וּבְכָל תַּרְמִית רֶשַׁע הַמְיֻעָדִים לִבְנֵי הָאֲבַדּוֹן, וְזֹאת מִפְּנֵי שֶׁלֹּא קִבְּלוּ אֶת אַהֲבַת הָאֱמֶת אֲשֶׁר יָכְלוּ לְהִוָּשַׁע בָּהּ; 11 עַל כֵּן אֱלֹהִים שׁוֹלֵחַ לָהֶם מַדּוּחִים לְהַטְעוֹתָם לְהַאֲמִין לַשֶּׁקֶר,(מלכים א', כב' כב') 12 לְמַעַן יִדּוֹנוּ כָּל אֲשֶׁר לֹא הֶאֱמִינוּ לָאֱמֶת אֶלָּא חָפְצוּ בָּעַוְלָה.

נחמת המאמינים

13 חַיָּבִים אָנוּ לְהוֹדוֹת לֵאלֹהִים עֲלֵיכֶם בְּכָל עֵת, אַחַי אֲהוּבֵי הָאָדוֹן, כִּי מֵרֵאשִׁית בָּחַר בָּכֶם אֱלֹהִים לִישׁוּעָה בְּקִדּוּשׁ עַל-יְדֵי הָרוּחַ וּבֶאֱמוּנָה בָּאֱמֶת; 14 וּבְאֶמְצָעוּת בְּשׂוֹרָתֵנוּ אַף קָרָא אֶתְכֶם לְנַחֲלָה בְּתִפְאַרְתּוֹ שֶׁל אֲדוֹנֵנוּ יֵשׁוּעַ הַמָּשִׁיחַ. 15 לְפִיכָךְ, אַחַי, עִמְדוּ עֲמִידָה אֵיתָנָה וְהַחֲזִיקוּ בַּמָּסוֹרוֹת שֶׁלֻּמַּדְתֶּם הֵן עַל-יְדֵי דְּבָרִים שֶׁאָמַרְנוּ וְהֵן עַל-יְדֵי אִגֶּרֶת שֶׁכָּתַבְנוּ. 16 וְהוּא, אֲדוֹנֵנוּ יֵשׁוּעַ הַמָּשִׁיחַ, וֵאלֹהִים אָבִינוּ, אֲשֶׁר אָהַב אוֹתָנוּ וּבְחַסְדּוֹ נָתַן לָנוּ נֶחָמַת עוֹלָם וְתִקְוָה טוֹבָה, 17 הוּא יְנַחֵם אֶת לְבַבְכֶם וִיכוֹנֵן אֶתְכֶם בְּכָל דִּבּוּר אוֹ מַעֲשֶׂה טוֹב.
