לפרק הקודם

אגרת שאול הראשונה אל טימותיאוס
פרק ב

הוראות בנוגע לתפילה ולהתנהגות

1 קֹדֶם כֹּל אֲנִי מְבַקֵּשׁ מִכֶּם לָשֵׂאת תְּחִנּוֹת וּתְפִלּוֹת וּבַקָּשׁוֹת וְהוֹדָיוֹת בְּעַד כָּל בְּנֵי אָדָם 2 בְּעַד מְלָכִים וְכָל רָאשֵׁי הַשִּׁלְטוֹן, לְמַעַן נִחְיֶה חַיֵּי שַׁלְוָה וְהַשְׁקֵט בִּמְלוֹא חֲסִידוּת וְדֶרֶךְ אֶרֶץ.(עזרא ו' י'; ירמיהו כט' ז') 3 טוֹב הַדָּבָר הַזֶּה וְרָצוּי בְּעֵינֵי אֱלֹהִים מוֹשִׁיעֵנוּ, 4 הֶחָפֵץ שֶׁכָּל בְּנֵי אָדָם יִוָּשְׁעוּ וְיַגִּיעוּ לְהַכָּרַת הָאֱמֶת.(יחזקאל יח' כג') 5 הֵן אֶחָד הָאֱלֹהִים, וְאֶחָד הַמְתַוֵּךְ בֵּין אֱלֹהִים לִבְנֵי אָדָם -- הָאָדָם הַמָּשִׁיחַ יֵשׁוּעַ,(ישעיהו מה' יד') 6 אֲשֶׁר נָתַן אֶת עַצְמוֹ כֹּפֶר בְּעַד הַכֹּל.(ישעיהו נג' יא') זֹאת הָעֵדוּת בְּעִתּוֹתֶיהָ, 7 אֲשֶׁר אֲנִי נִתְמַנֵּיתִי לָהּ לְכָרוֹז וּלְשָׁלִיחַ -- אֱמֶת אֲנִי מְדַבֵּר, אֵינֶנִּי מְשַׁקֵּר -- לְמוֹרֵה אֱמוּנָה וֶאֱמֶת לַגּוֹיִים. 8 רְצוֹנִי שֶׁהָאֲנָשִׁים יִתְפַּלְּלוּ בְּכָל מָקוֹם בִּנְשִׂיאַת יָדַיִם טְהוֹרוֹת, בְּלֹא כַּעַס וּבְלֹא מַחֲלֹקֶת.(תהילים קלד' ב') 9 כְּמוֹ כֵן תַּעֲטֶינָה הַנָּשִׁים תִּלְבֹּשֶׁת הוֹלֶמֶת, בִּצְנִיעוּת וּבְאִפּוּק; לֹא בּהִתְגַּנְדְּרוּת שֵׂעָר, לֹא בְּזָהָב וּפְנִינִים, וְלֹא בְּמַלְבּוּשִׁים יְקָרִים, 10 אֶלָּא בְּמַה שֶּׁיָּאֶה לְנָשִׁים הַמַּצְהִירוֹת עַל יִרְאַת שָׁמַיִם -- בְּמַעֲשִׂים טוֹבִים. 11 הָאִשָּׁה תִּלְמַד בְּדוּמִיָּה, בְּהַכְנָעָה שְׁלֵמָה. 12 אֵינֶנִּי מַרְשֶׁה לָאִשָּׁה לְלַמֵּד, אַף לֹא לְהִשְׂתָּרֵר עַל הָאִישׁ, אֶלָּא לְהִשָּׁאֵר בְּדוּמִיָּה; 13 כִּי אָדָם נוֹצַר רִאשׁוֹנָה וְאַחֲרֵי כֵן חַוָּה.(בראשית ב' כא'-כב') 14 אָדָם לֹא נִפְתָּה, אֶלָּא הָאִשָּׁה שָׁמְעָה לְקוֹל הַמְפַתֶּה וּבָאָה לִידֵי עֲבֵרָה.(בראשית ג' ו') 15 אֲבָל הִיא תִּוָּשַׁע בְּלִדְתָּהּ יְלָדִים, אִם תַּעֲמֹד בָּאֱמוּנָה, בָּאַהֲבָה, בַּקְּדֻשָּׁה וּבַצְּנִיעוּת.
